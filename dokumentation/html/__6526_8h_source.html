41 /* Define a structure with the 6526 register offsets.
42 ** NOTE: The timer registers are not declared as 16 bit registers, because
43 ** the order in which the two 8 bit halves are written is important, and
44 ** the compiler doesn't guarantee any order when writing 16 bit values.
45 */
46 struct __6526 {
47  unsigned char pra; /* Port register A */
48  unsigned char prb; /* Port register B */
49  unsigned char ddra; /* Data direction register A */
50  unsigned char ddrb; /* Data direction register B */
51  unsigned char ta_lo; /* Timer A, low byte */
52  unsigned char ta_hi; /* Timer A, high byte */
53  unsigned char tb_lo; /* Timer B, low byte */
54  unsigned char tb_hi; /* Timer B, high byte */
55  unsigned char tod_10; /* TOD, 1/10 sec. */
56  unsigned char tod_sec; /* TOD, seconds */
57  unsigned char tod_min; /* TOD, minutes */
58  unsigned char tod_hour; /* TOD, hours */
59  unsigned char sdr; /* Serial data register */
60  unsigned char icr; /* Interrupt control register */
61  unsigned char cra; /* Control register A */
62  unsigned char crb; /* Control register B */
63 };
