124 FilterModelConfig8580::FilterModelConfig8580() :
125  voice_voltage_range(0.4), // FIXME measure
126  voice_DC_voltage(4.80), // FIXME was 4.76
127  C(22e-9),
128  Vdd(9.09),
129  Vth(0.80),
130  Ut(26.0e-3),
131  k(1.3), // FIXME this is just a hack, k must be less than one, likely around 0.7
132  uCox(55e-6), // FIXME measure
133  kVddt(k * (Vdd - Vth)),
134  vmin(opamp_voltage[0].x),
135  vmax(kVddt < opamp_voltage[0].y ? opamp_voltage[0].y : kVddt),
136  denorm(vmax - vmin),
137  norm(1.0 / denorm),
138  N16(norm * ((1 << 16) - 1))
139 {
140  // Convert op-amp voltage transfer to 16 bit values.
141 
142  Spline::Point scaled_voltage[OPAMP_SIZE];
143 
144  for (unsigned int i = 0; i < OPAMP_SIZE; i++)
145  {
146  scaled_voltage[i].x = N16 * (opamp_voltage[i].x - opamp_voltage[i].y + denorm) / 2.;
147  scaled_voltage[i].y = N16 * (opamp_voltage[i].x - vmin);
148  }
149 
150  // Create lookup table mapping capacitor voltage to op-amp input voltage:
151 
152  Spline s(scaled_voltage, OPAMP_SIZE);
153 
154  for (int x = 0; x < (1 << 16); x++)
155  {
156  const Spline::Point out = s.evaluate(x);
157  double tmp = out.x;
158  if (tmp < 0.) tmp = 0.;
159  assert(tmp < 65535.5);
160  opamp_rev[x] = static_cast<unsigned short>(tmp + 0.5);
161  }
162 
163  // Create lookup tables for gains / summers.
164 
165  OpAmp opampModel(opamp_voltage, OPAMP_SIZE, kVddt);
166 
167  // The filter summer operates at n ~ 1, and has 5 fundamentally different
168  // input configurations (2 - 6 input "resistors").
169  //
170  // Note that all "on" transistors are modeled as one. This is not
171  // entirely accurate, since the input for each transistor is different,
172  // and transistors are not linear components. However modeling all
173  // transistors separately would be extremely costly.
174  for (int i = 0; i < 5; i++)
175  {
176  const int idiv = 2 + i; // 2 - 6 input "resistors".
177  const int size = idiv << 16;
178  const double n = idiv;
179  opampModel.reset();
180  summer[i] = new unsigned short[size];
181 
182  for (int vi = 0; vi < size; vi++)
183  {
184  const double vin = vmin + vi / N16 / idiv; /* vmin .. vmax */
185  const double tmp = (opampModel.solve(n, vin) - vmin) * N16;
186  assert(tmp > -0.5 && tmp < 65535.5);
187  summer[i][vi] = static_cast<unsigned short>(tmp + 0.5);
188  }
189  }
190 
191  // The audio mixer operates at n ~ 8/6, and has 8 fundamentally different
192  // input configurations (0 - 7 input "resistors").
193  //
194  // All "on", transistors are modeled as one - see comments above for
195  // the filter summer.
196  for (int i = 0; i < 8; i++)
197  {
198  const int idiv = (i == 0) ? 1 : i;
199  const int size = (i == 0) ? 1 : i << 16;
200  const double n = i * 8.0 / 6.0;
201  opampModel.reset();
202  mixer[i] = new unsigned short[size];
203 
204  for (int vi = 0; vi < size; vi++)
205  {
206  const double vin = vmin + vi / N16 / idiv; /* vmin .. vmax */
207  const double tmp = (opampModel.solve(n, vin) - vmin) * N16;
208  assert(tmp > -0.5 && tmp < 65535.5);
209  mixer[i][vi] = static_cast<unsigned short>(tmp + 0.5);
210  }
211  }
212 
213  // 4 bit "resistor" ladders in the audio output gain
214  // necessitate 16 gain tables.
215  // From die photographs of the volume "resistor" ladders
216  // it follows that gain ~ vol/8 (assuming ideal op-amps
217  for (int n8 = 0; n8 < 16; n8++)
218  {
219  const int size = 1 << 16;
220  const double n = n8 / 8.0;
221  opampModel.reset();
222  gain_vol[n8] = new unsigned short[size];
223 
224  for (int vi = 0; vi < size; vi++)
225  {
226  const double vin = vmin + vi / N16; /* vmin .. vmax */
227  const double tmp = (opampModel.solve(n, vin) - vmin) * N16;
228  assert(tmp > -0.5 && tmp < 65535.5);
229  gain_vol[n8][vi] = static_cast<unsigned short>(tmp + 0.5);
230  }
231  }
232 
233  // 4 bit "resistor" ladders in the bandpass resonance gain
234  // necessitate 16 gain tables.
235  // From die photographs of the bandpass and volume "resistor" ladders
236  // it follows that 1/Q ~ 2^((4 - res)/8) (assuming ideal
237  // op-amps and ideal "resistors").
238  for (int n8 = 0; n8 < 16; n8++)
239  {
240  const int size = 1 << 16;
241  opampModel.reset();
242  gain_res[n8] = new unsigned short[size];
243 
244  for (int vi = 0; vi < size; vi++)
245  {
246  const double vin = vmin + vi / N16; /* vmin .. vmax */
247  const double tmp = (opampModel.solve(resGain[n8], vin) - vmin) * N16;
248  assert(tmp > -0.5 && tmp < 65535.5);
249  gain_res[n8][vi] = static_cast<unsigned short>(tmp + 0.5);
250  }
251  }
252 }
