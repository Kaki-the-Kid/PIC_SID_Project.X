104 FilterModelConfig::FilterModelConfig() :
105  voice_voltage_range(1.5),
106  voice_DC_voltage(5.0),
107  C(470e-12),
108  Vdd(12.18),
109  Vth(1.31),
110  Ut(26.0e-3),
111  k(1.0),
112  uCox(20e-6),
113  WL_vcr(9.0 / 1.0),
114  WL_snake(1.0 / 115.0),
115  kVddt(k * (Vdd - Vth)),
116  dac_zero(6.65),
117  dac_scale(2.63),
118  vmin(opamp_voltage[0].x),
119  vmax(kVddt < opamp_voltage[0].y ? opamp_voltage[0].y : kVddt),
120  denorm(vmax - vmin),
121  norm(1.0 / denorm),
122  N16(norm * ((1 << 16) - 1)),
123  dac(DAC_BITS)
124 {
125  dac.kinkedDac(MOS6581);
126 
127  // Convert op-amp voltage transfer to 16 bit values.
128 
129  Spline::Point scaled_voltage[OPAMP_SIZE];
130 
131  for (unsigned int i = 0; i < OPAMP_SIZE; i++)
132  {
133  scaled_voltage[i].x = N16 * (opamp_voltage[i].x - opamp_voltage[i].y + denorm) / 2.;
134  scaled_voltage[i].y = N16 * (opamp_voltage[i].x - vmin);
135  }
136 
137  // Create lookup table mapping capacitor voltage to op-amp input voltage:
138 
139  Spline s(scaled_voltage, OPAMP_SIZE);
140 
141  for (int x = 0; x < (1 << 16); x++)
142  {
143  const Spline::Point out = s.evaluate(x);
144  double tmp = out.x;
145  if (tmp < 0.) tmp = 0.;
146  assert(tmp < 65535.5);
147  opamp_rev[x] = static_cast<unsigned short>(tmp + 0.5);
148  }
149 
150  // Create lookup tables for gains / summers.
151 
152  OpAmp opampModel(opamp_voltage, OPAMP_SIZE, kVddt);
153 
154  // The filter summer operates at n ~ 1, and has 5 fundamentally different
155  // input configurations (2 - 6 input "resistors").
156  //
157  // Note that all "on" transistors are modeled as one. This is not
158  // entirely accurate, since the input for each transistor is different,
159  // and transistors are not linear components. However modeling all
160  // transistors separately would be extremely costly.
161  for (int i = 0; i < 5; i++)
162  {
163  const int idiv = 2 + i; // 2 - 6 input "resistors".
164  const int size = idiv << 16;
165  const double n = idiv;
166  opampModel.reset();
167  summer[i] = new unsigned short[size];
168 
169  for (int vi = 0; vi < size; vi++)
170  {
171  const double vin = vmin + vi / N16 / idiv; /* vmin .. vmax */
172  const double tmp = (opampModel.solve(n, vin) - vmin) * N16;
173  assert(tmp > -0.5 && tmp < 65535.5);
174  summer[i][vi] = static_cast<unsigned short>(tmp + 0.5);
175  }
176  }
177 
178  // The audio mixer operates at n ~ 8/6, and has 8 fundamentally different
179  // input configurations (0 - 7 input "resistors").
180  //
181  // All "on", transistors are modeled as one - see comments above for
182  // the filter summer.
183  for (int i = 0; i < 8; i++)
184  {
185  const int idiv = (i == 0) ? 1 : i;
186  const int size = (i == 0) ? 1 : i << 16;
187  const double n = i * 8.0 / 6.0;
188  opampModel.reset();
189  mixer[i] = new unsigned short[size];
190 
191  for (int vi = 0; vi < size; vi++)
192  {
193  const double vin = vmin + vi / N16 / idiv; /* vmin .. vmax */
194  const double tmp = (opampModel.solve(n, vin) - vmin) * N16;
195  assert(tmp > -0.5 && tmp < 65535.5);
196  mixer[i][vi] = static_cast<unsigned short>(tmp + 0.5);
197  }
198  }
199 
200  // 4 bit "resistor" ladders in the bandpass resonance gain and the audio
201  // output gain necessitate 16 gain tables.
202  // From die photographs of the bandpass and volume "resistor" ladders
203  // it follows that gain ~ vol/8 and 1/Q ~ ~res/8 (assuming ideal
204  // op-amps and ideal "resistors").
205  for (int n8 = 0; n8 < 16; n8++)
206  {
207  const int size = 1 << 16;
208  const double n = n8 / 8.0;
209  opampModel.reset();
210  gain[n8] = new unsigned short[size];
211 
212  for (int vi = 0; vi < size; vi++)
213  {
214  const double vin = vmin + vi / N16; /* vmin .. vmax */
215  const double tmp = (opampModel.solve(n, vin) - vmin) * N16;
216  assert(tmp > -0.5 && tmp < 65535.5);
217  gain[n8][vi] = static_cast<unsigned short>(tmp + 0.5);
218  }
219  }
220 
221  const double nkVddt = N16 * kVddt;
222  const double nVmin = N16 * vmin;
223 
224  for (unsigned int i = 0; i < (1 << 16); i++)
225  {
226  // The table index is right-shifted 16 times in order to fit in
227  // 16 bits; the argument to sqrt is thus multiplied by (1 << 16).
228  //
229  // The returned value must be corrected for translation. Vg always
230  // takes part in a subtraction as follows:
231  //
232  // k*Vg - Vx = (k*Vg - t) - (Vx - t)
233  //
234  // I.e. k*Vg - t must be returned.
235  const double Vg = nkVddt - sqrt((double)(i << 16));
236  const double tmp = k * Vg - nVmin;
237  assert(tmp > -0.5 && tmp < 65535.5);
238  vcr_kVg[i] = static_cast<unsigned short>(tmp + 0.5);
239  }
240 
241  // EKV model:
242  //
243  // Ids = Is*(if - ir)
244  // Is = 2*u*Cox*Ut^2/k*W/L
245  // if = ln^2(1 + e^((k*(Vg - Vt) - Vs)/(2*Ut))
246  // ir = ln^2(1 + e^((k*(Vg - Vt) - Vd)/(2*Ut))
247 
248  const double kVt = k * Vth;
249  const double Is = 2. * uCox * Ut * Ut / k * WL_vcr;
250 
251  // Normalized current factor for 1 cycle at 1MHz.
252  const double N15 = norm * ((1 << 15) - 1);
253  const double n_Is = N15 * 1.0e-6 / C * Is;
254 
255  // kVg_Vx = k*Vg - Vx
256  // I.e. if k != 1.0, Vg must be scaled accordingly.
257  for (int kVg_Vx = 0; kVg_Vx < (1 << 16); kVg_Vx++)
258  {
259  const double log_term = log1p(exp((kVg_Vx / N16 - kVt) / (2. * Ut)));
260  // Scaled by m*2^15
261  const double tmp = n_Is * log_term * log_term;
262  assert(tmp > -0.5 && tmp < 65535.5);
263  vcr_n_Ids_term[kVg_Vx] = static_cast<unsigned short>(tmp + 0.5);
264  }
265 }

285 unsigned short* FilterModelConfig::getDAC(double adjustment) const
286 {
287  const double dac_zero = getDacZero(adjustment);
288 
289  unsigned short* f0_dac = new unsigned short[1 << DAC_BITS];
290 
291  for (unsigned int i = 0; i < (1 << DAC_BITS); i++)
292  {
293  const double fcd = dac.getOutput(i);
294  const double tmp = N16 * (dac_zero + fcd * dac_scale / (1 << DAC_BITS) - vmin);
295  assert(tmp > -0.5 && tmp < 65535.5);
296  f0_dac[i] = static_cast<unsigned short>(tmp + 0.5);
297  }
298 
299  return f0_dac;
300 }

302 std::unique_ptr<Integrator> FilterModelConfig::buildIntegrator()
303 {
304  // Vdd - Vth, normalized so that translated values can be subtracted:
305  // k*Vddt - x = (k*Vddt - t) - (x - t)
306  double tmp = N16 * (kVddt - vmin);
307  assert(tmp > -0.5 && tmp < 65535.5);
308  const unsigned short nkVddt = static_cast<unsigned short>(tmp + 0.5);
309 
310  // Normalized snake current factor, 1 cycle at 1MHz.
311  // Fit in 5 bits.
312  tmp = denorm * (1 << 13) * (uCox / (2. * k) * WL_snake * 1.0e-6 / C);
313  assert(tmp > -0.5 && tmp < 65535.5);
314  const unsigned short n_snake = static_cast<unsigned short>(tmp + 0.5);
315 
316  return std::unique_ptr<Integrator>(new Integrator(vcr_kVg, vcr_n_Ids_term, opamp_rev, nkVddt, n_snake));
317 }
