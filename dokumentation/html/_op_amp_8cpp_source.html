33 double OpAmp::solve(double n, double vi) const
34 {
35  // Start off with an estimate of x and a root bracket [ak, bk].
36  // f is decreasing, so that f(ak) > 0 and f(bk) < 0.
37  double ak = vmin;
38  double bk = vmax;
39 
40  const double a = n + 1.;
41  const double b = kVddt;
42  double b_vi = (b - vi);
43  if (b_vi < 0.) b_vi = 0.;
44  const double c = n * (b_vi * b_vi);
45 
46  for (;;)
47  {
48  const double xk = x;
49 
50  // Calculate f and df.
51 
52  Spline::Point out = opamp->evaluate(x);
53  const double vo = out.x;
54  const double dvo = out.y;
55 
56  double b_vx = b - x;
57  if (b_vx < 0.) b_vx = 0.;
58  double b_vo = b - vo;
59  if (b_vo < 0.) b_vo = 0.;
60 
61  // f = a*(b - vx)^2 - c - (b - vo)^2
62  const double f = a * (b_vx * b_vx) - c - (b_vo * b_vo);
63 
64  // df = 2*((b - vo)*dvo - a*(b - vx))
65  const double df = 2. * (b_vo * dvo - a * b_vx);
66 
67  // Newton-Raphson step: xk1 = xk - f(xk)/f'(xk)
68  x -= f / df;
69 
70  if (unlikely(fabs(x - xk) < EPSILON))
71  {
72  out = opamp->evaluate(x);
73  return out.x;
74  }
75 
76  // Narrow down root bracket.
77  (f < 0. ? bk : ak) = xk;
78 
79  if (unlikely(x <= ak) || unlikely(x >= bk))
80  {
81  // Bisection step (ala Dekker's method).
82  x = (ak + bk) * 0.5;
83  }
84  }
85 }
