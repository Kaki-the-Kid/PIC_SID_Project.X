127 unsigned int SidTuneBase::selectSong(unsigned int selectedSong)
128 {
129  // Check whether selected song is valid, use start song if not
130  const unsigned int song = (selectedSong == 0 || selectedSong > info->m_songs) ? info->m_startSong : selectedSong;
131 
132  // Copy any song-specific variable information
133  // such a speed/clock setting to the info structure.
134  info->m_currentSong = song;
135 
136  // Retrieve song speed definition.
137  switch (info->m_compatibility)
138  {
139  case SidTuneInfo::COMPATIBILITY_R64:
140  info->m_songSpeed = SidTuneInfo::SPEED_CIA_1A;
141  break;
142  case SidTuneInfo::COMPATIBILITY_PSID:
143  // This does not take into account the PlaySID bug upon evaluating the
144  // SPEED field. It would most likely break compatibility to lots of
145  // sidtunes, which have been converted from .SID format and vice versa.
146  // The .SID format does the bit-wise/song-wise evaluation of the SPEED
147  // value correctly, like it is described in the PlaySID documentation.
148  info->m_songSpeed = songSpeed[(song - 1) & 31];
149  break;
150  default:
151  info->m_songSpeed = songSpeed[song - 1];
152  break;
153  }
154 
155  info->m_clockSpeed = clockSpeed[song - 1];
156 
157  return info->m_currentSong;
158 }

162 void SidTuneBase::placeSidTuneInC64mem(sidmemory& mem)
163 {
164  // The Basic ROM sets these values on loading a file.
165  // Program end address
166  const uint_least16_t start = info->m_loadAddr;
167  const uint_least16_t end = start + info->m_c64dataLen;
168  mem.writeMemWord(0x2d, end); // Variables start
169  mem.writeMemWord(0x2f, end); // Arrays start
170  mem.writeMemWord(0x31, end); // Strings start
171  mem.writeMemWord(0xac, start);
172  mem.writeMemWord(0xae, end);
173 
174  // Copy data from cache to the correct destination.
175  mem.fillRam(info->m_loadAddr, &cache[fileOffset], info->m_c64dataLen);
176 }

178 void SidTuneBase::loadFile(const char* fileName, buffer_t& bufferRef)
179 {
180  std::ifstream inFile(fileName, std::ifstream::binary);
181 
182  if (!inFile.is_open())
183  {
184  throw loadError(ERR_CANT_OPEN_FILE);
185  }
186 
187  inFile.seekg(0, inFile.end);
188  const int fileLen = inFile.tellg();
189 
190  if (fileLen <= 0)
191  {
192  throw loadError(ERR_EMPTY);
193  }
194 
195  inFile.seekg(0, inFile.beg);
196 
197  buffer_t fileBuf;
198  fileBuf.reserve(fileLen);
199 
200  try
201  {
202  fileBuf.assign(std::istreambuf_iterator<char>(inFile), std::istreambuf_iterator<char>());
203  }
204  catch (std::exception &ex)
205  {
206  throw loadError(ex.what());
207  }
208 
209  if (inFile.bad())
210  {
211  throw loadError(ERR_CANT_LOAD_FILE);
212  }
213 
214  inFile.close();
215 
216  bufferRef.swap(fileBuf);
217 }

250 SidTuneBase* SidTuneBase::getFromBuffer(const uint_least8_t* const buffer, uint_least32_t bufferLen)
251 {
252  if (buffer == nullptr || bufferLen == 0)
253  {
254  throw loadError(ERR_EMPTY);
255  }
256 
257  if (bufferLen > MAX_FILELEN)
258  {
259  throw loadError(ERR_FILE_TOO_LONG);
260  }
261 
262  buffer_t buf1(buffer, buffer + bufferLen);
263 
264  // Here test for the possible single file formats.
265  std::unique_ptr<SidTuneBase> s(PSID::load(buf1));
266  if (s.get() == nullptr) s.reset(MUS::load(buf1, true));
267  if (s.get() == nullptr) throw loadError(ERR_UNRECOGNIZED_FORMAT);
268 
269  s->acceptSidTune("-", "-", buf1, false);
270  return s.release();
271 }

273 void SidTuneBase::acceptSidTune(const char* dataFileName, const char* infoFileName,
274  buffer_t& buf, bool isSlashedFileName)
275 {
276  // Make a copy of the data file name and path, if available.
277  if (dataFileName != nullptr)
278  {
279  const size_t fileNamePos = isSlashedFileName ?
280  SidTuneTools::slashedFileNameWithoutPath(dataFileName) :
281  SidTuneTools::fileNameWithoutPath(dataFileName);
282  info->m_path = std::string(dataFileName, fileNamePos);
283  info->m_dataFileName = std::string(dataFileName + fileNamePos);
284  }
285 
286  // Make a copy of the info file name, if available.
287  if (infoFileName != nullptr)
288  {
289  const size_t fileNamePos = isSlashedFileName ?
290  SidTuneTools::slashedFileNameWithoutPath(infoFileName) :
291  SidTuneTools::fileNameWithoutPath(infoFileName);
292  info->m_infoFileName = std::string(infoFileName + fileNamePos);
293  }
294 
295  // Fix bad sidtune set up.
296  if (info->m_songs > MAX_SONGS)
297  {
298  info->m_songs = MAX_SONGS;
299  }
300  else if (info->m_songs == 0)
301  {
302  info->m_songs = 1;
303  }
304 
305  if (info->m_startSong == 0
306  || info->m_startSong > info->m_songs)
307  {
308  info->m_startSong = 1;
309  }
310 
311  info->m_dataFileLen = buf.size();
312  info->m_c64dataLen = buf.size() - fileOffset;
313 
314  // Calculate any remaining addresses and then
315  // confirm all the file details are correct
316  resolveAddrs(&buf[fileOffset]);
317 
318  if (checkRelocInfo() == false)
319  {
320  throw loadError(ERR_BAD_RELOC);
321  }
322  if (checkCompatibility() == false)
323  {
324  throw loadError(ERR_BAD_ADDR);
325  }
326 
327  if (info->m_dataFileLen >= 2)
328  {
329  // We only detect an offset of two. Some position independent
330  // sidtunes contain a load address of 0xE000, but are loaded
331  // to 0x0FFE and call player at 0x1000.
332  info->m_fixLoad = (endian_little16(&buf[fileOffset])==(info->m_loadAddr+2));
333  }
334 
335  // Check the size of the data.
336  if (info->m_c64dataLen > MAX_MEMORY)
337  {
338  throw loadError(ERR_DATA_TOO_LONG);
339  }
340  else if (info->m_c64dataLen == 0)
341  {
342  throw loadError(ERR_EMPTY);
343  }
344 
345  cache.swap(buf);
346 }

359 SidTuneBase* SidTuneBase::getFromFiles(const char* fileName, const char **fileNameExtensions, bool separatorIsSlash)
360 {
361  buffer_t fileBuf1;
362 
363  loadFile(fileName, fileBuf1);
364 
365  // File loaded. Now check if it is in a valid single-file-format.
366  std::unique_ptr<SidTuneBase> s(PSID::load(fileBuf1));
367  if (s.get() == nullptr)
368  {
369  // Try some native C64 file formats
370  s.reset(MUS::load(fileBuf1, true));
371  if (s.get() != nullptr)
372  {
373  // Try to find second file.
374  std::string fileName2;
375  int n = 0;
376  while (fileNameExtensions[n] != nullptr)
377  {
378  createNewFileName(fileName2, fileName, fileNameExtensions[n]);
379  // 1st data file was loaded into "fileBuf1",
380  // so we load the 2nd one into "fileBuf2".
381  // Do not load the first file again if names are equal.
382  if (!stringutils::equal(fileName, fileName2.data(), fileName2.size()))
383  {
384  try
385  {
386  buffer_t fileBuf2;
387 
388  loadFile(fileName2.c_str(), fileBuf2);
389  // Check if tunes in wrong order and therefore swap them here
390  if (stringutils::equal(fileNameExtensions[n], ".mus"))
391  {
392  std::unique_ptr<SidTuneBase> s2(MUS::load(fileBuf2, fileBuf1, 0, true));
393  if (s2.get() != nullptr)
394  {
395  s2->acceptSidTune(fileName2.c_str(), fileName, fileBuf2, separatorIsSlash);
396  return s2.release();
397  }
398  }
399  else
400  {
401  std::unique_ptr<SidTuneBase> s2(MUS::load(fileBuf1, fileBuf2, 0, true));
402  if (s2.get() != nullptr)
403  {
404  s2->acceptSidTune(fileName, fileName2.c_str(), fileBuf1, separatorIsSlash);
405  return s2.release();
406  }
407  }
408  // The first tune loaded ok, so ignore errors on the
409  // second tune, may find an ok one later
410  }
411  catch (loadError const &) {}
412  }
413  n++;
414  }
415  }
416  }
417  if (s.get() == nullptr) s.reset(p00::load(fileName, fileBuf1));
418  if (s.get() == nullptr) s.reset(prg::load(fileName, fileBuf1));
419  if (s.get() == nullptr) throw loadError(ERR_UNRECOGNIZED_FORMAT);
420 
421  s->acceptSidTune(fileName, nullptr, fileBuf1, separatorIsSlash);
422  return s.release();
423 }

425 void SidTuneBase::convertOldStyleSpeedToTables(uint_least32_t speed, SidTuneInfo::clock_t clock)
426 {
427  // Create the speed/clock setting tables.
428  //
429  // This routine implements the PSIDv2NG compliant speed conversion. All tunes
430  // above 32 use the same song speed as tune 32
431  // NOTE: The cast here is used to avoid undefined references
432  // as the std::min function takes its parameters by reference
433  const unsigned int toDo = std::min(info->m_songs, static_cast<unsigned int>(MAX_SONGS));
434  for (unsigned int s = 0; s < toDo; s++)
435  {
436  clockSpeed[s] = clock;
437  songSpeed[s] = (speed & 1) ? SidTuneInfo::SPEED_CIA_1A : SidTuneInfo::SPEED_VBI;
438 
439  if (s < 31)
440  {
441  speed >>= 1;
442  }
443  }
444 }

446 bool SidTuneBase::checkRelocInfo()
447 {
448  // Fix relocation information
449  if (info->m_relocStartPage == 0xFF)
450  {
451  info->m_relocPages = 0;
452  return true;
453  }
454  else if (info->m_relocPages == 0)
455  {
456  info->m_relocStartPage = 0;
457  return true;
458  }
459 
460  // Calculate start/end page
461  const uint_least8_t startp = info->m_relocStartPage;
462  const uint_least8_t endp = (startp + info->m_relocPages - 1) & 0xff;
463  if (endp < startp)
464  {
465  return false;
466  }
467 
468  { // Check against load range
469  const uint_least8_t startlp = (uint_least8_t) (info->m_loadAddr >> 8);
470  const uint_least8_t endlp = startlp + (uint_least8_t) ((info->m_c64dataLen - 1) >> 8);
471 
472  if (((startp <= startlp) && (endp >= startlp))
473  || ((startp <= endlp) && (endp >= endlp)))
474  {
475  return false;
476  }
477  }
478 
479  // Check that the relocation information does not use the following
480  // memory areas: 0x0000-0x03FF, 0xA000-0xBFFF and 0xD000-0xFFFF
481  if ((startp < 0x04)
482  || ((0xa0 <= startp) && (startp <= 0xbf))
483  || (startp >= 0xd0)
484  || ((0xa0 <= endp) && (endp <= 0xbf))
485  || (endp >= 0xd0))
486  {
487  return false;
488  }
489 
490  return true;
491 }

493 void SidTuneBase::resolveAddrs(const uint_least8_t *c64data)
494 {
495  // Originally used as a first attempt at an RSID
496  // style format. Now reserved for future use
497  if (info->m_playAddr == 0xffff)
498  {
499  info->m_playAddr = 0;
500  }
501 
502  // loadAddr = 0 means, the address is stored in front of the C64 data.
503  if (info->m_loadAddr == 0)
504  {
505  if (info->m_c64dataLen < 2)
506  {
507  throw loadError(ERR_CORRUPT);
508  }
509 
510  info->m_loadAddr = endian_16(*(c64data+1), *c64data);
511  fileOffset += 2;
512  info->m_c64dataLen -= 2;
513  }
514 
515  if (info->m_compatibility == SidTuneInfo::COMPATIBILITY_BASIC)
516  {
517  if (info->m_initAddr != 0)
518  {
519  throw loadError(ERR_BAD_ADDR);
520  }
521  }
522  else if (info->m_initAddr == 0)
523  {
524  info->m_initAddr = info->m_loadAddr;
525  }
526 }

528 bool SidTuneBase::checkCompatibility()
529 {
530  if (info->m_compatibility == SidTuneInfo::COMPATIBILITY_R64)
531  {
532  // Check valid init address
533  switch (info->m_initAddr >> 12)
534  {
535  case 0x0A:
536  case 0x0B:
537  case 0x0D:
538  case 0x0E:
539  case 0x0F:
540  return false;
541  default:
542  if ((info->m_initAddr < info->m_loadAddr)
543  || (info->m_initAddr > (info->m_loadAddr + info->m_c64dataLen - 1)))
544  {
545  return false;
546  }
547  }
548 
549  // Check tune is loadable on a real C64
550  if (info->m_loadAddr < SIDTUNE_R64_MIN_LOAD_ADDR)
551  {
552  return false;
553  }
554  }
555 
556  return true;
557 }

559 std::string SidTuneBase::petsciiToAscii(SmartPtr_sidtt<const uint8_t>& spPet)
560 {
561  std::string buffer;
562 
563  do
564  {
565  const uint8_t petsciiChar = *spPet;
566  spPet++;
567 
568  if ((petsciiChar == 0x00) || (petsciiChar == 0x0d))
569  break;
570 
571  // If character is 0x9d (left arrow key) then move back.
572  if ((petsciiChar == 0x9d) && (!buffer.empty()))
573  {
574  buffer.resize(buffer.size() - 1);
575  }
576  else
577  {
578  // ASCII CHR$ conversion
579  const char asciiChar = CHR_tab[petsciiChar];
580  if ((asciiChar >= 0x20) && (buffer.length() <= 31))
581  buffer.push_back(asciiChar);
582  }
583  }
584  while (!spPet.fail());
585 
586  return buffer;
587 }
