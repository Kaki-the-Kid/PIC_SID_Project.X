59 /*
60  * This is what happens when the lfsr is clocked:
61  *
62  * cycle 0: bit 19 of the accumulator goes from low to high, the noise register acts normally,
63  * the output may overwrite a bit;
64  *
65  * cycle 1: first phase of the shift, the bits are interconnected and the output of each bit
66  * is latched into the following. The output may overwrite the latched value.
67  *
68  * cycle 2: second phase of the shift, the latched value becomes active in the first
69  * half of the clock and from the second half the register returns to normal operation.
70  *
71  * When the test or reset lines are active the first phase is executed at every cyle
72  * until the signal is released triggering the second phase.
73  */
74 void WaveformGenerator::clock_shift_register(unsigned int bit0)
75 {
76  shift_register = (shift_register >> 1) | bit0;
77 
78  // New noise waveform output.
79  set_noise_output();
80 }

82 unsigned int WaveformGenerator::get_noise_writeback()
83 {
84  return
85  ~(
86  (1 << 2) | // Bit 20
87  (1 << 4) | // Bit 18
88  (1 << 8) | // Bit 14
89  (1 << 11) | // Bit 11
90  (1 << 13) | // Bit 9
91  (1 << 17) | // Bit 5
92  (1 << 20) | // Bit 2
93  (1 << 22) // Bit 0
94  ) |
95  ((waveform_output & (1 << 11)) >> 9) | // Bit 11 -> bit 20
96  ((waveform_output & (1 << 10)) >> 6) | // Bit 10 -> bit 18
97  ((waveform_output & (1 << 9)) >> 1) | // Bit 9 -> bit 14
98  ((waveform_output & (1 << 8)) << 3) | // Bit 8 -> bit 11
99  ((waveform_output & (1 << 7)) << 6) | // Bit 7 -> bit 9
100  ((waveform_output & (1 << 6)) << 11) | // Bit 6 -> bit 5
101  ((waveform_output & (1 << 5)) << 15) | // Bit 5 -> bit 2
102  ((waveform_output & (1 << 4)) << 18); // Bit 4 -> bit 0
103 }

105 void WaveformGenerator::write_shift_register()
106 {
107  if (unlikely(waveform > 0x8) && likely(!test) && likely(shift_pipeline != 1))
108  {
109  // Write changes to the shift register output caused by combined waveforms
110  // back into the shift register. This happens only when the register is clocked
111  // (see $D1+$81_wave_test [1]) or when the test bit is set.
112  // A bit once set to zero cannot be changed, hence the and'ing.
113  //
114  // [1] ftp://ftp.untergrund.net/users/nata/sid_test/$D1+$81_wave_test.7z
115  //
116  // FIXME: Write test program to check the effect of 1 bits and whether
117  // neighboring bits are affected.
118 
119  shift_register &= get_noise_writeback();
120 
121  noise_output &= waveform_output;
122  no_noise_or_noise_output = no_noise | noise_output;
123  }
124 }

132 void WaveformGenerator::set_noise_output()
133 {
134  noise_output =
135  ((shift_register & (1 << 2)) << 9) | // Bit 20 -> bit 11
136  ((shift_register & (1 << 4)) << 6) | // Bit 18 -> bit 10
137  ((shift_register & (1 << 8)) << 1) | // Bit 14 -> bit 9
138  ((shift_register & (1 << 11)) >> 3) | // Bit 11 -> bit 8
139  ((shift_register & (1 << 13)) >> 6) | // Bit 9 -> bit 7
140  ((shift_register & (1 << 17)) >> 11) | // Bit 5 -> bit 6
141  ((shift_register & (1 << 20)) >> 15) | // Bit 2 -> bit 5
142  ((shift_register & (1 << 22)) >> 18); // Bit 0 -> bit 4
143 
144  no_noise_or_noise_output = no_noise | noise_output;
145 }

153 {
154  is6581 = chipModel == MOS6581;

156  Dac dacBuilder(DAC_BITS);
157  dacBuilder.kinkedDac(chipModel);

186  // This need more investigation

198 void WaveformGenerator::writeCONTROL_REG(unsigned char control)
199 {
200  const unsigned int waveform_prev = waveform;
201  const bool test_prev = test;
202 
203  waveform = (control >> 4) & 0x0f;
204  test = (control & 0x08) != 0;
205  sync = (control & 0x02) != 0;
206 
207  // Substitution of accumulator MSB when sawtooth = 0, ring_mod = 1.
208  ring_msb_mask = ((~control >> 5) & (control >> 2) & 0x1) << 23;
209 
210  if (waveform != waveform_prev)
211  {
212  // Set up waveform table.
213  wave = (*model_wave)[waveform & 0x7];
214 
215  // no_noise and no_pulse are used in set_waveform_output() as bitmasks to
216  // only let the noise or pulse influence the output when the noise or pulse
217  // waveforms are selected.
218  no_noise = (waveform & 0x8) != 0 ? 0x000 : 0xfff;
219  no_noise_or_noise_output = no_noise | noise_output;
220  no_pulse = (waveform & 0x4) != 0 ? 0x000 : 0xfff;
221 
222  if (waveform == 0)
223  {
224  // Change to floating DAC input.
225  // Reset fading time for floating DAC input.
226  floating_output_ttl = is6581 ? FLOATING_OUTPUT_TTL_6581 : FLOATING_OUTPUT_TTL_8580;
227  }
228  }
229 
230  if (test != test_prev)
231  {
232  if (test)
233  {
234  // Reset accumulator.
235  accumulator = 0;
236 
237  // Flush shift pipeline.
238  shift_pipeline = 0;
239 
240  // Set reset time for shift register.
241  shift_register_reset = model_shift_register_reset;
242 
243  // New noise waveform output.
244  set_noise_output();
245  }
246  else
247  {
248  // When the test bit is falling, the second phase of the shift is
249  // completed by enabling SRAM write.
250 
251  // During first phase of the shift the bits are interconnected
252  // and the output of each bit is latched into the following.
253  // The output may overwrite the latched value.
254  if (do_pre_writeback(waveform_prev, waveform, is6581)) {
255  shift_register &= get_noise_writeback();
256  }
257 
258  // bit0 = (bit22 | test) ^ bit17 = 1 ^ bit17 = ~bit17
259  clock_shift_register((~shift_register << 17) & (1 << 22));
260  }
261  }
262 }
