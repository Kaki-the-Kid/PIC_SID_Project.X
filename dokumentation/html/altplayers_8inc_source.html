194 ;Just-intonation frequency-table (in key of C) for PAL machines (based on integer-based ratios for pure intervals)
195 ;---------------------------------------------------------------
196 ; (reference document: http://www.phy.mtu.edu/~suits/scales.html )
197 JUST_C1_FREQ= 16.351875 ;C1 note's frequency (Hz) - A4 note = 436.05Hz this way
198 JUST_RATIOS = [1.00, 25.0/24.0, 9.0/8.0, 6.0/5.0, 5.0/4.0, 4.0/3.0, 45.0/32.0, 3.0/2.0, 8.0/5.0, 5.0/3.0, 9.0/5.0, 15.0/8.0]
199 ;DISP_NOTES = ["C","C#","D","D#","E","F","F#","G","G#","A","A#","H"]

201 FreqTableJustPalHi
202 .for i=0, i<8, i=i+1 ;i counts octaves
203  .byte >( (256**3/PAL_CLOCKFREQ) * JUST_C1_FREQ*JUST_RATIOS * 2**i )
204  ;.for j=0, j<12, j=j+1 ;j counts notes
205  ;.warn DISP_NOTES[j],"-",i+1,": ", JUST_C1_FREQ*JUST_RATIOS[j] * 2**i , " Hz"
206  ;.next
207 .next
