5  *
6  * BASS_VST allows the usage of VST plugins in BASS. BASS_VST was
7  * written to work with Silverjuke(R). Any other usage is for your own risk -
8  * but you're welcome and I'm sure it will work :-)
9  *
10  * If you use BASS_VST in your projects, you must not charge anything for this
11  * feature. I would also be happy to hear from you if you use BASS_VST. A
12  * note like "BASS_VST by Bjoern Petersen @ www.silverjuke.net" in the about
13  * box would be nice, however, I'm not angry if you decide not to do so.
14  *
15  * BASS_VST is a VST host compatible up to the VST 2.4 implementation and
16  * (hopefully) implements all needed features - incl. the new double precision
17  * processing.
18  *
19  * While the normal usage is very easy (only one function -
20  * BASS_VST_ChannelSetDSP() - is really needed), BASS_VST also allows advanced
21  * usage of the VST features - incl. embedding an editor or subclassing
22  * the whole library. The latter may require the VST SDK which can be
23  * obtained for free from Steinberg. Moreover, BASS_VST also supports
24  * VST instruments (VSTi plugins).

164 /* BASS_VST_ChannelSetDSP() sets any VST effect plugin (defined by a DLL file
165  * name) to any channel handle. VST instrument plugins cannot be used with
166  * this function. Flags:
167  *
168  * BASS_UNICODE Treat the dllFile pointer as UNICODE instead of ANSI
169  * (0x80000000)
170  *
171  * BASS_VST_KEEP_CHANS By default, mono effects assigned to stereo channels
172  * are mixed down before processing and converted back
173  * to stereo afterwards. Set this flag to avoid this
174  * behaviour in which case only the first channel is
175  * affected by processing (0x00000001)
176  *
177  * The priority parameter has the same meaning as for BASS_ChannelSetDSP() -
178  * DSPs with higher priority are called before those with lower.
179  *
180  * On success, the function returns the new vstHandle that must be given to
181  * the other functions. For errors, 0 is returned and BASS_ErrorGetCode()
182  * will specify the reason.
183  *
184  * For testing if a DLL is a valid VST effect, you can set chHandle to 0 -
185  * however, do not forget to call BASS_VST_ChannelRemoveDSP() even in this
186  * case.
187  *
188  * You may safely assign the same DLL to different channels at the same time -
189  * the library makes sure, every channel is processed indepeningly. But take
190  * care to use the correct vstHandles in this case.
191  *
192  * Finally, you can use any number of VST effects on a channel.
193  */
194 BASS_VSTSCOPE DWORD BASS_VSTDEF(BASS_VST_ChannelSetDSP)
195  (DWORD chHandle, const void* dllFile, DWORD flags, int priority);
196 
197 #define BASS_VST_KEEP_CHANS 0x00000001 /* flag that may be used for BASS_VST_ChannelSetDSP(), see the comments above */

437 /* With BASS_VST_SetBypass() you can bypass the VST plugin processing
438  * (state=TRUE) or switch back to normal processing (state=FALSE). By default,
439  * bypassing is off and the VST plugin will be processed normally.
440  * BASS_VST_GetBypass() returns the current state.
441  *
442  * Note, that the bypassing is completely done by BASS_VST, we're not using
443  * the so-called "soft" bypass that is implemented by some VST plugins. This is
444  * for the following reasons:
445  *
446  * - Soft-bypassing is not supported by all VST plugins
447  * - The state of soft-bypassing cannot be queried safely
448  * - Soft-bypassing would not be a real bypass as some channel transformations
449  * may still be needed
450  * - Performance reasons - soft-bypassing would require still most of the
451  * needed BASS_VST transformations
452  * - Finally, I do not see any advantages of the soft-bypassing
453  *
454  * If you really need the soft bypassing, you can do the following using the
455  * VST SDK:
456  *
457  * #include <aeffectx.h> // you can get this file from Steinberg
458  *
459  * BASS_VST_INFO info;
460  * BASS_VST_GetInfo(vstHandle, &info);
461  *
462  * if( info.aeffect->dispatcher(info.aeffect, effCanDo, 0, 0, "bypass", 0.0) )
463  * {
464  * info.aeffect->dispatcher(info.aeffect, effSetBypass, 0, (BOOL)softBypassState, NULL, 0.0);
465  * }
466  */
467 BASS_VSTSCOPE BOOL BASS_VSTDEF(BASS_VST_SetBypass)

476 /* BASS_VST_GetInfo() writes some information about a vstHandle to a
477  * BASS_VST_INFO structure.
478  *
479  * Some words to the number of input/output channels:
480  *
481  * VST plugins that have no input channels (so called "instruments") are not
482  * loaded by BASS_VST. You can assume chansIn and chansOut to be at least 1.
483  *
484  * Multi-channel streams should work correctly, if supported by a VST plugin.
485  * If not, only the first chansIn channels are processed by the plugin, the
486  * other ones stay unaffected. The opposite, eg. assigning multi-channel
487  * VST plugins to stereo channels, should be no problem at all.
488  *
489  * If mono plugins are assigned to stereo channels, the result will be mono,
490  * expanded to both channels. This behaviour can be switched of using the
491  * BASS_VST_KEEP_CHANS in BASS_VST_ChannelSetDSP().
492  */
493 typedef struct
494 {
495  DWORD channelHandle; /* the channelHandle as given to BASS_VST_ChannelSetDSP() or returned by BASS_VST_ChannelCreate; 0 if no channel was assigned to the VST plugin */
496  DWORD uniqueID; /* a unique ID for the VST plugin (the IDs are registered at Steinberg) */
497  char effectName[80]; /* the plugin's name */
498  DWORD effectVersion; /* the plugin's version */
499  DWORD effectVstVersion; /* the VST version, the plugin was written for */
500  DWORD hostVstVersion; /* the VST version supported by BASS_VST, currently 2.4 */
501  char productName[80]; /* the product name, may be empty */
502  char vendorName[80]; /* the vendor name, may be empty */
503  DWORD vendorVersion; /* vendor-specific version number */
504  DWORD chansIn; /* max. number of possible input channels */
505  DWORD chansOut; /* max. number of possible output channels */
506  DWORD initialDelay; /* for algorithms which need input in the first place, in samples */
507  DWORD hasEditor; /* can the BASS_VST_EmbedEditor() function be called? */
508  DWORD editorWidth; /* initial/current width of the editor, also note BASS_VST_EDITOR_RESIZED; if the editor is not yet opened, this value may be 0 for some (very few) plugins! */
509  DWORD editorHeight; /* same for the height */
510 #ifdef __aeffect__
511  AEffect* aeffect; /* the underlying AEffect object (see aeffectx.h in the VST SDK) */
512 #else
513  void* aeffect;
514 #endif
515  DWORD isInstrument; /* 1=the VST plugin is an instrument, 0=the VST plugin is an effect */
516  DWORD dspHandle; /* the internal DSP handle */
517 } BASS_VST_INFO;

564 /* With BASS_VST_SetCallback() you can assign a callback function of the type
565  * VSTPROC* to a vstHandle. The callback function is called with the BASS_VST_*
566  * actions defined below then. Unless defined otherwise, the callback function
567  * should always return 0. The "user" parameter given to BASS_VST_SetCallback()
568  * is just forwarded to the callback.
569  *
570  * Every vstHandle can have only one callback function; subsequent calls to
571  * BASS_VST_SetCallback() for the same vstHandle will just change the callback
572  * function. You can remove a callback function from a vstHandle with
573  * BASS_VST_SetCallback(vstHandle, NULL, 0); however, this is not needed from
574  * the view of BASS_VST.
575  */
576 typedef DWORD (CALLBACK VSTPROC)(DWORD vstHandle, DWORD action, DWORD param1, DWORD param2, void* user);

587 /* Subclassing BASS_VST: By using BASS_VST_SetCallback() and catching the
588  * BASS_VST_AUDIO_MASTER events, you can subclass the communication between the
589  * plugin and BASS_VST. You'll find all needed parameters in a
590  * BASS_VST_AUDIO_MASTER_PARAM structure given in param1 to the callback
591  * function. To avoid the BASS_VST default processing on some opcodes, just
592  * set the "doDefault" member to 0 - in this case BASS_VST just forwards your
593  * return value to the plugin and does nothing else. Initially, "doDefault"
594  * is always set to 1 and BASS_VST will do what's needed. For a documentation
595  * about the possible requests, see the VST SDK from Steinberg.
596  *
597  * Example for replacing the BASS_VST file selector:

599  * #include <aeffectx.h> // you can get this file from Steinberg
600  *
601  * DWORD openMyFileSelector(VstFileSelect* fileSelect)
602  * {
603  * // do what to do here ...
604  * }
605  *
606  * DWORD myCallback(DWORD vstHandle, DWORD action, DWORD param1, DWORD param2, void* user)
607  * {
608  * if( action == BASS_VST_AUDIO_MASTER )
609  * {
610  * BASS_VST_AUDIO_MASTER_PARAM* audioMaster = (BASS_VST_AUDIO_MASTER_PARAM*)param1;
611  * if( audioMaster->opcode == audioMasterOpenFileSelector )
612  * {
613  * openMyFileSelector((VstFileSelect*)audioMaster->ptr);
614  * audioMaster->doDefault = 0;
615  * return 1;
616  * }
617  * }
618  * return 0;
619  * }
620  *
621  * BASS_VST_SetCallback(vstHandle, myCallback, 0);
