32 {
33  // check if any parameters have beed changed ...
34  assert( _CrtCheckMemory() );
35 
36  bool paramChanged = false;
37  int oldParamCount = this_->numLastValues;
38  int newParamCount = oldParamCount;
39 
40  if( this_->aeffect->getParameter )
41  {
42  enterVstCritical(this_);
43 
44  newParamCount = validateLastValues(this_);
45  float param, *params = this_->lastValues;
46  for( int paramIndex = 0; paramIndex < newParamCount; paramIndex++ )
47  {
48  param = this_->aeffect->getParameter(this_->aeffect, paramIndex);
49  if( param != params[paramIndex] )

51  params[paramIndex] = param;
52  paramChanged = true;
53  }
54  }

88 void idleDo()
89 {
90  static bool s_inHere = false;
91  if( !s_inHere )
92  {
93  s_inHere = true;
94  EnterCriticalSection(&s_idleCritical);
95 
96  assert( _CrtCheckMemory() );
97  BASS_VST_PLUGIN* this_;
98  DWORD vstHandle;
99  bool removeElem;
100  sjhashElem *elem = sjhashFirst(&s_idleHash), *next;
101  while( elem )
102  {
103  next = sjhashNext(elem);
104  removeElem = false;
105 
106  vstHandle = (DWORD)sjhashKeysize(elem);
107  this_ = refHandle(vstHandle);
108  if( this_ )
109  {
110  if( this_->needsIdle & NEEDS_IDLE_OUTSIDE_EDIT )
111  {
112  // idle outside edit
113  enterVstCritical(this_);
114  if( this_->aeffect->dispatcher(this_->aeffect, __effIdleDeprecated, 0, 0, NULL, 0.0) == 0 )
115  this_->needsIdle &= ~NEEDS_IDLE_OUTSIDE_EDIT;
116  leaveVstCritical(this_);
117  }
118 
119  if( this_->needsIdle & NEEDS_EDIT_IDLE )
120  {
121  // call the effect idle routine
122  enterVstCritical(this_);
123  this_->aeffect->dispatcher(this_->aeffect, effEditIdle, 0, 0, NULL, 0.0);
124  leaveVstCritical(this_);
125 
126  checkForChangedParam(this_);
127  }
128 
129  if( this_->needsIdle == 0 )
130  removeElem = true;
131 
132  unrefHandle(vstHandle);
133  }
134  else
135  {
136  assert( _CrtCheckMemory() );
137  removeElem = true;
138  }
139 
140  // any more idle needed for this effect?
141  if( removeElem )
142  {
143  sjhashInsert(&s_idleHash, NULL, /*pKey, not needed*/ (int)vstHandle, /*nKey (keySize)*/
144  (void*)0/*pData - 0 = remove*/);
145  }
146 
147  elem = next;
148  }
149 
150  // unload pending instances
151  if( sjhashCount(&s_unloadPendingInstances) )
152  {
153  s_unloadPendingCountdown--;
154  if( s_unloadPendingCountdown <= 0 )
155  {
156  elem = sjhashFirst(&s_unloadPendingInstances);
157  while( elem )
158  {
159  HINSTANCE inst = (HINSTANCE)sjhashKeysize(elem);
160  long unloadCount = (long)sjhashData(elem);
161 
162  while( unloadCount > 0 )
163  {
164  FreeLibrary(inst);
165  unloadCount--;
166  }
167 
168  elem = sjhashNext(elem);
169  }
170  sjhashClear(&s_unloadPendingInstances);
171  }
172  }
173 
174  // kill the timer?
175  if( sjhashCount(&s_idleHash) == 0
176  && sjhashCount(&s_unloadPendingInstances) == 0 )
177  {
178  killIdleTimers();
179  }
180 
181  LeaveCriticalSection(&s_idleCritical);
182  s_inHere = false;
183  }
184 }

188 void updateIdleTimers(BASS_VST_PLUGIN* this_)
189 {
190  EnterCriticalSection(&s_idleCritical);
191 
192  sjhashInsert(&s_idleHash, NULL, /*pKey, not needed*/ (int)this_->vstHandle, /*nKey*/
193  (void*)this_->needsIdle/*pData - 0 = remove*/);
194 
195  if( sjhashCount(&s_idleHash) || sjhashCount(&s_unloadPendingInstances) )
196  {
197  // add the idle timer, if not yet done (this is checked in createIdleTimers())
198  createIdleTimers();
199  }
200  else
201  {
202  // remove the idle timer, if it exists (this is checked in killIdleTimers())
203  killIdleTimers();
204  }
205 
206  LeaveCriticalSection(&s_idleCritical);
207 }
