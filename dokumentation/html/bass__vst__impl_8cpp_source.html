189 static void calcVstTimeInfo(BASS_VST_PLUGIN* this_, VstIntPtr toCalc)
190 {
191  this_->vstTimeInfo.flags = kVstTransportPlaying;
192  this_->vstTimeInfo.sampleRate = getSampleRate(this_);
193 
194  if( toCalc & kVstNanosValid )
195  {
196  this_->vstTimeInfo.nanoSeconds = (double)timeGetTime() * 1000000.0L;
197  this_->vstTimeInfo.flags |= kVstNanosValid;
198  }
199 
200  if( toCalc & kVstTimeSigValid )
201  {
202  this_->vstTimeInfo.timeSigNumerator = 4;
203  this_->vstTimeInfo.timeSigDenominator = 4;
204  this_->vstTimeInfo.flags |= kVstTimeSigValid;
205  }
206 
207  if( toCalc & (kVstTempoValid|kVstPpqPosValid) )
208  {
209  this_->vstTimeInfo.tempo = 120;
210  this_->vstTimeInfo.flags |= kVstTempoValid;
211  }
212 
213  if( toCalc & kVstPpqPosValid )
214  {
215  double dPos = this_->vstTimeInfo.samplePos / this_->vstTimeInfo.sampleRate;
216  this_->vstTimeInfo.ppqPos = dPos * this_->vstTimeInfo.tempo / 60.L;
217  this_->vstTimeInfo.flags |= kVstPpqPosValid;
218  }
219 
220  if( toCalc & kVstSmpteValid )
221  {
222  static const double fSmpteDiv[] = {24.f,25.f,24.f,30.f,29.97f,30.f};
223  this_->vstTimeInfo.smpteFrameRate = kVstSmpte24fps;
224 
225  double dPos = this_->vstTimeInfo.samplePos / this_->vstTimeInfo.sampleRate;
226  double dOffsetInSecond = dPos - floor(dPos);
227  this_->vstTimeInfo.smpteOffset = (long)(dOffsetInSecond * fSmpteDiv[this_->vstTimeInfo.smpteFrameRate] * 80.L);
228  this_->vstTimeInfo.flags |= kVstSmpteValid;
229  }
230 
231  if( toCalc & kVstCyclePosValid )
232  {
233  this_->vstTimeInfo.cycleStartPos = 0;
234  this_->vstTimeInfo.cycleEndPos = 0;
235  this_->vstTimeInfo.flags |= kVstCyclePosValid;
236  }

247 static VstIntPtr audioMasterCallbackImpl(AEffect* aeffect_, // on load, aeffect_ this may be NULL or incomplete, use this_->aeffeet insteed
248  VstInt32 opcode, VstInt32 index,
249  VstIntPtr value, void *ptr, float opt)
250 {
251  VstIntPtr ret = 0;
252 
253  DWORD vstHandle = (aeffect_ && aeffect_->resvd1)? // litte bug fix for 2.4.0.2: we also check for aeffect_->resvd1 now, so we will _always_ get a handle as
254  (DWORD)aeffect_->resvd1 : s_inConstructionVstHandle; // s_inConstructionVstHandle is always valid until aeffect_->resvd1 is set.
255  BASS_VST_PLUGIN* this_ = refHandle(vstHandle);
256  if( this_ == NULL )
257  return 0;
258 
259  // check if processing is done by the user
261 
262  if( this_->callback )
263  {
264  VSTPROC* callback = this_->callback;
265  void* callbackUserData = this_->callbackUserData;
266  BASS_VST_AUDIO_MASTER_PARAM amp;
267  amp.aeffect = this_->aeffect;
268  amp.opcode = opcode;
269  amp.index = index;
270  amp.value = value;
271  amp.ptr = ptr;
272  amp.opt = opt;
273  amp.doDefault = 1;
274  unrefHandle(vstHandle);
275  ret = callback(vstHandle, BASS_VST_AUDIO_MASTER, (DWORD)&amp, 0, callbackUserData);
276  if( amp.doDefault == 0 )
277  return ret;
278  ret = 0;
279  this_ = refHandle(vstHandle); // reallocate the handle
280  if( this_ == NULL )
281  return 0;
282  }
283 
284 
285  switch( opcode )
286  {
287  // VST 1.00 opcodes
289 
290  case audioMasterAutomate: // Notifies us a about a parameter change in the editor --
291  break; // however, we do not rely on this as some plugins do not send this message.
292  // Instead, we poll for parameter changes in our idle routine.
293 
294  case audioMasterVersion: // VST Version supported (for example 2200 for VST 2.2) --
295  ret = kVstVersion; // 2 for VST 2.00, 2100 for VST 2.1, 2200 for VST 2.2 etc.
296  break; // We use 2.4 although we do not support all features
297  // as some plugins relies on this without checking :-(
298  // (eg. TripleComp crashes if the version is set to 2)
299 
300  case audioMasterCurrentId: // Returns the unique id of a plug that's currently loading
301  ret = vstHandle;
302  break;
303 
304  case audioMasterIdle: // Call application idle routine (this will
305  idleDo(); // call effEditIdle for all open editors too)
306  break;
307 
308  case audioMasterPinConnected:
309  ret = index >= 2? 1 : 0; // Inquire if an input or output is beeing connected;
310  if( this_->channelHandle ) // index enumerates input or output counting from zero,
311  { // value is 0 for input and != 0 otherwise. note: the
312  BASS_CHANNELINFO info; // return value is 0 for <true> such that older versions will always return true.
313  if( BASS_ChannelGetInfo(this_->channelHandle, &info) )
314  {
315  ret = index >= (long)info.chans? 1 : 0;
316  }
317  }
318  break;
319 
320  // VST 2.00 opcodes
322 
323  case audioMasterUpdateDisplay: // the plug-in reported an update (e.g. after a program load/rename or any other param change)
324  if (this_->effStartProcessCalled)
325  {
326  enterVstCritical(this_);
327  int oldParamCount = this_->numLastValues;
328  int newParamCount = validateLastValues(this_);
329  leaveVstCritical(this_);
330  if( this_->callback )
331  this_->callback(this_->vstHandle, BASS_VST_PARAM_CHANGED, oldParamCount, newParamCount, this_->callbackUserData);
332  }
333  break;
334 
335  case audioMasterGetTime:
336  calcVstTimeInfo(this_, value);
337  ret =(VstIntPtr)&this_->vstTimeInfo;// the structure needs to be valid "a moment"; it is overwritten on the next call to audioMasterGetTime
338  break;
339 
340  case __audioMasterNeedIdleDeprecated: // plug needs idle calls (outside its editor window) (although this is deprecated it is heavily used by VST 2.3 and sooner)
341  this_->needsIdle |= NEEDS_IDLE_OUTSIDE_EDIT;
342  updateIdleTimers(this_);
343  ret = 1;
344  break;
345 
346  case audioMasterSizeWindow: // index: width, value: height
347  if( this_->callback )
348  {
349  this_->callback(vstHandle, BASS_VST_EDITOR_RESIZED, index, (DWORD)value, this_->callbackUserData);
350  }
351  break;
352 
353  case audioMasterGetSampleRate:
354  ret = getSampleRate(this_);
355  break;
356 
357  case audioMasterGetVendorString: // fills <ptr> with a string identifying the vendor (max 64 char)
358  strcpy((char*)ptr, "Bjoern Petersen Software Design and Development"/*max 64 char!*/);
359  ret = true;
360  break;
361 
362  case audioMasterGetProductString: // fills <ptr> with a string with product name (max 64 char)
363  strcpy((char*)ptr, "BASS_VST @ Silverjuke.Net");
364  ret = true;
365  break;
366 
367  case audioMasterGetVendorVersion: // returns vendor-specific version
368  ret = BASS_VST_VERSION_HEX;
369  break;
370 
371  case audioMasterCanDo: // string in ptr, see below
372  if( strcasecmp((char*)ptr, "supplyidle")==0
373  || strcasecmp((char*)ptr, "sendvstevents")==0 // we can (and will!) send events to the plugin ...
374  || strcasecmp((char*)ptr, "sendvstmidievent")==0 // ... esp. MIDI event for VSTi
375  || strcasecmp((char*)ptr, "sendvsttimeinfo")==0 // on request, we can send timing information to the plugin (see this_->vstTimeInfo)
376  || strcasecmp((char*)ptr, "openfileselector")==0 // we support audioMasterOpenFileSelector?
377  || strcasecmp((char*)ptr, "closefileselector")==0 // we support audioMasterCloseFileSelector?
378  || strcasecmp((char*)ptr, "sizewindow")==0
379  || strcasecmp((char*)ptr, "startstopprocess")==0 ) // we calls effStartProcess and effStopProcess
380  {
381  ret = 1;
382  }
383  break;
384 
385  case audioMasterGetLanguage: // see enum
386  ret = s_language;
387  break;
388 
389  // VST 2.10 opcodes
391 
392 
393  case audioMasterOpenFileSelector: // open a fileselector window with VstFileSelect* in <ptr>
394  ret = fileSelOpen(this_, (VstFileSelect*)ptr);
395  break;
396 
397  // VST 2.20 opcodes
399 
400  case audioMasterCloseFileSelector:
401  fileSelClose(this_, (VstFileSelect*)ptr);
402  break;
403 
404  // Misc / Debug only
406 
407  #if defined(_DEBUG) && defined(_WIN32)
408  case __audioMasterWantMidiDeprecated:
409  OutputDebugString("plug called audioMasterWantMidi\n");
410  break;
411 
412  default:
413  {
414  char buf[256]; sprintf(buf, "plug called opcode #%i\n", (int)opcode);
415  OutputDebugString(buf);
416  }
417  break;
418  #endif
419  }
420 
421  // Unknown opcode, always return 0
423 
424  unrefHandle(vstHandle);
425  return ret;
426 }

457 static BOOL loadVstLibrary(BASS_VST_PLUGIN* this_, const void* dllFile, DWORD createFlags)
458 {
459  dllMainEntryFuncType dllMainEntryFuncPtr;
460 
461  // init some values
462  this_->createFlags = createFlags;
463 
464  // load the library
465  if( createFlags & BASS_UNICODE )
466  this_->hinst = LoadLibraryW((const LPCWSTR)dllFile);
467  else
468  this_->hinst = LoadLibraryA((const char*)dllFile);
469 
470  if( this_->hinst == NULL )
471  {
472  SET_ERROR(BASS_ERROR_FILEOPEN);
473  return false;
474  }
475 
476  // get the plugin pointer
477  dllMainEntryFuncPtr = (dllMainEntryFuncType)GetProcAddress(this_->hinst, "VSTPluginMain");
478  if( dllMainEntryFuncPtr == NULL )
479  {
480  dllMainEntryFuncPtr = (dllMainEntryFuncType)GetProcAddress(this_->hinst, "main");
481  if( dllMainEntryFuncPtr == NULL )
482  {
483  SET_ERROR(BASS_ERROR_FILEFORM);
484  return false;
485  }
486  }
487 
488  // get the aeffect instance
489  s_inConstructionVstHandle = this_->vstHandle;
490  this_->aeffect = (dllMainEntryFuncPtr)(audioMasterCallbackImpl);
491  if( this_->aeffect == NULL
492  || this_->aeffect->magic != kEffectMagic
493  || (this_->aeffect->__processDeprecated == NULL && this_->aeffect->processReplacing == NULL && !canDoubleReplacing(this_))
494  || this_->aeffect->dispatcher == NULL )
495  {
496  SET_ERROR(BASS_ERROR_FILEFORM);
497  s_inConstructionVstHandle = 0;
498  return false;
499  }
500  this_->aeffect->resvd1 = (long)this_->vstHandle;
501  s_inConstructionVstHandle = 0;
502 
503  // check if there are enough inputs / outputs
504  if( this_->type==VSTeffect && this_->aeffect->numInputs <= 0 )
505  {
506  SET_ERROR(BASS_VST_ERROR_NOINPUTS);
507  return false;
508  }
509 
510  if( this_->aeffect->numOutputs <= 0
511  || (this_->type==VSTinstrument && !(this_->aeffect->flags&effFlagsIsSynth)) )
512  {
513  SET_ERROR(BASS_VST_ERROR_NOOUTPUTS);
514  return false;
515  }
516 
517  // call effOpen - call this before dispatching anything else!
518  this_->aeffect->dispatcher(this_->aeffect, effOpen, 0, 0, NULL, 0.0); // effOpen has no return value for errors
519  this_->effOpenCalled = true;
520 
521  // check of the module supports real time processing -
522  // checking for (PluginCanDo("noRealTime")&&PluginCanDo("offline")) is not correct: this only means a realtime plugin can also do offline processing
523  long plugCategory = (long)this_->aeffect->dispatcher(this_->aeffect, effGetPlugCategory, 0, 0, NULL, 0.0);
524  if( plugCategory == kPlugCategOfflineProcess )
525  {
526  SET_ERROR(BASS_VST_ERROR_NOREALTIME);
527  return false;
528  }
529 
530  // init the sample rate
531  long sampleRate = getSampleRate(this_);
532  this_->aeffect->dispatcher(this_->aeffect, effSetSampleRate, 0, 0, NULL, (float)sampleRate);
533  // dodgy hack to force some plugins to initialise the sample rate..
534  if (this_->aeffect->getParameter && (!(this_->aeffect->flags&effFlagsHasEditor)) && this_->aeffect->numParams > 0)
535  {
536  const float old = this_->aeffect->getParameter(this_->aeffect, 0);
537  this_->aeffect->setParameter(this_->aeffect, 0, (old < 0.5f) ? 1.0f : 0.0f);
538  this_->aeffect->setParameter(this_->aeffect, 0, old);
539  }
540 
541  // this is a safety measure against some plugins that only set their buffers
542  // ONCE - this should ensure that they allocate a buffer that's large enough.
543  // Normally, the buffer size is set dynamically in PostprocessSamples()
544  this_->aeffect->dispatcher(this_->aeffect, effSetBlockSize, 0, sampleRate/*one second*/, NULL, 0.0);
545  this_->aeffect->dispatcher(this_->aeffect, effMainsChanged, 0, 1/*resume*/, NULL, 0.0);
546 
547  this_->numDefaultValues = 0;
548  this_->numLastValues = 0;
549 
550  if( this_->aeffect->getParameter )
551  {
552  // select the first program
553  if( this_->aeffect->dispatcher(this_->aeffect, effGetProgram, 0, 0, NULL, 0.0) != 0 )
554  {
555  this_->aeffect->dispatcher(this_->aeffect, effSetProgram, 0, 0/*select first program*/, NULL, 0.0);
556  }
557 
558  int paramCount = this_->aeffect->numParams;
559  if (paramCount >= 0)
560  {
561  // remember all defaultValues and init lastValues
562  long bytesNeeded = sizeof(float) * paramCount;
563  if (bytesNeeded == 0)
564  bytesNeeded = sizeof(float) * 24;
565  this_->defaultValues = (float*)malloc(bytesNeeded);
566  this_->lastValues = (float*)malloc(bytesNeeded);
567 
568  if( this_->defaultValues == NULL || this_->lastValues == NULL)
569  {
570  SET_ERROR(BASS_ERROR_MEM);
571  return false;
572  }
573 
574  memset(this_->defaultValues, 0, bytesNeeded);
575  memset(this_->lastValues, 0, bytesNeeded);
576 
577  this_->numDefaultValues = paramCount;
578  this_->numLastValues = paramCount;
579 
580  int paramIndex;
581  for( paramIndex = 0; paramIndex < paramCount; paramIndex++ )
582  {
583  this_->defaultValues[paramIndex] = this_->aeffect->getParameter(this_->aeffect, paramIndex);
584  }
585  }
586  }
587 
588  // success
589  return true;
590 }
