39 static void cnvFloatToDouble(const float* fBufStart, double* dBufStart, long numBytes)
40 {
41  // copy backward to allow using the same buffers
42  // not that dBuf needs to be two times larger than fBuf
43  long numSamples = numBytes / sizeof(float);
44  const float* f = &fBufStart[numSamples-1];
45  double* d = &dBufStart[numSamples-1];
46  while( f >= fBufStart )
47  {
48  *d = *f;
49  d--;
50  f--;
51  }
52 }

88 static void cnvFloatToPcm16(const float* fBuf, signed short* sBuf, long numBytes)
89 {
90  // copy forward to allow using the same buffers
91  long numSamples = numBytes / sizeof(float);
92  const float* fBufEnd = &fBuf[numSamples];
93  float sample;
94  while( fBuf < fBufEnd )
95  {
96  sample = *fBuf * 32767.0F;
97  if( sample < -32768.0F ) sample = -32768.0F;
98  if( sample > 32767.0F ) sample = 32767.0F;
99  *sBuf = (signed short)sample;
100 
101  fBuf++;
102  sBuf++;
103  }
104 }

108 static void cnvFloatLLRR_To_Mono(float* bufferL, float* bufferR, long sampleCount, float gain=1.0F)
109 {
110  // To-mono-conversions sums the two channels to the left channel
111  // which will have double gain by default. The right channel is
112  // void after conversion.
113 
114  float* bufferLEnd = bufferL+sampleCount;
115 
116  while( bufferL != bufferLEnd )
117  {
118  *bufferL = (*bufferL+*bufferR) * gain;
119  bufferL++;
120  bufferR++;
121  }
122 }

126 static void cnvFloatLLRR_To_Stereo(float* bufferL, float* bufferR, long sampleCount)
127 {
128  // To-stereo-conversions duplicates the left channel to the right
129  // channel and halfs the gain of both channels.
130  float* bufferLEnd = bufferL+sampleCount;
131 
132  while( bufferL != bufferLEnd )
133  {
134  *bufferL /= 2;
135  *bufferR = *bufferL;
136  bufferL++;
137  bufferR++;
138  }

160 static bool allocChanBuffers(BASS_VST_PLUGIN* this_, long numInputs, long numOutputs, long numBytes)
161 {
162  if( numInputs > MAX_CHANS
163  || numInputs <= 0
164  || numOutputs > MAX_CHANS
165  || numOutputs <= 0 )
166  {
167  freeChansBuffers(this_);
168  return false;
169  }
170 
171  if( numBytes > this_->bytesPerInOutBuffer
172  || this_->buffersIn[numInputs-1] == NULL
173  || this_->buffersOut[numOutputs-1] == NULL )
174  {
175  // we allcate silently the double number of bytes to be prepared for double processing ...
176  #define BUFFER_HEADROOM_MULT 2
177  assert( sizeof(double)==sizeof(float)*BUFFER_HEADROOM_MULT );
178 
179  // free previously allocated buffers
180  freeChansBuffers(this_);
181 
182  // allocated the new buffers
183  int i;
184  for( i = 0; i < numInputs; i++ )
185  {
186  if( (this_->buffersIn[i]=(float*)malloc(numBytes*BUFFER_HEADROOM_MULT)) == NULL )
187  {
188  freeChansBuffers(this_);
189  return false;
190  }
191  }
192 
193  for( i = 0; i < numOutputs; i++ )
194  {
195  if( (this_->buffersOut[i]=(float*)malloc(numBytes*BUFFER_HEADROOM_MULT)) == NULL )
196  {
197  freeChansBuffers(this_);
198  return false;
199  }
200  }
201 
202  this_->bytesPerInOutBuffer = numBytes;
203 
204  this_->effBlockSize = numBytes/sizeof(float)/*sample count*/;
205  callMainsChanged(this_, this_->effBlockSize);
206 
207  EnterCriticalSection(&s_forwardCritical);
208  for( i = 0; i < this_->forwardDataToOtherCnt; i++ )
209  {
210  BASS_VST_PLUGIN* other_ = refHandle(this_->forwardDataToOtherVstHandles[i]);
211  if( other_ && other_->effStartProcessCalled )
212  callMainsChanged(other_, this_->effBlockSize);
213  unrefHandle(this_->forwardDataToOtherVstHandles[i]);
214  }
215  LeaveCriticalSection(&s_forwardCritical);
216  }
217 
218  return true;
219 }

302 static void callProcess(BASS_VST_PLUGIN* this_, BASS_VST_PLUGIN* buffers, long numSamples)
303 {
304  if( this_->effStartProcessCalled )
305  {
306  // do MIDI processing
307  EnterCriticalSection(&this_->midiCritical_);
308  if( this_->midiEventsCurr && this_->midiEventsCurr->numEvents )
309  {
310  this_->aeffect->dispatcher(this_->aeffect, effProcessEvents, 0, 0, this_->midiEventsCurr, 0.0);
311 
312  // prepare for the next round ... use the other buffer
313  VstEvents* temp = this_->midiEventsCurr; this_->midiEventsCurr = this_->midiEventsPrev; this_->midiEventsPrev = temp;
314  if( this_->midiEventsCurr )
315  this_->midiEventsCurr->numEvents = 0;
316  }
317  LeaveCriticalSection(&this_->midiCritical_);
318 
319  if( this_->aeffect->processReplacing
320  && ( (this_->aeffect->flags & effFlagsCanReplacing) || this_->aeffect->__processDeprecated == NULL) )
321  {
322  // do the normal float processing
323  this_->aeffect->processReplacing(this_->aeffect, buffers->buffersIn, buffers->buffersOut, numSamples);
324  }
325  else if( this_->aeffect->__processDeprecated )
326  {
327  // do the "old" float processing - better than the overhead for the double replacing
328  this_->aeffect->__processDeprecated(this_->aeffect, this_->buffersIn, this_->buffersOut, numSamples);
329  }
330  else if( canDoubleReplacing(this_) )
331  {
332  // convert all buffers to double; the output buffer is already emptied incl. the double headroom
333  double* doubleIn[MAX_CHANS];
334  double* doubleOut[MAX_CHANS];
335  int i;
336  for( i = 0; i < MAX_CHANS; i++ )
337  {
338  doubleIn[i] = (double*)buffers->buffersIn[i];
339  if( doubleIn[i] )
340  cnvFloatToDouble(buffers->buffersIn[i], doubleIn[i], numSamples*sizeof(float));
341 
342  doubleOut[i] = (double*)buffers->buffersOut[i];
343  }
344 
345  // do process double replacing
346  this_->aeffect->processDoubleReplacing(this_->aeffect, doubleIn, doubleOut, numSamples);
347 
348  // convert all buffers back to floats; this is also needed for the input buffers as
349  // callProcess() may be called for several instances of effects (eg. for editor forwarding)
350  for( i = 0; i < MAX_CHANS; i++ )
351  {
352  if( doubleIn[i] )
353  cnvDoubleToFloat(doubleIn[i], buffers->buffersIn[i], numSamples*sizeof(double));
354 
355  if( doubleOut[i] )
356  cnvDoubleToFloat(doubleOut[i], buffers->buffersOut[i], numSamples*sizeof(double));
357  }
358  }
359  }
360 }

364 void CALLBACK doEffectProcess(HDSP dspHandle, DWORD channelHandle, void* buffer__, DWORD bufferBytes__, USERPTR vstHandle__)
365 {
366  DWORD vstHandle = (DWORD)vstHandle__;
367  BASS_CHANNELINFO channelInfo;
368  int i;
369  long requiredInputs;
370  long requiredOutputs;
371 
372  float* floatBuffer;
373  long numSamples;
374  bool cnvPcm2Float;
375  bool cnvMonoToStereo = false;
376 
377  BASS_VST_PLUGIN* this_ = refHandle(vstHandle);
378  if( this_ == NULL || channelHandle != this_->channelHandle || dspHandle != this_->dspHandle || buffer__ == NULL || bufferBytes__ <= 0 )
379  goto Cleanup; // error already logged
380 
381  // get the channel information
382  if( !BASS_ChannelGetInfo(channelHandle, &channelInfo)
383  || channelInfo.chans <= 0 )
384  goto Cleanup;
385 
386  requiredInputs = this_->aeffect->numInputs;
387  if( (long)channelInfo.chans > requiredInputs )
388  requiredInputs = channelInfo.chans;
389 
390  requiredOutputs = this_->aeffect->numOutputs;
391  if( (long)channelInfo.chans > requiredOutputs )
392  requiredOutputs = channelInfo.chans;
393 
394  // get the data as floats.
395  // this is not lossy.
396  cnvPcm2Float = ((channelInfo.flags&BASS_SAMPLE_FLOAT)==0 && (this_->type==VSTinstrument || BASS_GetConfig(BASS_CONFIG_FLOATDSP)==0));
397  if( cnvPcm2Float )
398  {
399  if( channelInfo.flags & BASS_SAMPLE_8BITS )
400  goto Cleanup; // can't and won't do this
401 
402  if( !allocTempBuffer(this_, bufferBytes__*2) )
403  goto Cleanup;
404  cnvPcm16ToFloat((signed short*)buffer__, this_->bufferTemp, bufferBytes__);
405 
406  floatBuffer = this_->bufferTemp;
407  numSamples = (bufferBytes__ / sizeof(signed short)) / channelInfo.chans;
408  }
409  else
410  {
411  floatBuffer = (float*)buffer__;
412  numSamples = (bufferBytes__ / sizeof(float)) / channelInfo.chans;
413  }
414 
415  if( numSamples <= 0 )
416  goto Cleanup;
417 
418  // copy the given LRLRLR buffer to the VST LLLRRR buffers
419  // this is not lossy
420  if( !allocChanBuffers(this_, requiredInputs, requiredOutputs, numSamples*sizeof(float)) )
421  goto Cleanup;
422 
423  {
424  long chans = channelInfo.chans, c = 0;
425  float* buffer = (float*)floatBuffer;
426  float* end = &buffer[numSamples * chans];
427  float** in = this_->buffersIn;
428  i = 0;
429  while( buffer < end )
430  {
431  in[c][i] = *buffer;
432  buffer ++;
433  c++;
434  if( c == chans )
435  {
436  c = 0;
437  i++;
438  }
439  }
440 
441  for( c = chans; c < requiredInputs; c++ )
442  memset(in[c], 0, numSamples * sizeof(float));
443  }
444 
445  // special mono-processing effect handling
446  if( this_->aeffect->numInputs == 1
447  && this_->aeffect->numOutputs <= 2
448  && channelInfo.chans > 1
449  && !(this_->createFlags&BASS_VST_KEEP_CHANS) )
450  {
451  cnvFloatLLRR_To_Mono(this_->buffersIn[0], this_->buffersIn[1], numSamples,
452  this_->aeffect->numOutputs == 1? 1.0F : 0.5F);
453 
454  if( this_->aeffect->numOutputs == 1 )
455  cnvMonoToStereo = true;
456  }
457 
458  // empty the output buffers and process
459  // (normally this is needed only for process() and not for processReplacing();
460  // however, this has to be done even in processReplacing() since some VSTIs
461  // (most notably those from Steinberg... hehe) obviously don't implement
462  // processReplacing() as a separate function but rather use process())
463  enterVstCritical(this_);
464  if( !this_->doBypass )
465  {
466  this_->vstTimeInfo.samplePos += numSamples;
467  if( this_->vstTimeInfo.samplePos < 0.0 )
468  this_->vstTimeInfo.samplePos = 0.0;
469 
470  EnterCriticalSection(&s_forwardCritical);
471  for( i = 0; i < this_->forwardDataToOtherCnt; i++ )
472  {
473  clearOutputBuffers(this_, numSamples);
474  BASS_VST_PLUGIN* other_ = refHandle(this_->forwardDataToOtherVstHandles[i]);
475  if( other_ )
476  {
477  if( tryEnterVstCritical(other_) )
478  {
479  callProcess(other_, this_/*buffers to use*/, numSamples);
480  leaveVstCritical(other_);
481  }
482  }
483  unrefHandle(this_->forwardDataToOtherVstHandles[i]);
484  }
485  LeaveCriticalSection(&s_forwardCritical);
486 
487  // the "real" sound processing (the one above is only for the editors to get data)
488  clearOutputBuffers(this_, numSamples);
489  callProcess(this_, this_/*buffers to use*/, numSamples);
490 
491  // special mono-processing effect handling
492  if( cnvMonoToStereo )
493  {
494  cnvFloatLLRR_To_Stereo(this_->buffersOut[0], this_->buffersOut[1], numSamples);
495  }
496 
497  // convert the returned data back to our channel representation (LLLLLRRRRR to LRLRLRLR)
498  // this is not lossy
499  {
500  long chans = channelInfo.chans, c = 0;
501  float* buffer = (float*)floatBuffer;
502  float* end = &buffer[numSamples * chans];
503  float** out = this_->buffersOut;
504  i = 0;
505  while( buffer < end )
506  {
507  *buffer = out[c][i];
508  buffer++;
509  c++;
510  if( c == chans )
511  {
512  c = 0;
513  i++;
514  }
515  }
516  }
517 
518  // convert the data back to PCM, if needed
519  // this is lossy
520  if( cnvPcm2Float )
521  {
522  cnvFloatToPcm16(floatBuffer, (signed short*)buffer__, numSamples * sizeof(float) * channelInfo.chans);
523  }
524  }
525  leaveVstCritical(this_);
526 
527  // done
528 Cleanup:
529  unrefHandle(vstHandle);
530 }

552 {
553  // really not yet opened?
554  if( this_->effStartProcessCalled )
555  {
556  return false;
557  }
558 
559  // do we have a BASS handle?
560  if( info_->channelHandle == 0 )
561  {
562  return false;
563  }
564 
565  // get information about the channel
566  BASS_CHANNELINFO channelInfo;
567  if( !BASS_ChannelGetInfo(info_->channelHandle, &channelInfo) )
568  {
569  return false; // error already logged
570  }
571

573

577  {

580  }
581
