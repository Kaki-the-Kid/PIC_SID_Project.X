278 void c64::interruptIRQ(bool state)
279 {
280  if (state)
281  {
282  if (irqCount == 0)
283  cpu.triggerIRQ();
284 
285  irqCount ++;
286  }
287  else
288  {
289  irqCount --;
290  if (irqCount == 0)
