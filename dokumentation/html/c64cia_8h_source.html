68  void poke(uint_least16_t address, uint8_t value) override
69  {
70  write(endian_16lo8(address), value);
71 
72  // Save the value written to Timer A
73  if (address == 0xDC04 || address == 0xDC05)
74  {
75  if (timerA.getTimer() != 0)
76  last_ta = timerA.getTimer();
77  }
78  }
79

81  {
82  return read(endian_16lo8(address));
83  }
84 
85  void reset() override
86  {

89  }
90
