210 static int _exSID_thread_output(void *arg)
211 {
212  xsdbg("thread started\n");
213  while (1) {
214  mtx_lock(&frontbuf_mtx);
215 
216  // wait for frontbuf ready (not empty)
217  while (!frontbuf_idx)
218  cnd_wait(&frontbuf_ready_cnd, &frontbuf_mtx);
219 
220  if (unlikely(frontbuf_idx < 0)) { // exit condition
221  xsdbg("thread exiting!\n");
222  mtx_unlock(&frontbuf_mtx);
223  thrd_exit(0);
224  }
225 
226  xSwrite(frontbuf, frontbuf_idx);
227  frontbuf_idx = 0;
228 
229  // xSread() and xSoutb() are in the same thread of execution
230  // so it can only be one or the other waiting.
231  cnd_signal(&frontbuf_done_cnd);
232  mtx_unlock(&frontbuf_mtx);
233  }
234  return 0; // make the compiler happy

246 static void xSoutb(uint8_t byte, int flush)
247 {
248 #ifndef EXSID_THREADED
249  static unsigned char backbuf[XS_BUFFSZ];
250  static int backbuf_idx = 0;
251 #else
252  unsigned char * bufptr = NULL;
253 #endif
254 
255  backbuf[backbuf_idx++] = (unsigned char)byte;
256 
257  if (likely((backbuf_idx < XS_BUFFSZ) && !flush))
258  return;
259 
260 #ifdef EXSID_THREADED
261  // buffer dance
262  mtx_lock(&frontbuf_mtx);
263 
264  // wait for frontbuf available (empty). Only triggers if previous
265  // write buffer hasn't been consummed before we get here again.
266  while (unlikely(frontbuf_idx))
267  cnd_wait(&frontbuf_done_cnd, &frontbuf_mtx);
268 
269  if (unlikely(flush < 0)) // indicate exit request
270  frontbuf_idx = -1;
271  else { // flip buffers
272  bufptr = frontbuf;
273  frontbuf = backbuf;
274  frontbuf_idx = backbuf_idx;
275  backbuf = bufptr;
276  backbuf_idx = 0;
277  }
278 
279  cnd_signal(&frontbuf_ready_cnd);
280  mtx_unlock(&frontbuf_mtx);
281 #else // unthreaded
282  xSwrite(backbuf, backbuf_idx);
283  backbuf_idx = 0;
284 #endif
285 }

294 int exSID_init(void)
295 {
296  unsigned char dummy;
297  int i, found, ret;
298 
299  if (ftdi) {
300  xserror("Device already open!");
301  return -1;
302  }
303 
304  if (xSfw_dlopen()) {
305  //xserror("Failed to open dynamic loader"); // already sets its own error message
306  return -1;
307  }
308 
309  /* Attempt to open all supported devices until first success.
310  * Cleanup ftdi after each try to avoid passing garbage around as we don't know what
311  * the FTDI open routines do with the pointer.
312  * FIXME: despite that, some combinations still don't seem to work */
313  found = 0;
314  for (i = 0; i < ARRAY_SIZE(xSsupported); i++) {
315  if (xSfw_new) {
316  ftdi = xSfw_new();
317  if (!ftdi) {
318  xserror("ftdi_new failed");
319  return -1;
320  }
321  }
322 
323  xsdbg("Trying %s...\n", xSsupported[i].desc);
324  xSpriv = &xSsupported[i].xsp; // setting unconditionnally avoids segfaults if user code does the wrong thing.
325  ftdi_status = xSfw_usb_open_desc(&ftdi, xSsupported[i].vid, xSsupported[i].pid, xSsupported[i].desc, NULL);
326  if (ftdi_status >= 0) {
327  xsdbg("Opened!\n");
328  found = 1;
329  break;
330  }
331  else {
332  xsdbg("Failed: %d (%s)\n", ftdi_status, xSfw_get_error_string(ftdi));
333  if (xSfw_free)
334  xSfw_free(ftdi);
335  ftdi = NULL;
336  }
337  }
338 
339  if (!found) {
340  xserror("No device could be opened");
341  return -1;
342  }
343 
344  ftdi_status = xSfw_usb_setup(ftdi, XS_BDRATE, XS_USBLAT);
345  if (ftdi_status < 0) {
346  // xserror("Failed to setup device"); // already sets its own error message
347  return -1;
348  }
349 
350  // success - device is ready
351  xsdbg("Device ready\n");
352 
353 
354 #ifdef EXSID_THREADED
355  xsdbg("Thread setup\n");
356  ret = mtx_init(&frontbuf_mtx, mtx_plain);
357  ret |= cnd_init(&frontbuf_ready_cnd);
358  ret |= cnd_init(&frontbuf_done_cnd);
359  backbuf_idx = frontbuf_idx = 0;
360  ret |= thrd_create(&thread_output, _exSID_thread_output, NULL);
361  if (ret) {
362  xserror("Thread setup failed");
363  return -1;
364  }
365 #endif
366 
367  xSfw_usb_purge_buffers(ftdi); // Purge both Rx and Tx buffers
368 
369  // Wait for device ready by trying to read FV and wait for the answer
370  // XXX Broken with libftdi due to non-blocking read :-/
371  xSoutb(XS_AD_IOCTFV, 1);
372  xSread(&dummy, 1);
373 
374  xsdbg("Rock'n'roll!\n");
375 
376 #ifdef DEBUG
377  exSID_hwversion();
378  xsdbg("XS_BUFFSZ: %d bytes\n", XS_BUFFSZ);
379 #endif
380 
381  return 0;
382 }

627 static void xSlongdelay(uint_fast32_t cycles)
628 {
629  int multiple, flush;
630  uint_fast32_t delta;
631  unsigned char dummy;
632 
633  flush = (XS_MODEL_STD == xSpriv->model);
634 
635  multiple = cycles - xSpriv->ldelay_offs;
636  delta = multiple % XS_LDMULT;
637  multiple /= XS_LDMULT;
638 
639  //xsdbg("ldelay: %" PRIdFAST32 ", multiple: %d, delta: %" PRIdFAST32 "\n", cycles, multiple, delta);
640 
641  if (unlikely(multiple < 0)) {
642  xsdbg("Wrong delay!\n");
643  return;
644  }
645 
646 #ifdef DEBUG
647  accdelay += (cycles - delta);
648 #endif
649 
650  while (multiple >= 255) {
651  _exSID_write(XS_AD_IOCTLD, 255, flush);
652  if (flush)
653  xSread(&dummy, 1); // wait for answer with blocking read
654  multiple -= 255;
655  }
656 
657  if (multiple) {
658  _exSID_write(XS_AD_IOCTLD, (unsigned char)multiple, flush);
659  if (flush)
660  xSread(&dummy, 1); // wait for answer with blocking read
661  }
662 
663  // deal with remainder
664  xSdelay(delta);
665 }

720 void exSID_clkdwrite(uint_fast32_t cycles, uint_least8_t addr, uint8_t data)
721 {
722  static int adj = 0;
723 
724 #ifdef DEBUG
725  if (unlikely(addr > 0x18)) {
726  xsdbg("Invalid write: %.2" PRIxLEAST8 "\n", addr);
727  exSID_delay(cycles);
728  return;
729  }
730 #endif
731 
732  // actual write will cost write_cycles. Delay for cycles - write_cycles then account for the write
733  clkdrift += cycles;
734  if (clkdrift > xSpriv->write_cycles)
735  xSdelay(clkdrift - xSpriv->write_cycles);
736 
737  clkdrift -= xSpriv->write_cycles; // write is going to consume write_cycles clock ticks
738 
739 #ifdef DEBUG
740  if (clkdrift >= xSpriv->mindel_cycles)
741  xsdbg("Impossible drift adjustment! %" PRIdFAST32 " cycles\n", clkdrift);
742  else if (clkdrift < 0)
743  accdrift += clkdrift;
744 #endif
745 
746  /* if we are still going to be early, delay actual write by up to XS_MAXAD ticks
747  At this point it is guaranted that clkdrift will be < mindel_cycles. */
748  if (likely(clkdrift >= 0)) {
749  adj = clkdrift % (xSpriv->max_adj+1);
750  /* if max_adj is >= clkdrift, modulo will give the same results
751  as the correct test:
752  adj = (clkdrift < max_adj ? clkdrift : max_adj)
753  but without an extra conditional branch. If is is < max_adj, then it
754  seems to provide better results by evening jitter accross writes. So
755  it's the preferred solution for all cases. */
756  addr = (unsigned char)(addr | (adj << 5)); // final delay encoded in top 3 bits of address
757 #ifdef DEBUG
758  accdrift += (clkdrift - adj);
759 #endif
760  //xsdbg("drft: %d, adj: %d, addr: %.2hhx, data: %.2hhx\n", clkdrift, adj, (char)(addr | (adj << 5)), data);
761  }
762 
763 #ifdef DEBUG
764  acccycle += cycles;
765  accioops++;
766 #endif
767 
768  //xsdbg("delay: %d, clkdrift: %d\n", cycles, clkdrift);
769  _exSID_write(addr, data, 0);
770 }

816 uint8_t exSID_clkdread(uint_fast32_t cycles, uint_least8_t addr)
817 {
818  static int adj = 0;
819 
820 #ifdef DEBUG
821  if (unlikely((addr < 0x19) || (addr > 0x1C))) {
822  xsdbg("Invalid read: %.2" PRIxLEAST8 "\n", addr);
823  exSID_delay(cycles);
824  return 0xFF;
825  }
826 #endif
827 
828  // actual read will happen after read_pre_cycles. Delay for cycles - read_pre_cycles then account for the read
829  clkdrift += xSpriv->read_offset_cycles; // 2-cycle offset adjustement, see function documentation.
830  clkdrift += cycles;
831  if (clkdrift > xSpriv->read_pre_cycles)
832  xSdelay(clkdrift - xSpriv->read_pre_cycles);
833 
834  clkdrift -= xSpriv->read_pre_cycles; // read request is going to consume read_pre_cycles clock ticks
835 
836 #ifdef DEBUG
837  if (clkdrift > xSpriv->mindel_cycles)
838  xsdbg("Impossible drift adjustment! %" PRIdFAST32 " cycles", clkdrift);
839  else if (clkdrift < 0) {
840  accdrift += clkdrift;
841  xsdbg("Late read request! %" PRIdFAST32 " cycles\n", clkdrift);
842  }
843 #endif
844 
845  // if we are still going to be early, delay actual read by up to max_adj ticks
846  if (likely(clkdrift >= 0)) {
847  adj = clkdrift % (xSpriv->max_adj+1); // see clkdwrite()
848  addr = (unsigned char)(addr | (adj << 5)); // final delay encoded in top 3 bits of address
849 #ifdef DEBUG
850  accdrift += (clkdrift - adj);
851 #endif
852  //xsdbg("drft: %d, adj: %d, addr: %.2hhx, data: %.2hhx\n", clkdrift, adj, (char)(addr | (adj << 5)), data);
853  }
854 
855 #ifdef DEBUG
856  acccycle += cycles;
857  accioops++;
858 #endif
859 
860  // after read has completed, at least another read_post_cycles will have been spent
861  clkdrift -= xSpriv->read_post_cycles;
862 
863  //xsdbg("delay: %d, clkdrift: %d\n", cycles, clkdrift);
864  return _exSID_read(addr, 1);
