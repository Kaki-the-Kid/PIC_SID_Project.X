646 void mousecommands(void)
647 {
648  int c;
649 
650  if (!mouseb) return;
651 
652  // Pattern editpos & pattern number selection
653  for (c = 0; c < MAX_CHN; c++)
654  {
655  if ((mousey == 2) && (mousex >= 13 + c*15) && (mousex <= 14 + c*15))
656  {
657  if ((!prevmouseb) || (mouseheld > HOLDDELAY))
658  {
659  if (mouseb & MOUSEB_LEFT)
660  {
661  epchn = c;
662  nextpattern();
663  }
664  if (mouseb & MOUSEB_RIGHT)
665  {
666  epchn = c;
667  prevpattern();
668  }
669  }
670  }
671  else
672  {
673  if ((mousey >= 2) && (mousey <= 34) && (mousex >= 6 + c*15) && (mousex <= 14 + c*15))
674  {
675  int x = mousex-6-c*15;
676  int newpos = mousey-3+epview;
677  if (newpos < 0) newpos = 0;
678  if (newpos > pattlen[epnum[epchn]]) newpos = pattlen[epnum[epchn]];
679 
680  editmode = EDIT_PATTERN;
681 
682  if ((mouseb & (MOUSEB_RIGHT|MOUSEB_MIDDLE)) && (!prevmouseb))
