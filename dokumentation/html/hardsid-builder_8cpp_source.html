70 // Create a new sid emulation.
71 unsigned int HardSIDBuilder::create(unsigned int sids)
72 {
73  m_status = true;
74 
75  // Check available devices
76  unsigned int count = availDevices();
77  if (count == 0)
78  {
79  m_errorBuffer = "HARDSID ERROR: No devices found (run HardSIDConfig)";
80  goto HardSIDBuilder_create_error;
81  }
82 
83  if (count < sids)
84  sids = count;
85 
86  for (count = 0; count < sids; count++)
87  {
88  try
89  {
90  std::unique_ptr<libsidplayfp::HardSID> sid(new libsidplayfp::HardSID(this));
91 
92  // SID init failed?
93  if (!sid->getStatus())
94  {
95  m_errorBuffer = sid->error();
96  goto HardSIDBuilder_create_error;
97  }
98  sidobjs.insert(sid.release());
99  }
100  // Memory alloc failed?
101  catch (std::bad_alloc const &)
102  {
103  m_errorBuffer.assign(name()).append(" ERROR: Unable to create HardSID object");
104  goto HardSIDBuilder_create_error;
105  }
106 
107 
108  }
109  return count;
110 
111 HardSIDBuilder_create_error:
112  m_status = false;
113  return count;
114 }

145 // Load the library and initialise the hardsid
146 int HardSIDBuilder::init()
147 {
148  HINSTANCE dll;
149 
150  if (hsid2.Instance)
151  return 0;
152 
153  m_status = false;
154 
155  dll = LoadLibrary(TEXT("HARDSID.DLL"));
156 
157  if (!dll)
158  {
159  DWORD err = GetLastError();
160  if (err == ERROR_DLL_INIT_FAILED)
161  m_errorBuffer = "HARDSID ERROR: hardsid.dll init failed!";
162  else
163  m_errorBuffer = "HARDSID ERROR: hardsid.dll not found!";
164  goto HardSID_init_error;
165  }
166 
167  { // Export Needed Version 1 Interface
168  HsidDLL1_InitMapper_t mapper;
169  mapper = (HsidDLL1_InitMapper_t) GetProcAddress(dll, "InitHardSID_Mapper");
170 
171  if (mapper)
172  mapper();
173  else
174  {
175  m_errorBuffer = "HARDSID ERROR: hardsid.dll is corrupt!";
176  goto HardSID_init_error;
177  }
178  }
179 
180  { // Check for the Version 2 interface
181  HsidDLL2_Version_t version;
182  version = (HsidDLL2_Version_t) GetProcAddress(dll, "HardSID_Version");
183  if (!version)
184  {
185  m_errorBuffer = "HARDSID ERROR: hardsid.dll not V2";
186  goto HardSID_init_error;
187  }
188  hsid2.Version = version ();
189  }
190 
191  {
192  WORD version = hsid2.Version;
193  if ((version >> 8) != (HSID_VERSION_MIN >> 8))
194  {
195  std::ostringstream ss;
196  ss << "HARDSID ERROR: hardsid.dll not V" << (HSID_VERSION_MIN >> 8) << std::endl;
197  m_errorBuffer = ss.str();
198  goto HardSID_init_error;
199  }
200 
201  if (version < HSID_VERSION_MIN)
202  {
203  std::ostringstream ss;
204  ss.fill('0');
205  ss << "HARDSID ERROR: hardsid.dll hardsid.dll must be V";
206  ss << std::setw(2) << (HSID_VERSION_MIN >> 8);
207  ss << ".";
208  ss << std::setw(2) << (HSID_VERSION_MIN & 0xff);
209  ss << " or greater" << std::endl;
210  m_errorBuffer = ss.str();
211  goto HardSID_init_error;
212  }
213  }
214 
215  // Export Needed Version 2 Interface
216  hsid2.Delay = (HsidDLL2_Delay_t) GetProcAddress(dll, "HardSID_Delay");
217  hsid2.Devices = (HsidDLL2_Devices_t) GetProcAddress(dll, "HardSID_Devices");
218  hsid2.Filter = (HsidDLL2_Filter_t) GetProcAddress(dll, "HardSID_Filter");
219  hsid2.Flush = (HsidDLL2_Flush_t) GetProcAddress(dll, "HardSID_Flush");
220  hsid2.MuteAll = (HsidDLL2_MuteAll_t) GetProcAddress(dll, "HardSID_MuteAll");
221  hsid2.Read = (HsidDLL2_Read_t) GetProcAddress(dll, "HardSID_Read");
222  hsid2.Sync = (HsidDLL2_Sync_t) GetProcAddress(dll, "HardSID_Sync");
223  hsid2.Write = (HsidDLL2_Write_t) GetProcAddress(dll, "HardSID_Write");
224 
225  if (hsid2.Version < HSID_VERSION_204)
226  hsid2.Reset = (HsidDLL2_Reset_t) GetProcAddress(dll, "HardSID_Reset");
227  else
228  {
229  hsid2.Lock = (HsidDLL2_Lock_t) GetProcAddress(dll, "HardSID_Lock");
230  hsid2.Unlock = (HsidDLL2_Unlock_t) GetProcAddress(dll, "HardSID_Unlock");
231  hsid2.Reset2 = (HsidDLL2_Reset2_t) GetProcAddress(dll, "HardSID_Reset2");
232  }
233 
234  if (hsid2.Version < HSID_VERSION_207)
235  hsid2.Mute = (HsidDLL2_Mute_t) GetProcAddress(dll, "HardSID_Mute");
236  else
237  hsid2.Mute2 = (HsidDLL2_Mute2_t) GetProcAddress(dll, "HardSID_Mute2");
238 
239  hsid2.Instance = dll;
240  m_status = true;
241  return 0;
242 
243 HardSID_init_error:
244  if (dll)
245  FreeLibrary(dll);
246  return -1;
247 }

254 // Find the number of sid devices. We do not care about
255 // stuppid device numbering or drivers not loaded for the
256 // available nodes.
257 int HardSIDBuilder::init()
258 {
259  DIR *dir = opendir("/dev");
260  if (!dir)
261  return -1;
262 
263  m_count = 0;
264 
265  while (dirent *entry=readdir(dir))
266  {
267  // SID device
268  if (strncmp ("sid", entry->d_name, 3))
269  continue;
270 
271  // if it is truely one of ours then it will be
272  // followed by numerics only
273  const char *p = entry->d_name+3;
274  unsigned int index = 0;
275  while (*p)
276  {
277  if (!isdigit(*p))
278  continue;
279  index = index * 10 + (*p++ - '0');
280  }
281  index++;
282  if (m_count < index)
283  m_count = index;
284  }
285 
286  closedir(dir);
287  return 0;
288 }
