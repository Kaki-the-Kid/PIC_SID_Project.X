86 event_clock_t HardSID::delay()
87 {
88  event_clock_t cycles = eventScheduler->getTime(m_accessClk, EVENT_CLOCK_PHI1);
89  m_accessClk += cycles;
90 
91  while (cycles > 0xFFFF)
92  {
93  hsid2.Delay((BYTE) m_instance, 0xFFFF);
94  cycles -= 0xFFFF;
95  }
96 
97  return cycles;
98 }

119 void HardSID::reset(uint8_t volume)
120 {
121  m_accessClk = 0;
122  // Clear hardsid buffers
123  hsid2.Flush ((BYTE) m_instance);
124  if (hsid2.Version >= HSID_VERSION_204)
125  hsid2.Reset2((BYTE) m_instance, volume);
126  else
127  hsid2.Reset((BYTE) m_instance);
128  hsid2.Sync((BYTE) m_instance);
129 
130  if (eventScheduler != nullptr)
131  eventScheduler->schedule(*this, HARDSID_DELAY_CYCLES, EVENT_CLOCK_PHI1);
132 }

167 void HardSID::event()
168 {
169  event_clock_t cycles = eventScheduler->getTime(m_accessClk, EVENT_CLOCK_PHI1);
170  if (cycles < HARDSID_DELAY_CYCLES)
171  {
172  eventScheduler->schedule(*this, HARDSID_DELAY_CYCLES - cycles,
173  EVENT_CLOCK_PHI1);
174  }
175  else
176  {
177  m_accessClk += cycles;
178  hsid2.Delay((BYTE) m_instance, (WORD) cycles);
179  eventScheduler->schedule(*this, HARDSID_DELAY_CYCLES, EVENT_CLOCK_PHI1);
180  }
181 }
