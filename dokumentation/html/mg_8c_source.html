300 void isr_tc0_coma() {
301  in _Sreg, SREG //Save regs...
302  movw _Zreg, ZL
303  movw _Yreg, YL
304 
305  ldiw Y, Notes //Process all notes
306  clrw T2 //Clear accumlator
307 
308 tone_lp:
309  ldd EH, Y+ns_rptr //Load wave table pointer
310  lddw Z, Y+ns_rptr+1
311  lpm EL, Z //Get a sample
312  lddw T4, Y+ns_freq //Load angular speed
313  add EH, T4L //Increase wave table ptr (next angle)
314  adc ZL, T4H //
315  adc ZH, _0
316  cpi ZH, high(wt_end*2) //Repeat sustain area
317  brcs PC+4 //
318  subiw Z, (wt_end-wt_loop)*2 //
319  std Y+ns_wrap, _0
320  std Y+ns_rptr, EH //Save wave table ptr
321  stdw Y+ns_rptr+1, Z
322  ldd EH, Y+ns_lvl //Apply envelope curve
323  MULT
324  addw T2, T0 //Add the sample to accumlator
325  adiw YL, 9 //Next note
326  cpi YL, low(Notes+nsize*N_NOTE)//

328

333  brlt PC+2

372 /******************************************************************
373  * Wavetable Synthesizer
374  * The wavetable synthesizer synthesizes a sound with pre-sampled
375  * (pre-synthesized) waveform. This is a method that have established
376  * at dawn of digital synthesizer. MOD Sound on AMIGA is the first time
377  * that wavetable synthesizer was realized on the PC if my memory serves.
378  * Ofcourse the wavetable synthesizer is embedded as a sound driver into
379  * recent PCs. The theory of wavetable synthesizer is very simple and its
380  * processor load is relatively low. However, it requires large amount of
381  * memory because there are wavetable data for source of sound. Therefore,
382  * following technics are applied to existing wavetable synthesizer.
383  *
384  * Not reserve wavetables for each tone. It can also synthesize any tone
385  * with a wavetable with changing play back speed.
386  *
387  * Not reserve all part of wavetable. Monotonous part can be replaced with
388  * a loop.
389  *
390  * Decaying waveform can be simulated in envelope genarotor.
391  ******************************************************************/
392
