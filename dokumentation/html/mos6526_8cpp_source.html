98 void InterruptSource6526::trigger(uint8_t interruptMask)
99 {
100  // timer b bug
101  if (interruptMask == InterruptSource::INTERRUPT_UNDERFLOW_B)
102  {
103  tbBug = (eventScheduler.getTime(EVENT_CLOCK_PHI2) == last_clear+1);
104  }
105 
106  InterruptSource::trigger(interruptMask);
107 
108  if (!interruptMasked())
109  return;
110 
111  if (eventScheduler.getTime(EVENT_CLOCK_PHI2) == last_clear)
112  {
113  return;
114  }
115 
116  if (tbBug)
117  {
118  triggerBug();
119  tbBug = false;
120  }
121 
122  if (!interruptTriggered())
123  {
124  schedule();
125  }
126 }

221 uint8_t MOS6526::read(uint_least8_t addr)
222 {
223  addr &= 0x0f;
224 
225  timerA.syncWithCpu();
226  timerA.wakeUpAfterSyncWithCpu();
227  timerB.syncWithCpu();
228  timerB.wakeUpAfterSyncWithCpu();
229 
230  switch (addr)
231  {
232  case PRA: // Simulate a serial port
233  return (regs[PRA] | ~regs[DDRA]);
234  case PRB:
235  {
236  uint8_t data = regs[PRB] | ~regs[DDRB];
237  // Timers can appear on the port
238  if (regs[CRA] & 0x02)
239  {
240  data &= 0xbf;
241  if (timerA.getPb(regs[CRA]))
242  data |= 0x40;
243  }
244  if (regs[CRB] & 0x02)
245  {
246  data &= 0x7f;
247  if (timerB.getPb(regs[CRB]))
248  data |= 0x80;
249  }
250  return data;
251  }
252  case TAL:
253  return endian_16lo8(timerA.getTimer());
254  case TAH:
255  return endian_16hi8(timerA.getTimer());
256  case TBL:
257  return endian_16lo8(timerB.getTimer());
258  case TBH:
259  return endian_16hi8(timerB.getTimer());
260  case TOD_TEN:
261  case TOD_SEC:
262  case TOD_MIN:
263  case TOD_HR:
264  return tod.read(addr - TOD_TEN);
265  case IDR:
266  return interruptSource->clear();
267  case CRA:
268  return (regs[CRA] & 0xee) | (timerA.getState() & 1);
269  case CRB:
270  return (regs[CRB] & 0xee) | (timerB.getState() & 1);
271  default:
272  return regs[addr];
273  }
274 }

276 void MOS6526::write(uint_least8_t addr, uint8_t data)
277 {
278  addr &= 0x0f;
279 
280  timerA.syncWithCpu();
281  timerB.syncWithCpu();
282 
283  const uint8_t oldData = regs[addr];
284  regs[addr] = data;
285 
286  switch (addr)
287  {
288  case PRA:
289  case DDRA:
290  portA();
291  break;
292  case PRB:
293  case DDRB:
294  portB();
295  break;
296  case TAL:
297  timerA.latchLo(data);
298  break;
299  case TAH:
300  timerA.latchHi(data);
301  break;
302  case TBL:
303  timerB.latchLo(data);
304  break;
305  case TBH:
306  timerB.latchHi(data);
307  break;
308  case TOD_TEN:
309  case TOD_SEC:
310  case TOD_MIN:
311  case TOD_HR:
312  tod.write(addr - TOD_TEN, data);
313  break;
314  case SDR:
315  if (regs[CRA] & 0x40)
316  serialPort.setBuffered();
317  break;
318  case ICR:
319  interruptSource->set(data);
320  break;
321  case CRA:
322  if ((data & 1) && !(oldData & 1))
323  {
324  // Reset the underflow flipflop for the data port
325  timerA.setPbToggle(true);
326  }
327  timerA.setControlRegister(data);
328  break;
329  case CRB:
330  if ((data & 1) && !(oldData & 1))
331  {
332  // Reset the underflow flipflop for the data port
333  timerB.setPbToggle(true);
334  }
335  timerB.setControlRegister(data | (data & 0x40) >> 1);
336  break;
337  }
338 
339  timerA.wakeUpAfterSyncWithCpu();
340  timerB.wakeUpAfterSyncWithCpu();
341 }
