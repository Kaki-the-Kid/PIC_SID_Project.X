112 /* operator struct definition
113  For OPL2 all 9 channels consist of two operators each, carrier and modulator.
114  Channel x has operators x as modulator and operators (9+x) as carrier.
115  For OPL3 all 18 channels consist either of two operators (2op mode) or four
116  operators (4op mode) which is determined through register4 of the second
117  adlib register set.
118  Only the channels 0,1,2 (first set) and 9,10,11 (second set) can act as
119  4op channels. The two additional operators for a channel y come from the
120  2op channel y+3 so the operatorss y, (9+y), y+3, (9+y)+3 make up a 4op
121  channel.
122 */
123 typedef struct operator_struct {
124  Bit32s cval, lastcval; // current output/last output (used for feedback)
125  Bit32u tcount, wfpos, tinc; // time (position in waveform) and time increment
126  fltype amp, step_amp; // and amplification (envelope)
127  fltype vol; // volume
128  fltype sustain_level; // sustain level
129  Bit32s mfbi; // feedback amount
130  fltype a0, a1, a2, a3; // attack rate function coefficients
131  fltype decaymul, releasemul; // decay/release rate functions
132  Bit32u op_state; // current state of operator (attack/decay/sustain/release/off)
133  Bit32u toff;
134  Bit32s freq_high; // highest three bits of the frequency, used for vibrato calculations
135  Bit16s* cur_wform; // start of selected waveform
136  Bit32u cur_wmask; // mask for selected waveform
137  Bit32u act_state; // activity state (regular, percussion)
138  bool sus_keep; // keep sustain level when decay finished
139  bool vibrato,tremolo; // vibrato/tremolo enable bits
140 
141  // variables used to provide non-continuous envelopes
142  Bit32u generator_pos; // for non-standard sample rates we need to determine how many samples have passed
143  Bits cur_env_step; // current (standardized) sample position
144  Bits env_step_a,env_step_d,env_step_r; // number of std samples of one step (for attack/decay/release mode)
145  Bit8u step_skip_pos_a; // position of 8-cyclic step skipping (always 2^x to check against mask)
146  Bits env_step_skip_a; // bitmask that determines if a step is skipped (respective bit is zero then)
147 
148  bool is_4op,is_4op_attached; // base of a 4op channel/part of a 4op channel
149  Bit32s left_pan,right_pan; // opl3 stereo panning amount
150 } op_type;
