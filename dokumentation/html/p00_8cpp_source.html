41 // File format from PC64. PC64 automatically generates
42 // the filename from the cbm name (16 to 8 conversion)
43 // but we only need to worry about that when writing files
44 // should we want pc64 compatibility. The extension numbers
45 // are just an index to try to avoid repeats. Name conversion
46 // works by creating an initial filename from alphanumeric
47 // and ' ', '-' characters only with the later two being
48 // converted to '_'. Then it parses the filename
49 // from end to start removing characters stopping as soon
50 // as the filename becomes <= 8. The removal of characters
51 // occurs in three passes, the first removes all '_', then
52 // vowels and finally numerics. If the filename is still
53 // greater than 8 it is truncated.
54 
55 struct X00Header

83 SidTuneBase* p00::load(const char *fileName, buffer_t& dataBuf)
84 {
85  const char *ext = SidTuneTools::fileExtOfPath(fileName);
86 
87  // Combined extension & magic field identification
88  if (strlen(ext) != 4)
89  return nullptr;
90 
91  if (!isdigit(ext[2]) || !isdigit(ext[3]))
92  return nullptr;
93 
94  const char *format = nullptr;
95  X00Format type;
96 
97  switch (toupper(ext[1]))
98  {
99  case 'D':
100  type = X00_DEL;
101  format = TXT_FORMAT_DEL;
102  break;
103  case 'S':
104  type = X00_SEQ;
105  format = TXT_FORMAT_SEQ;
106  break;
107  case 'P':
108  type = X00_PRG;
109  format = TXT_FORMAT_PRG;
110  break;
111  case 'U':
112  type = X00_USR;
113  format = TXT_FORMAT_USR;
114  break;
115  case 'R':
116  type = X00_REL;
117  format = TXT_FORMAT_REL;
118  break;
119  default:
120  return nullptr;
121  }
122 
123  // Verify the file is what we think it is
124  const buffer_t::size_type bufLen = dataBuf.size();
125  if (bufLen < X00_ID_LEN)
126  return nullptr;
127 
128  X00Header pHeader;
129  memcpy(pHeader.id, &dataBuf[0], X00_ID_LEN);
130  memcpy(pHeader.name, &dataBuf[X00_ID_LEN], X00_NAME_LEN);
131  pHeader.length = dataBuf[X00_ID_LEN + X00_NAME_LEN];
132 
133  if (strcmp(pHeader.id, P00_ID))
134  return nullptr;
135 
136  // File types current supported
137  if (type != X00_PRG)
138  throw loadError("Not a PRG inside X00");
139 
140  if (bufLen < sizeof(X00Header) + 2)
141  throw loadError(ERR_TRUNCATED);
142 
143  std::unique_ptr<p00> tune(new p00());
144  tune->load(format, &pHeader);
145 
146  return tune.release();
147 }
