116 void Player::initialise()
117 {
118  m_isPlaying = STOPPED;
119 
120  m_c64.reset();
121 
122  const SidTuneInfo* tuneInfo = m_tune->getInfo();
123 
124  const uint_least32_t size = static_cast<uint_least32_t>(tuneInfo->loadAddr()) + tuneInfo->c64dataLen() - 1;
125  if (size > 0xffff)
126  {
127  throw configError(ERR_UNSUPPORTED_SIZE);
128  }
129 
130  uint_least16_t powerOnDelay = m_cfg.powerOnDelay;
131  // Delays above MAX result in random delays
132  if (powerOnDelay > SidConfig::MAX_POWER_ON_DELAY)
133  { // Limit the delay to something sensible.
134  powerOnDelay = (uint_least16_t)((m_rand.next() >> 3) & SidConfig::MAX_POWER_ON_DELAY);
135  }
136 
137  psiddrv driver(m_tune->getInfo());
138  driver.powerOnDelay(powerOnDelay);
139  if (!driver.drvReloc())
140  {
141  throw configError(driver.errorString());
142  }
143 
144  m_info.m_driverAddr = driver.driverAddr();
145  m_info.m_driverLength = driver.driverLength();
146  m_info.m_powerOnDelay = powerOnDelay;
147 
148  driver.install(m_c64.getMemInterface(), videoSwitch);
149 
150  if (!m_tune->placeSidTuneInC64mem(m_c64.getMemInterface()))
151  {
152  throw configError(m_tune->statusString());
153  }
154 
155  m_c64.resetCpu();
156 }

191 uint_least32_t Player::play(short *buffer, uint_least32_t count)
192 {
193  // Make sure a tune is loaded
194  if (m_tune == nullptr)
195  return 0;
196 
197  // Start the player loop
198  if (m_isPlaying == STOPPED)
199  m_isPlaying = PLAYING;
200 
201  if (m_isPlaying == PLAYING)
202  {
203  m_mixer.begin(buffer, count);
204 
205  try
206  {
207  if (m_mixer.getSid(0) != nullptr)
208  {
209  if (count && buffer != nullptr)
210  {
211  // Clock chips and mix into output buffer
212  while (m_isPlaying && m_mixer.notFinished())
213  {
214  run(sidemu::OUTPUTBUFFERSIZE);
215 
216  m_mixer.clockChips();
217  m_mixer.doMix();
218  }
219  count = m_mixer.samplesGenerated();
220  }
221  else
222  {
223  // Clock chips and discard buffers
224  int size = m_c64.getMainCpuSpeed() / m_cfg.frequency;
225  while (m_isPlaying && --size)
226  {
227  run(sidemu::OUTPUTBUFFERSIZE);
228 
229  m_mixer.clockChips();
230  m_mixer.resetBufs();
231  }
232  }
233  }
234  else
235  {
236  // Clock the machine
237  int size = m_c64.getMainCpuSpeed() / m_cfg.frequency;
238  while (m_isPlaying && --size)
239  {
240  run(sidemu::OUTPUTBUFFERSIZE);
241  }
242  }
243  }
244  catch (MOS6510::haltInstruction const &)
245  {
246  m_errorString = "Illegal instruction executed";
247  m_isPlaying = STOPPING;
248  }
249  }
250 
251  if (m_isPlaying == STOPPING)
252  {
253  try
254  {
255  initialise();
256  }
257  catch (configError const &) {}
258  m_isPlaying = STOPPED;
259  }
260 
261  return count;
262 }

272 bool Player::config(const SidConfig &cfg, bool force)
273 {
274  // Check if configuration have been changed or forced
275  if (!force && !m_cfg.compare(cfg))
276  {
277  return true;
278  }
279 
280  // Check for base sampling frequency
281  if (cfg.frequency < 8000)
282  {
283  m_errorString = ERR_UNSUPPORTED_FREQ;
284  return false;
285  }
286 
287  // Only do these if we have a loaded tune
288  if (m_tune != nullptr)
289  {
290  const SidTuneInfo* tuneInfo = m_tune->getInfo();
291 
292  try
293  {
294  sidRelease();
295 
296  std::vector<unsigned int> addresses;
297  const uint_least16_t secondSidAddress = tuneInfo->sidChipBase(1) != 0 ?
298  tuneInfo->sidChipBase(1) :
299  cfg.secondSidAddress;
300  if (secondSidAddress != 0)
301  addresses.push_back(secondSidAddress);
302 
303  const uint_least16_t thirdSidAddress = tuneInfo->sidChipBase(2) != 0 ?
304  tuneInfo->sidChipBase(2) :
305  cfg.thirdSidAddress;
306  if (thirdSidAddress != 0)
307  addresses.push_back(thirdSidAddress);
308 
309  // SID emulation setup (must be performed before the
310  // environment setup call)
311  sidCreate(cfg.sidEmulation, cfg.defaultSidModel, cfg.digiBoost, cfg.forceSidModel, addresses);
312 
313  // Determine clock speed
314  const c64::model_t model = c64model(cfg.defaultC64Model, cfg.forceC64Model);
315 
316  m_c64.setModel(model);
317  m_c64.setCiaModel(cfg.ciaModel);
318 
319  sidParams(m_c64.getMainCpuSpeed(), cfg.frequency, cfg.samplingMethod, cfg.fastSampling);
320 
321  // Configure, setup and install C64 environment/events
322  initialise();
323  }
324  catch (configError const &e)
325  {
326  m_errorString = e.message();
327  m_cfg.sidEmulation = 0;
328  if (&m_cfg != &cfg)
329  {
330  config(m_cfg);
331  }
332  return false;
333  }
334  }
335 
336  const bool isStereo = cfg.playback == SidConfig::STEREO;
337  m_info.m_channels = isStereo ? 2 : 1;
338 
339  m_mixer.setStereo(isStereo);
340  m_mixer.setVolume(cfg.leftVolume, cfg.rightVolume);
341 
342  // Update Configuration
343  m_cfg = cfg;
344 
345  return true;
346 }

349 c64::model_t Player::c64model(SidConfig::c64_model_t defaultModel, bool forced)
350 {
351  const SidTuneInfo* tuneInfo = m_tune->getInfo();
352 
353  SidTuneInfo::clock_t clockSpeed = tuneInfo->clockSpeed();
354 
355  c64::model_t model;
356 
357  // Use preferred speed if forced or if song speed is unknown
358  if (forced || clockSpeed == SidTuneInfo::CLOCK_UNKNOWN || clockSpeed == SidTuneInfo::CLOCK_ANY)
359  {
360  switch (defaultModel)
361  {
362  case SidConfig::PAL:
363  clockSpeed = SidTuneInfo::CLOCK_PAL;
364  model = c64::PAL_B;
365  videoSwitch = 1;
366  break;
367  case SidConfig::DREAN:
368  clockSpeed = SidTuneInfo::CLOCK_PAL;
369  model = c64::PAL_N;
370  videoSwitch = 1; // TODO verify
371  break;
372  case SidConfig::NTSC:
373  clockSpeed = SidTuneInfo::CLOCK_NTSC;
374  model = c64::NTSC_M;
375  videoSwitch = 0;
376  break;
377  case SidConfig::OLD_NTSC:
378  clockSpeed = SidTuneInfo::CLOCK_NTSC;
379  model = c64::OLD_NTSC_M;
380  videoSwitch = 0;
381  break;
382  case SidConfig::PAL_M:
383  clockSpeed = SidTuneInfo::CLOCK_NTSC;
384  model = c64::PAL_M;
385  videoSwitch = 0; // TODO verify
386  break;
387  }
388  }
389  else
390  {
391  switch (clockSpeed)
392  {
393  default:
394  case SidTuneInfo::CLOCK_PAL:
395  model = c64::PAL_B;
396  videoSwitch = 1;
397  break;
398  case SidTuneInfo::CLOCK_NTSC:
399  model = c64::NTSC_M;
400  videoSwitch = 0;
401  break;
402  }
403  }
404 
405  switch (clockSpeed)
406  {
407  case SidTuneInfo::CLOCK_PAL:
408  if (tuneInfo->songSpeed() == SidTuneInfo::SPEED_CIA_1A)
409  m_info.m_speedString = TXT_PAL_CIA;
410  else if (tuneInfo->clockSpeed() == SidTuneInfo::CLOCK_NTSC)
411  m_info.m_speedString = TXT_PAL_VBI_FIXED;
412  else
413  m_info.m_speedString = TXT_PAL_VBI;
414  break;
415  case SidTuneInfo::CLOCK_NTSC:
416  if (tuneInfo->songSpeed() == SidTuneInfo::SPEED_CIA_1A)
417  m_info.m_speedString = TXT_NTSC_CIA;
418  else if (tuneInfo->clockSpeed() == SidTuneInfo::CLOCK_PAL)
419  m_info.m_speedString = TXT_NTSC_VBI_FIXED;
420  else
421  m_info.m_speedString = TXT_NTSC_VBI;
422  break;
423  default:
424  break;
425  }
426

492 void Player::sidCreate(sidbuilder *builder, SidConfig::sid_model_t defaultModel, bool digiboost,
493  bool forced, const std::vector<unsigned int> &extraSidAddresses)
494 {
495  if (builder != nullptr)
496  {
497  const SidTuneInfo* tuneInfo = m_tune->getInfo();
498 
499  // Setup base SID
500  const SidConfig::sid_model_t userModel = getSidModel(tuneInfo->sidModel(0), defaultModel, forced);
501  sidemu *s = builder->lock(m_c64.getEventScheduler(), userModel, digiboost);
502  if (!builder->getStatus())
503  {
504  throw configError(builder->error());
505  }
506 
507  m_c64.setBaseSid(s);
508  m_mixer.addSid(s);
509 
510  // Setup extra SIDs if needed
511  if (extraSidAddresses.size() != 0)
512  {
513  // If bits 6-7 are set to Unknown then the second SID will be set to the same SID
514  // model as the first SID.
515  defaultModel = userModel;
516 
517  const unsigned int extraSidChips = extraSidAddresses.size();
518 
519  for (unsigned int i = 0; i < extraSidChips; i++)
520  {
521  const SidConfig::sid_model_t userModel = getSidModel(tuneInfo->sidModel(i+1), defaultModel, forced);
522 
523  sidemu *s = builder->lock(m_c64.getEventScheduler(), userModel, digiboost);
524  if (!builder->getStatus())
525  {
526  throw configError(builder->error());
527  }
528 
529  if (!m_c64.addExtraSid(s, extraSidAddresses[i]))
530  throw configError(ERR_UNSUPPORTED_SID_ADDR);
531 
532  m_mixer.addSid(s);
533  }
534  }
535  }
