64 void copyPoweronPattern(sidmemory& mem)
65 {
66  uint_least16_t addr = 0;
67  for (unsigned int i = 0; i < sizeof(POWERON);)
68  {
69  uint8_t off = POWERON[i++];
70  uint8_t count = 0;
71  bool compressed = false;
72 
73  // Determine data count/compression
74  if (off & 0x80)
75  {
76  // fixup offset
77  off &= 0x7f;
78  count = POWERON[i++];
79  if (count & 0x80)
80  {
81  // fixup count
82  count &= 0x7f;
83  compressed = true;
84  }
85  }
86 
87  // Fix count off by ones (see format details)
88  count++;
89  addr += off;
90 
91  if (compressed)
92  {
93  // Extract compressed data
94  const uint8_t data = POWERON[i++];
95  while (count-- > 0)
96  {
97  mem.writeMemByte(addr++, data);
98  }
99  }
100  else
101  {
102  // Extract uncompressed data
103  while (count-- > 0)
104  {
105  mem.writeMemByte(addr++, POWERON[i++]);
106  }
107  }
108  }
109 }

111 uint8_t psiddrv::iomap(uint_least16_t addr) const
112 {
113  // Force Real C64 Compatibility
114  if (m_tuneInfo->compatibility() == SidTuneInfo::COMPATIBILITY_R64
115  || m_tuneInfo->compatibility() == SidTuneInfo::COMPATIBILITY_BASIC
116  || addr == 0)
117  {
118  // Special case, set to 0x37 by the psid driver
119  return 0;
120  }
121 
122  /*
123  * $34 for init/play in $d000 - $dfff
124  * $35 for init/play in $e000 - $ffff
125  * $36 for load end/play in $a000 - $ffff
126  * $37 for the rest
127  */
128  if (addr < 0xa000)
129  return 0x37; // Basic-ROM, Kernal-ROM, I/O
130  if (addr < 0xd000)
131  return 0x36; // Kernal-ROM, I/O
132  if (addr >= 0xe000)
133  return 0x35; // I/O only
134 
135  return 0x34; // RAM only
136 }

138 bool psiddrv::drvReloc()
139 {
140  const int startlp = m_tuneInfo->loadAddr() >> 8;
141  const int endlp = (m_tuneInfo->loadAddr() + (m_tuneInfo->c64dataLen() - 1)) >> 8;
142 
143  uint_least8_t relocStartPage = m_tuneInfo->relocStartPage();
144  uint_least8_t relocPages = m_tuneInfo->relocPages();
145 
146  if (m_tuneInfo->compatibility() == SidTuneInfo::COMPATIBILITY_BASIC)
147  {
148  // The psiddrv is only used for initialisation and to
149  // autorun basic tunes as running the kernel falls
150  // into a manual load/run mode
151  relocStartPage = 0x04;
152  relocPages = 0x03;
153  }
154 
155  // Check for free space in tune
156  if (relocStartPage == 0xff)
157  relocPages = 0;
158  // Check if we need to find the reloc addr
159  else if (relocStartPage == 0)
160  {
161  relocPages = 0;
162  // find area where to dump the driver in.
163  // It's only 1 block long, so any free block we can find
164  // between $0400 and $d000 will do.
165  for (int i = 4; i < 0xd0; i ++)
166  {
167  if (i >= startlp && i <= endlp)
168  continue;
169 
170  if (i >= 0xa0 && i <= 0xbf)
171  continue;
172 
173  relocStartPage = i;
174  relocPages = 1;
175  break;
176  }
177  }
178 
179  if (relocPages < 1)
180  {
181  m_errorString = ERR_PSIDDRV_NO_SPACE;
182  return false;
183  }
184 
185  // Place psid driver into ram
186  const uint_least16_t relocAddr = relocStartPage << 8;
187 
188  reloc_driver = psid_driver;
189  reloc_size = sizeof(psid_driver);
190 
191  reloc65 relocator;
192  relocator.setReloc(reloc65::TEXT, relocAddr - 10);
193  relocator.setExtract(reloc65::TEXT);
194  if (!relocator.reloc(&reloc_driver, &reloc_size))
195  {
196  m_errorString = ERR_PSIDDRV_RELOC;
197  return false;
198  }
199 
200  // Adjust size to not included initialisation data.
201  reloc_size -= 10;
202 
203  m_driverAddr = relocAddr;
204  m_driverLength = static_cast<uint_least16_t>(reloc_size);
205  // Round length to end of page
206  m_driverLength += 0xff;
207  m_driverLength &= 0xff00;
208 
209  return true;
210 }

212 void psiddrv::install(sidmemory& mem, uint8_t video) const
213 {
214  mem.fillRam(0, static_cast<uint8_t>(0), 0x3ff);
215 
216  if (m_tuneInfo->compatibility() >= SidTuneInfo::COMPATIBILITY_R64)
217  {
218  copyPoweronPattern(mem);
219  }
220 
221  // Set PAL/NTSC switch
222  mem.writeMemByte(0x02a6, video);
223 
224  mem.installResetHook(endian_little16(reloc_driver));
225 
226  // If not a basic tune then the psiddrv must install
227  // interrupt hooks and trap programs trying to restart basic
228  if (m_tuneInfo->compatibility() == SidTuneInfo::COMPATIBILITY_BASIC)
229  {
230  // Install hook to set subtune number for basic
231  mem.setBasicSubtune((uint8_t)(m_tuneInfo->currentSong() - 1));
232  mem.installBasicTrap(0xbf53);
233  }
234  else
235  {
236  // Only install irq handle for RSID tunes
237  mem.fillRam(0x0314, &reloc_driver[2], m_tuneInfo->compatibility() == SidTuneInfo::COMPATIBILITY_R64 ? 2 : 6);
238 
239  // Experimental restart basic trap
240  const uint_least16_t addr = endian_little16(&reloc_driver[8]);
241  mem.installBasicTrap(0xffe1);
242  mem.writeMemWord(0x0328, addr);
243  }
244 
245  int pos = m_driverAddr;
246 
247  // Install driver to ram
248  mem.fillRam(pos, &reloc_driver[10], reloc_size);
249 
250  // Set song number
251  mem.writeMemByte(pos, (uint8_t) (m_tuneInfo->currentSong() - 1));
252  pos++;
253 
254  // Set tunes speed (VIC/CIA)
255  mem.writeMemByte(pos, m_tuneInfo->songSpeed() == SidTuneInfo::SPEED_VBI ? 0 : 1);
256  pos++;
257 
258  // Set init address
259  mem.writeMemWord(pos, m_tuneInfo->compatibility() == SidTuneInfo::COMPATIBILITY_BASIC ?
260  0xbf55 : m_tuneInfo->initAddr());
261  pos += 2;
262 
263  // Set play address
264  mem.writeMemWord(pos, m_tuneInfo->playAddr());
265  pos += 2;
266 
267  mem.writeMemWord(pos, m_powerOnDelay);
268  pos += 2;
269 
270  // Set init address io bank value
271  mem.writeMemByte(pos, iomap(m_tuneInfo->initAddr()));
272  pos++;
273 
274  // Set play address io bank value
275  mem.writeMemByte(pos, iomap(m_tuneInfo->playAddr()));
276  pos++;
277 
278  // Set PAL/NTSC flag
279  mem.writeMemByte(pos, video);
280  pos++;
281 
282  // Set the required tune clock speed
283  uint8_t clockSpeed;
284  switch (m_tuneInfo->clockSpeed())
285  {
286  case SidTuneInfo::CLOCK_PAL:
287  clockSpeed = 1;
288  break;
289  case SidTuneInfo::CLOCK_NTSC:
290  clockSpeed = 0;
291  break;
292  default: // UNKNOWN or ANY
293  clockSpeed = video;
294  break;
295  }
296  mem.writeMemByte(pos, clockSpeed);
297  pos++;
298 
299  // Set default processor register flags on calling init
300  mem.writeMemByte(pos, m_tuneInfo->compatibility() >= SidTuneInfo::COMPATIBILITY_R64 ? 0 : 1 << MOS6510::SR_INTERRUPT);
301 }
