136 bool reloc65::reloc(unsigned char **buf, int *fsize)
137 {
138  unsigned char *tmpBuf = *buf;
139 
140  if (memcmp(tmpBuf, o65hdr, 5) != 0)
141  {
142  return false;
143  }
144 
145  const int mode = getWord(tmpBuf, 6);
146  if ((mode & 0x2000) // 32 bit size not supported
147  || (mode & 0x4000)) // pagewise relocation not supported
148  {
149  return false;
150  }
151 
152  const int hlen = HEADER_SIZE + read_options(tmpBuf + HEADER_SIZE);
153 
154  const int tbase = getWord(tmpBuf, 8);
155  const int tlen = getWord(tmpBuf, 10);
156  m_tdiff = m_tflag ? m_tbase - tbase : 0;
157 
158  const int dbase = getWord(tmpBuf, 12);
159  const int dlen = getWord(tmpBuf, 14);
160  m_ddiff = m_dflag ? m_dbase - dbase : 0;
161 
162  const int bbase = getWord(tmpBuf, 16);
163  const int blen SID_UNUSED = getWord(tmpBuf, 18);
164  m_bdiff = m_bflag ? m_bbase - bbase : 0;
165 
166  const int zbase = getWord(tmpBuf, 20);
167  const int zlen SID_UNUSED = getWord(tmpBuf, 21);
168  m_zdiff = m_zflag ? m_zbase - zbase : 0;
169 
170  unsigned char *segt = tmpBuf + hlen; // Text segment
171  unsigned char *segd = segt + tlen; // Data segment
172  unsigned char *utab = segd + dlen; // Undefined references list
173 
174  unsigned char *rttab = utab + read_undef(utab); // Text relocation table
175 
176  unsigned char *rdtab = reloc_seg(segt, tlen, rttab); // Data relocation table
177  unsigned char *extab = reloc_seg(segd, dlen, rdtab); // Exported globals list
178 
179  reloc_globals(extab);
180 
181  if (m_tflag)
182  {
183  setWord(tmpBuf, 8, m_tbase);
184  }
185  if (m_dflag)
186  {
187  setWord(tmpBuf, 12, m_dbase);
188  }
189  if (m_bflag)
190  {
191  setWord(tmpBuf, 16, m_bbase);
192  }
193  if (m_zflag)
194  {
195  setWord(tmpBuf, 20, m_zbase);
196  }
197 
198  switch (m_extract)
199  {
200  case WHOLE:
201  return true;
202  case TEXT:
203  *buf = segt;
204  *fsize = tlen;
205  return true;
206  case DATA:
207  *buf = segd;
208  *fsize = dlen;
209  return true;
210  default:
211  return false;
212  }
213 }

227 unsigned char* reloc65::reloc_seg(unsigned char *buf, int len, unsigned char *rtab)
228 {
229  int adr = -1;
230  while (*rtab)
231  {
232  if ((*rtab & 255) == 255)
233  {
234  adr += 254;
235  rtab++;
236  }
237  else
238  {
239  adr += *rtab & 255;
240  rtab++;
241  const unsigned char type = *rtab & 0xe0;
242  unsigned char seg = *rtab & 0x07;
243  rtab++;
244  switch(type)
245  {
246  case 0x80: {
247  const int oldVal = getWord(buf, adr);
248  const int newVal = oldVal + reldiff(seg);
249  setWord(buf, adr, newVal);
250  break; }
251  case 0x40: {
252  const int oldVal = buf[adr] * 256 + *rtab;
253  const int newVal = oldVal + reldiff(seg);
254  buf[adr] = (newVal >> 8) & 255;
255  *rtab = newVal & 255;
256  rtab++;
257  break; }
258  case 0x20: {
259  const int oldVal = buf[adr];
260  const int newVal = oldVal + reldiff(seg);
261  buf[adr] = newVal & 255;
262  break; }
263  }
264  if (seg == 0)
265  {
266  rtab += 2;
267  }
268  }
269 
270  if (adr > len)
271  {
272  // Warning: relocation table entries past segment end!
273  }
274  }
275 
276  return ++rtab;
277 }
