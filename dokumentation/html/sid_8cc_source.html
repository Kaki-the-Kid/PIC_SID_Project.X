131 // ----------------------------------------------------------------------------
132 // Read registers.
133 //
134 // Reading a write only register returns the last byte written to any SID
135 // register. The individual bits in this value start to fade down towards
136 // zero after a few cycles. All bits reach zero within approximately
137 // $2000 - $4000 cycles.
138 // It has been claimed that this fading happens in an orderly fashion, however
139 // sampling of write only registers reveals that this is not the case.
140 // NB! This is not correctly modeled.
141 // The actual use of write only registers has largely been made in the belief
142 // that all SID registers are readable. To support this belief the read
143 // would have to be done immediately after a write to the same register
144 // (remember that an intermediate write to another register would yield that
145 // value instead). With this in mind we return the last value written to
146 // any SID register for $4000 cycles without modeling the bit fading.
147 // ----------------------------------------------------------------------------
148 reg8 SID::read(reg8 offset)
149 {
150  switch (offset) {
151  case 0x19:
152  bus_value = potx.readPOT();
153  bus_value_ttl = databus_ttl;
154  break;
155  case 0x1a:
156  bus_value = poty.readPOT();
157  bus_value_ttl = databus_ttl;
158  break;
159  case 0x1b:
160  bus_value = voice[2].wave.readOSC();
161  bus_value_ttl = databus_ttl;
162  break;
163  case 0x1c:
164  bus_value = voice[2].envelope.readENV();
165  bus_value_ttl = databus_ttl;
166  break;
167  }
168  return bus_value;
169 }

322 // ----------------------------------------------------------------------------
323 // Read state.
324 // ----------------------------------------------------------------------------
325 SID::State SID::read_state()
326 {
327  State state;
328  int i, j;
329 
330  for (i = 0, j = 0; i < 3; i++, j += 7) {
331  WaveformGenerator& wave = voice[i].wave;
332  EnvelopeGenerator& envelope = voice[i].envelope;
333  state.sid_register[j + 0] = wave.freq & 0xff;
334  state.sid_register[j + 1] = wave.freq >> 8;
335  state.sid_register[j + 2] = wave.pw & 0xff;
336  state.sid_register[j + 3] = wave.pw >> 8;
337  state.sid_register[j + 4] =
338  (wave.waveform << 4)
339  | (wave.test ? 0x08 : 0)
340  | (wave.ring_mod ? 0x04 : 0)
341  | (wave.sync ? 0x02 : 0)
342  | (envelope.gate ? 0x01 : 0);
343  state.sid_register[j + 5] = (envelope.attack << 4) | envelope.decay;
344  state.sid_register[j + 6] = (envelope.sustain << 4) | envelope.release;
345  }
346 
347  state.sid_register[j++] = filter.fc & 0x007;
348  state.sid_register[j++] = filter.fc >> 3;
349  state.sid_register[j++] = (filter.res << 4) | filter.filt;
350  state.sid_register[j++] = filter.mode | filter.vol;
351 
352  // These registers are superfluous, but are included for completeness.
353  for (; j < 0x1d; j++) {
354  state.sid_register[j] = read(j);
355  }
356  for (; j < 0x20; j++) {
357  state.sid_register[j] = 0;
358  }
359 
360  state.bus_value = bus_value;
361  state.bus_value_ttl = bus_value_ttl;
362  state.write_pipeline = write_pipeline;
363  state.write_address = write_address;
364  state.voice_mask = filter.voice_mask;
365 
366  for (i = 0; i < 3; i++) {
367  state.accumulator[i] = voice[i].wave.accumulator;
368  state.shift_register[i] = voice[i].wave.shift_register;
369  state.shift_register_reset[i] = voice[i].wave.shift_register_reset;
370  state.shift_pipeline[i] = voice[i].wave.shift_pipeline;
371  state.pulse_output[i] = voice[i].wave.pulse_output;
372  state.floating_output_ttl[i] = voice[i].wave.floating_output_ttl;
373 
374  state.rate_counter[i] = voice[i].envelope.rate_counter;
375  state.rate_counter_period[i] = voice[i].envelope.rate_period;
376  state.exponential_counter[i] = voice[i].envelope.exponential_counter;
377  state.exponential_counter_period[i] = voice[i].envelope.exponential_counter_period;
378  state.envelope_counter[i] = voice[i].envelope.envelope_counter;
379  state.envelope_state[i] = voice[i].envelope.state;
380  state.hold_zero[i] = voice[i].envelope.hold_zero;
381  state.envelope_pipeline[i] = voice[i].envelope.envelope_pipeline;
382  }
383 
384  return state;
385 }

465 // ----------------------------------------------------------------------------
466 // I0() computes the 0th order modified Bessel function of the first kind.
467 // This function is originally from resample-1.5/filterkit.c by J. O. Smith.
468 // ----------------------------------------------------------------------------
469 double SID::I0(double x)
470 {
471  // Max error acceptable in I0.
472  const double I0e = 1e-6;
473 
474  double sum, u, halfx, temp;
475  int n;
476 
477  sum = u = n = 1;
478  halfx = x/2.0;
479 
480  do {
481  temp = halfx/n++;
482  u *= temp*temp;
483  sum += u;
484  } while (u >= I0e*sum);
485 
486  return sum;
487 }

490 // ----------------------------------------------------------------------------
491 // Setting of SID sampling parameters.
492 //
493 // Use a clock freqency of 985248Hz for PAL C64, 1022730Hz for NTSC C64.
494 // The default end of passband frequency is pass_freq = 0.9*sample_freq/2
495 // for sample frequencies up to ~ 44.1kHz, and 20kHz for higher sample
496 // frequencies.
497 //
498 // For resampling, the ratio between the clock frequency and the sample
499 // frequency is limited as follows:
500 // 125*clock_freq/sample_freq < 16384
501 // E.g. provided a clock frequency of ~ 1MHz, the sample frequency can not
502 // be set lower than ~ 8kHz. A lower sample frequency would make the
503 // resampling code overfill its 16k sample ring buffer.
504 //
505 // The end of passband frequency is also limited:
506 // pass_freq <= 0.9*sample_freq/2
507 
508 // E.g. for a 44.1kHz sampling rate the end of passband frequency is limited
509 // to slightly below 20kHz. This constraint ensures that the FIR table is
510 // not overfilled.
511 // ----------------------------------------------------------------------------
512 bool SID::set_sampling_parameters(double clock_freq, sampling_method method,
513  double sample_freq, double pass_freq, double filter_scale)
514 {
515  // Check resampling constraints.
516  if (method == SAMPLE_RESAMPLE || method == SAMPLE_RESAMPLE_FASTMEM)
517  {
518  // Check whether the sample ring buffer would overfill.
519  if (FIR_N*clock_freq/sample_freq >= RINGSIZE) {
520  return false;
521  }
522 
523  // The default passband limit is 0.9*sample_freq/2 for sample
524  // frequencies below ~ 44.1kHz, and 20kHz for higher sample frequencies.
525  if (pass_freq < 0) {
526  pass_freq = 20000;
527  if (2*pass_freq/sample_freq >= 0.9) {
528  pass_freq = 0.9*sample_freq/2;
529  }
530  }
531  // Check whether the FIR table would overfill.
532  else if (pass_freq > 0.9*sample_freq/2) {
533  return false;
534  }
535 
536  // The filter scaling is only included to avoid clipping, so keep
537  // it sane.
538  if (filter_scale < 0.9 || filter_scale > 1.0) {
539  return false;
540  }
541  }
542 
543  clock_frequency = clock_freq;
544  sampling = method;
545 
546  cycles_per_sample =
547  cycle_count(clock_freq/sample_freq*(1 << FIXP_SHIFT) + 0.5);
548 
549  sample_offset = 0;
550  sample_prev = 0;
551  sample_now = 0;
552 
553  // FIR initialization is only necessary for resampling.
554  if (method != SAMPLE_RESAMPLE && method != SAMPLE_RESAMPLE_FASTMEM)
555  {
556  delete[] sample;
557  delete[] fir;
558  sample = 0;
559  fir = 0;
560  return true;
561  }
562 
563  // Allocate sample buffer.
564  if (!sample) {
565  sample = new short[RINGSIZE*2];
566  }
567  // Clear sample buffer.
568  for (int j = 0; j < RINGSIZE*2; j++) {
569  sample[j] = 0;
570  }
571  sample_index = 0;
572 
573  const double pi = 3.1415926535897932385;
574 
575  // 16 bits -> -96dB stopband attenuation.
576  const double A = -20*log10(1.0/(1 << 16));
577  // A fraction of the bandwidth is allocated to the transition band,
578  double dw = (1 - 2*pass_freq/sample_freq)*pi*2;
579  // The cutoff frequency is midway through the transition band (nyquist)
580  double wc = pi;
581 
582  // For calculation of beta and N see the reference for the kaiserord
583  // function in the MATLAB Signal Processing Toolbox:
584  // http://www.mathworks.com/access/helpdesk/help/toolbox/signal/kaiserord.html
585  const double beta = 0.1102*(A - 8.7);
586  const double I0beta = I0(beta);
587 
588  // The filter order will maximally be 124 with the current constraints.
589  // N >= (96.33 - 7.95)/(2.285*0.1*pi) -> N >= 123
590  // The filter order is equal to the number of zero crossings, i.e.
591  // it should be an even number (sinc is symmetric about x = 0).
592  int N = int((A - 7.95)/(2.285*dw) + 0.5);
593  N += N & 1;
594 
595  double f_samples_per_cycle = sample_freq/clock_freq;
596  double f_cycles_per_sample = clock_freq/sample_freq;
597 
598  // The filter length is equal to the filter order + 1.
599  // The filter length must be an odd number (sinc is symmetric about x = 0).
600  int fir_N_new = int(N*f_cycles_per_sample) + 1;
601  fir_N_new |= 1;
602 
603  // We clamp the filter table resolution to 2^n, making the fixed point
604  // sample_offset a whole multiple of the filter table resolution.
605  int res = method == SAMPLE_RESAMPLE ?
606  FIR_RES : FIR_RES_FASTMEM;
607  int n = (int)ceil(log(res/f_cycles_per_sample)/log(2.0f));
608  int fir_RES_new = 1 << n;
609 
610  /* Determine if we need to recalculate table, or whether we can reuse earlier cached copy.
611  * This pays off on slow hardware such as current Android devices.
612  */
613  if (fir && fir_RES_new == fir_RES && fir_N_new == fir_N && beta == fir_beta && f_cycles_per_sample == fir_f_cycles_per_sample && fir_filter_scale == filter_scale) {
614  return true;
615  }
616  fir_RES = fir_RES_new;
617  fir_N = fir_N_new;
618  fir_beta = beta;
619  fir_f_cycles_per_sample = f_cycles_per_sample;
620  fir_filter_scale = filter_scale;
621 
622  // Allocate memory for FIR tables.
623  delete[] fir;
624  fir = new short[fir_N*fir_RES];
625 
626  // Calculate fir_RES FIR tables for linear interpolation.
627  for (int i = 0; i < fir_RES; i++) {
628  int fir_offset = i*fir_N + fir_N/2;
629  double j_offset = double(i)/fir_RES;
630  // Calculate FIR table. This is the sinc function, weighted by the
631  // Kaiser window.
632  for (int j = -fir_N/2; j <= fir_N/2; j++) {
633  double jx = j - j_offset;
634  double wt = wc*jx/f_cycles_per_sample;
635  double temp = jx/(fir_N/2);
636  double Kaiser = fabs(temp) <= 1 ? I0(beta*sqrt(1 - temp*temp))/I0beta : 0;
637  double sincwt = fabs(wt) >= 1e-6 ? sin(wt)/wt : 1;
638  double val = (1 << FIR_SHIFT)*filter_scale*f_samples_per_cycle*wc/pi*sincwt*Kaiser;
639  fir[fir_offset + j] = (short)round(val);
640  }
641  }
642 
643  return true;
644 }

647 // ----------------------------------------------------------------------------
648 // Adjustment of SID sampling frequency.
649 //
650 // In some applications, e.g. a C64 emulator, it can be desirable to
651 // synchronize sound with a timer source. This is supported by adjustment of
652 // the SID sampling frequency.
653 //
654 // NB! Adjustment of the sampling frequency may lead to noticeable shifts in
655 // frequency, and should only be used for interactive applications. Note also
656 // that any adjustment of the sampling frequency will change the
657 // characteristics of the resampling filter, since the filter is not rebuilt.
658 // ----------------------------------------------------------------------------
659 void SID::adjust_sampling_frequency(double sample_freq)
660 {
661  cycles_per_sample =
662  cycle_count(clock_frequency/sample_freq*(1 << FIXP_SHIFT) + 0.5);
663 }

666 // ----------------------------------------------------------------------------
667 // SID clocking - delta_t cycles.
668 // ----------------------------------------------------------------------------
669 void SID::clock(cycle_count delta_t)
670 {
671  int i;
672 
673  // Pipelined writes on the MOS8580.
674  if (unlikely(write_pipeline) && likely(delta_t > 0)) {
675  // Step one cycle by a recursive call to ourselves.
676  write_pipeline = 0;
677  clock(1);
678  write();
679  delta_t -= 1;
680  }
681 
682  if (unlikely(delta_t <= 0)) {
683  return;
684  }
685 
686  // Age bus value.
687  bus_value_ttl -= delta_t;
688  if (unlikely(bus_value_ttl <= 0)) {
689  bus_value = 0;
690  bus_value_ttl = 0;
691  }
692 
693  // Clock amplitude modulators.
694  for (i = 0; i < 3; i++) {
695  voice[i].envelope.clock(delta_t);
696  }
697 
698  // Clock and synchronize oscillators.
699  // Loop until we reach the current cycle.
700  cycle_count delta_t_osc = delta_t;
701  while (delta_t_osc) {
702  cycle_count delta_t_min = delta_t_osc;
703 
704  // Find minimum number of cycles to an oscillator accumulator MSB toggle.
705  // We have to clock on each MSB on / MSB off for hard sync to operate
706  // correctly.
707  for (i = 0; i < 3; i++) {
708  WaveformGenerator& wave = voice[i].wave;
709 
710  // It is only necessary to clock on the MSB of an oscillator that is
711  // a sync source and has freq != 0.
712  if (likely(!(wave.sync_dest->sync && wave.freq))) {
713  continue;
714  }
715 
716  reg16 freq = wave.freq;
717  reg24 accumulator = wave.accumulator;
718 
719  // Clock on MSB off if MSB is on, clock on MSB on if MSB is off.
720  reg24 delta_accumulator =
721  (accumulator & 0x800000 ? 0x1000000 : 0x800000) - accumulator;
722 
723  cycle_count delta_t_next = delta_accumulator/freq;
724  if (likely(delta_accumulator%freq)) {
725  ++delta_t_next;
726  }
727 
728  if (unlikely(delta_t_next < delta_t_min)) {
729  delta_t_min = delta_t_next;
730  }
731  }
732 
733  // Clock oscillators.
734  for (i = 0; i < 3; i++) {
735  voice[i].wave.clock(delta_t_min);
736  }
737 
738  // Synchronize oscillators.
739  for (i = 0; i < 3; i++) {
740  voice[i].wave.synchronize();
741  }
742 
743  delta_t_osc -= delta_t_min;
744  }
745 
746  // Calculate waveform output.
747  for (i = 0; i < 3; i++) {
748  voice[i].wave.set_waveform_output(delta_t);
749  }
750 
751  // Clock filter.
752  filter.clock(delta_t, voice[0].output(), voice[1].output(), voice[2].output());
753 
754  // Clock external filter.
755  extfilt.clock(delta_t, filter.output());
756 }

789 // ----------------------------------------------------------------------------
790 // SID clocking with audio sampling - delta clocking picking nearest sample.
791 // ----------------------------------------------------------------------------
792 int SID::clock_fast(cycle_count& delta_t, short* buf, int n, int interleave)
793 {
794  int s;
795 
796  for (s = 0; s < n; s++) {
797  cycle_count next_sample_offset = sample_offset + cycles_per_sample + (1 << (FIXP_SHIFT - 1));
798  cycle_count delta_t_sample = next_sample_offset >> FIXP_SHIFT;
799 
800  if (delta_t_sample > delta_t) {
801  delta_t_sample = delta_t;
802  }
803 
804  clock(delta_t_sample);
805 
806  if ((delta_t -= delta_t_sample) == 0) {
807  sample_offset -= delta_t_sample << FIXP_SHIFT;
808  break;
809  }
810 
811  sample_offset = (next_sample_offset & FIXP_MASK) - (1 << (FIXP_SHIFT - 1));
812  buf[s*interleave] = output();
813  }
814 
815  return s;
816 }

819 // ----------------------------------------------------------------------------
820 // SID clocking with audio sampling - cycle based with linear sample
821 // interpolation.
822 //
823 // Here the chip is clocked every cycle. This yields higher quality
824 // sound since the samples are linearly interpolated, and since the
825 // external filter attenuates frequencies above 16kHz, thus reducing
826 // sampling noise.
827 // ----------------------------------------------------------------------------
828 int SID::clock_interpolate(cycle_count& delta_t, short* buf, int n, int interleave)
829 {
830  int s;
831 
832  for (s = 0; s < n; s++) {
833  cycle_count next_sample_offset = sample_offset + cycles_per_sample;
834  cycle_count delta_t_sample = next_sample_offset >> FIXP_SHIFT;
835 
836  if (delta_t_sample > delta_t) {
837  delta_t_sample = delta_t;
838  }
839 
840  for (int i = delta_t_sample; i > 0; i--) {
841  clock();
842  if (unlikely(i <= 2)) {
843  sample_prev = sample_now;
844  sample_now = output();
845  }
846  }
847 
848  if ((delta_t -= delta_t_sample) == 0) {
849  sample_offset -= delta_t_sample << FIXP_SHIFT;
850  break;
851  }
852 
853  sample_offset = next_sample_offset & FIXP_MASK;
854 
855  buf[s*interleave] =
856  sample_prev + (sample_offset*(sample_now - sample_prev) >> FIXP_SHIFT);
857  }
858 
859  return s;
860 }

863 // ----------------------------------------------------------------------------
864 // SID clocking with audio sampling - cycle based with audio resampling.
865 //
866 // This is the theoretically correct (and computationally intensive) audio
867 // sample generation. The samples are generated by resampling to the specified
868 // sampling frequency. The work rate is inversely proportional to the
869 // percentage of the bandwidth allocated to the filter transition band.
870 //
871 // This implementation is based on the paper "A Flexible Sampling-Rate
872 // Conversion Method", by J. O. Smith and P. Gosset, or rather on the
873 // expanded tutorial on the "Digital Audio Resampling Home Page":
874 // http://www-ccrma.stanford.edu/~jos/resample/
875 //
876 // By building shifted FIR tables with samples according to the
877 // sampling frequency, the implementation below dramatically reduces the
878 // computational effort in the filter convolutions, without any loss
879 // of accuracy. The filter convolutions are also vectorizable on
880 // current hardware.
881 //
882 // Further possible optimizations are:
883 // * An equiripple filter design could yield a lower filter order, see
884 // http://www.mwrf.com/Articles/ArticleID/7229/7229.html
885 // * The Convolution Theorem could be used to bring the complexity of
886 // convolution down from O(n*n) to O(n*log(n)) using the Fast Fourier
887 // Transform, see http://en.wikipedia.org/wiki/Convolution_theorem
888 // * Simply resampling in two steps can also yield computational
889 // savings, since the transition band will be wider in the first step
890 // and the required filter order is thus lower in this step.
891 // Laurent Ganier has found the optimal intermediate sampling frequency
892 // to be (via derivation of sum of two steps):
893 // 2 * pass_freq + sqrt [ 2 * pass_freq * orig_sample_freq
894 // * (dest_sample_freq - 2 * pass_freq) / dest_sample_freq ]
895 //
896 // NB! the result of right shifting negative numbers is really
897 // implementation dependent in the C++ standard.
898 // ----------------------------------------------------------------------------
899 int SID::clock_resample(cycle_count& delta_t, short* buf, int n, int interleave)
900 {
901  int s;
902 
903  for (s = 0; s < n; s++) {
904  cycle_count next_sample_offset = sample_offset + cycles_per_sample;
905  cycle_count delta_t_sample = next_sample_offset >> FIXP_SHIFT;
906 
907  if (delta_t_sample > delta_t) {
908  delta_t_sample = delta_t;
909  }
910 
911  for (int i = 0; i < delta_t_sample; i++) {
912  clock();
913  sample[sample_index] = sample[sample_index + RINGSIZE] = output();
914  ++sample_index &= RINGMASK;
915  }
916 
917  if ((delta_t -= delta_t_sample) == 0) {
918  sample_offset -= delta_t_sample << FIXP_SHIFT;
919  break;
920  }
921 
922  sample_offset = next_sample_offset & FIXP_MASK;
923 
924  int fir_offset = sample_offset*fir_RES >> FIXP_SHIFT;
925  int fir_offset_rmd = sample_offset*fir_RES & FIXP_MASK;
926  short* fir_start = fir + fir_offset*fir_N;
927  short* sample_start = sample + sample_index - fir_N - 1 + RINGSIZE;
928 
929  // Convolution with filter impulse response.
930  int v1 = 0;
931  for (int j = 0; j < fir_N; j++) {
932  v1 += sample_start[j]*fir_start[j];
933  }
934 
935  // Use next FIR table, wrap around to first FIR table using
936  // next sample.
937  if (unlikely(++fir_offset == fir_RES)) {
938  fir_offset = 0;
939  ++sample_start;
940  }
941  fir_start = fir + fir_offset*fir_N;
942 
943  // Convolution with filter impulse response.
944  int v2 = 0;
945  for (int k = 0; k < fir_N; k++) {
946  v2 += sample_start[k]*fir_start[k];
947  }
948 
949  // Linear interpolation.
950  // fir_offset_rmd is equal for all samples, it can thus be factorized out:
951  // sum(v1 + rmd*(v2 - v1)) = sum(v1) + rmd*(sum(v2) - sum(v1))
952  int v = v1 + int((unsigned(fir_offset_rmd)*unsigned(v2 - v1)) >> FIXP_SHIFT);
953 
954  v >>= FIR_SHIFT;
955 
956  // Saturated arithmetics to guard against 16 bit sample overflow.
957  const int half = 1 << 15;
958  if (v >= half) {
959  v = half - 1;
960  }
961  else if (v < -half) {
962  v = -half;
963  }
964 
965  buf[s*interleave] = v;
966  }
967 
968  return s;
969 }
