27 /* A complete hash table is an instance of the following structure.
28  * The internals of this structure are intended to be opaque -- client
29  * code should not attempt to access or modify the fields of this structure
30  * directly. Change this structure only by using the routines below.
31  * However, many of the "procedures" and "functions" for modifying and
32  * accessing this structure are really macros, so we can't really make
33  * this structure opaque.
34  */

51 /* Each element in the hash table is an instance of the following

67 /*
68  * There are 4 different modes of operation for a hash table:
69  *
70  * SJHASH_INT nKey is used as the key and pKey is ignored.
71  *
72  * SJHASH_POINTER pKey is used as the key and nKey is ignored.
73  *
74  * SJHASH_STRING pKey points to a string that is nKey bytes long
75  * (including the null-terminator, if any). Case
76  * is ignored in comparisons.
77  *
78  * SJHASH_BINARY pKey points to binary data nKey bytes long.
79  * memcmp() is used to compare keys.
80  *
81  * A copy of the key is made for SJHASH_STRING and SJHASH_BINARY
82  * if the copyKey parameter to SjHashInit is 1.
83  */
84 #define SJHASH_INT 1
85 #define SJHASH_POINTER 2
86 #define SJHASH_STRING 3
87 #define SJHASH_BINARY 4
