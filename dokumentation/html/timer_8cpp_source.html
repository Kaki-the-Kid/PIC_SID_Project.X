37 void Timer::syncWithCpu()
38 {
39  if (ciaEventPauseTime > 0)
40  {
41  eventScheduler.cancel(m_cycleSkippingEvent);
42  const event_clock_t elapsed = eventScheduler.getTime(EVENT_CLOCK_PHI2) - ciaEventPauseTime;
43 
44  // It's possible for CIA to determine that it wants to go to sleep starting from the next
45  // cycle, and then have its plans aborted by CPU. Thus, we must avoid modifying
46  // the CIA state if the first sleep clock was still in the future.
47  if (elapsed >= 0)
48  {
49  timer -= elapsed;
50  clock();
51  }
52  }
53  if (ciaEventPauseTime == 0)
54  {
55  eventScheduler.cancel(*this);
56  }
57  ciaEventPauseTime = -1;
58 }

80 void Timer::clock()
81 {
82  if (timer != 0 && (state & CIAT_COUNT3) != 0)
83  {
84  timer--;
85  }
86 
87  /* ciatimer.c block start */
88  int_least32_t adj = state & (CIAT_CR_START | CIAT_CR_ONESHOT | CIAT_PHI2IN);
89  if ((state & (CIAT_CR_START | CIAT_PHI2IN)) == (CIAT_CR_START | CIAT_PHI2IN))
90  {
91  adj |= CIAT_COUNT2;
92  }
93  if ((state & CIAT_COUNT2) != 0
94  || (state & (CIAT_STEP | CIAT_CR_START)) == (CIAT_STEP | CIAT_CR_START))
95  {
96  adj |= CIAT_COUNT3;
97  }
98  // CR_FLOAD -> LOAD1, CR_ONESHOT -> ONESHOT0, LOAD1 -> LOAD, ONESHOT0 -> ONESHOT
99  adj |= (state & (CIAT_CR_FLOAD | CIAT_CR_ONESHOT | CIAT_LOAD1 | CIAT_ONESHOT0)) << 8;
100  state = adj;
101  /* ciatimer.c block end */
102 
103  if (timer == 0 && (state & CIAT_COUNT3) != 0)
104  {
105  state |= CIAT_LOAD | CIAT_OUT;
106 
107  if ((state & (CIAT_ONESHOT | CIAT_ONESHOT0)) != 0)
108  {
109  state &= ~(CIAT_CR_START | CIAT_COUNT2);
110  }
111 
112  // By setting bits 2&3 of the control register,
113  // PB6/PB7 will be toggled between high and low at each underflow.
114  const bool toggle = (lastControlValue & 0x06) == 6;
115  pbToggle = toggle && !pbToggle;
116 
117  // Implementation of the serial port
118  serialPort();
119 
120  // Timer A signals underflow handling: IRQ/B-count
121  underFlow();
122  }
123 
124  if ((state & CIAT_LOAD) != 0)
125  {
126  timer = latch;
127  state &= ~CIAT_COUNT3;
128  }
129 }
