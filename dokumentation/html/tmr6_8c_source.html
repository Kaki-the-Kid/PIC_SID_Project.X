119 void TMR6_ISR(void)
120 {
121  static volatile unsigned int CountCallBack = 0;
122 
123  // clear the TMR6 interrupt flag
124  PIR5bits.TMR6IF = 0;
125 
126  // callback function - called every 5000th pass
127  if (++CountCallBack >= TMR6_INTERRUPT_TICKER_FACTOR)
128  {
129  // ticker function call
130  TMR6_CallBack();
131 
132  // reset ticker counter
133  CountCallBack = 0;
134  }
135 }
