100 int vec_find(struct vec *p, cb_cmp * f, const void *in)
101 {
102  int lo;
103 
104  lo = -1;
105  if(p->flags & VEC_FLAG_SORTED)
106  {
107  int hi;
108  lo = 0;
109  hi = vec_count(p) - 1;
110  while(lo <= hi)
111  {
112  int next;
113  int val;
114 
115  next = (lo + hi) / 2;
116  val = f(in, vec_get(p, next));
117  if(val == 0)
118  {
119  /* match */
120  lo = -(next + 2);
121  break;
122  }
123  else if(val < 0)
124  {
125  hi = next - 1;
126  }
127  else
128  {
129  lo = next + 1;
130  }
131  }
132  }
133  return -(lo + 2);
134 }
