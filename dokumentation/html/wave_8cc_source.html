79 WaveformGenerator::WaveformGenerator()
80 {
81  static bool class_init;
82 
83  if (!class_init) {
84  // Calculate tables for normal waveforms.
85  accumulator = 0;
86  for (int i = 0; i < (1 << 12); i++) {
87  reg24 msb = accumulator & 0x800000;
88 
89  // Noise mask, triangle, sawtooth, pulse mask.
90  // The triangle calculation is made branch-free, just for the hell of it.
91  model_wave[0][0][i] = model_wave[1][0][i] = 0xfff;
92  model_wave[0][1][i] = model_wave[1][1][i] = ((accumulator ^ -!!msb) >> 11) & 0xffe;
93  model_wave[0][2][i] = model_wave[1][2][i] = accumulator >> 12;
94  model_wave[0][4][i] = model_wave[1][4][i] = 0xfff;
95 
96  accumulator += 0x1000;
97  }
98 
99  // Build DAC lookup tables for 12-bit DACs.
100  // MOS 6581: 2R/R ~ 2.20, missing termination resistor.
101  build_dac_table(model_dac[0], 12, 2.20, false);
102  // MOS 8580: 2R/R ~ 2.00, correct termination.
103  build_dac_table(model_dac[1], 12, 2.00, true);
104 
105  class_init = true;
106  }
107 
108  sync_source = this;
109 
110  sid_model = MOS6581;
111 
112  // Accumulator's even bits are high on powerup
113  accumulator = 0x555555;
114 
115  tri_saw_pipeline = 0x555;
116 
117  reset();
118 }

168 bool do_pre_writeback(reg8 waveform_prev, reg8 waveform, bool is6581)
169 {
170  // no writeback without combined waveforms
171  if (likely(waveform_prev <= 0x8))
172  return false;
173  // This need more investigation
174  if (waveform == 8)
175  return false;
176  // What's happening here?
177  if (is6581 &&
178  ((((waveform_prev & 0x3) == 0x1) && ((waveform & 0x3) == 0x2))
179  || (((waveform_prev & 0x3) == 0x2) && ((waveform & 0x3) == 0x1))))
180  return false;
181  // ok do the writeback
182  return true;
183 }

185 void WaveformGenerator::writeCONTROL_REG(reg8 control)
186 {
187  reg8 waveform_prev = waveform;
188  reg8 test_prev = test;
189  waveform = (control >> 4) & 0x0f;
190  test = control & 0x08;
191  ring_mod = control & 0x04;
192  sync = control & 0x02;
193 
194  // Set up waveform table.
195  wave = model_wave[sid_model][waveform & 0x7];
196 
197  // Substitution of accumulator MSB when sawtooth = 0, ring_mod = 1.
198  ring_msb_mask = ((~control >> 5) & (control >> 2) & 0x1) << 23;
199 
200  // no_noise and no_pulse are used in set_waveform_output() as bitmasks to
201  // only let the noise or pulse influence the output when the noise or pulse
202  // waveforms are selected.
203  no_noise = waveform & 0x8 ? 0x000 : 0xfff;
204  no_noise_or_noise_output = no_noise | noise_output;
205  no_pulse = waveform & 0x4 ? 0x000 : 0xfff;
206 
207  // Test bit rising.
208  // The accumulator is cleared, while the the shift register is prepared for
209  // shifting by interconnecting the register bits. The internal SRAM cells
210  // start to slowly rise up towards one. The SRAM cells reach one within
211  // approximately $8000 cycles, yielding a shift register value of
212  // 0x7fffff.
213  if (!test_prev && test) {
214  // Reset accumulator.
215  accumulator = 0;
216 
217  // Flush shift pipeline.
218  shift_pipeline = 0;
219 
220  // Set reset time for shift register.
221  shift_register_reset = (sid_model == MOS6581) ? SHIFT_REGISTER_RESET_6581 : SHIFT_REGISTER_RESET_8580;
222 
223  // The test bit sets pulse high.
224  pulse_output = 0xfff;
225  }
226  else if (test_prev && !test) {
227  // When the test bit is falling, the second phase of the shift is
228  // completed by enabling SRAM write.
229 
230  // During first phase of the shift the bits are interconnected
231  // and the output of each bit is latched into the following.
232  // The output may overwrite the latched value.
233  if (do_pre_writeback(waveform_prev, waveform, sid_model == MOS6581)) {
234  write_shift_register();
235  }
236 
237  // bit0 = (bit22 | test) ^ bit17 = 1 ^ bit17 = ~bit17
238  reg24 bit0 = (~shift_register >> 17) & 0x1;
239  shift_register = ((shift_register << 1) | bit0) & 0x7fffff;
240 
241  // Set new noise waveform output.
242  set_noise_output();
243  }
244 
245  if (waveform) {
246  // Set new waveform output.
247  set_waveform_output();
248  }
249  else if (waveform_prev) {
250  // Change to floating DAC input.
251  // Reset fading time for floating DAC input.
252  floating_output_ttl = (sid_model == MOS6581) ? FLOATING_OUTPUT_TTL_6581 : FLOATING_OUTPUT_TTL_8580;
253  }
254 
255  // The gate bit is handled by the EnvelopeGenerator.
256 }

266 void WaveformGenerator::reset()
267 {
268  // accumulator is not changed on reset
269  freq = 0;
270  pw = 0;
271 
272  msb_rising = false;
273 
274  waveform = 0;
275  test = 0;
276  ring_mod = 0;
277  sync = 0;
278 
279  wave = model_wave[sid_model][0];
280

282  no_noise = 0xfff;

284  pulse_output = 0xfff;
285 
286  // reset shift register
287  // when reset is released the shift register is clocked once
288  shift_register = 0x7ffffe;
289  shift_register_reset = 0;
290  set_noise_output();
291 
292  shift_pipeline = 0;
293

296  floating_output_ttl = 0;
