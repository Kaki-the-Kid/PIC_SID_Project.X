87 int __fastcall__ uncompress (unsigned char* dest, unsigned* destLen,
88  const unsigned char* source, unsigned sourceLen);
89 /*
90  Original zlib description:
91 
92  Decompresses the source buffer into the destination buffer. sourceLen is
93  the byte length of the source buffer. Upon entry, destLen is the total
94  size of the destination buffer, which must be large enough to hold the
95  entire uncompressed data. (The size of the uncompressed data must have
96  been saved previously by the compressor and transmitted to the decompressor
97  by some mechanism outside the scope of this compression library.)
98  Upon exit, destLen is the actual size of the compressed buffer.
99  This function can be used to decompress a whole file at once if the
100  input file is mmap'ed.
101 
102  uncompress returns Z_OK if success, Z_MEM_ERROR if there was not
103  enough memory, Z_BUF_ERROR if there was not enough room in the output
104  buffer, or Z_DATA_ERROR if the input data was corrupted.
105 
106  Implementation notes:
107 
108  This function expects data in the ZLIB format, described in RFC 1950
109  in the file ftp://ds.internic.net/rfc/rfc1950.txt. The ZLIB format is
110  essentially the DEFLATE format plus a very small header and Adler-32
111  checksum.
112 
113  Z_MEM_ERROR and Z_BUF_ERROR are never returned in this implementation.
114 */

118  unsigned len);
119 
120 /*
121  Original zlib description:
122 
123  Update a running Adler-32 checksum with the bytes buf[0..len-1] and
124  return the updated checksum. If buf is NULL, this function returns
125  the required initial value for the checksum.
126  An Adler-32 checksum is almost as reliable as a CRC32 but can be computed
127  much faster. Usage example:
128 
129  unsigned long adler = adler32(0L, Z_NULL, 0);
130 
131  while (read_buffer(buffer, length) != EOF) {
132  adler = adler32(adler, buffer, length);
133  }
134  if (adler != original_adler) error();
135 
136  Implementation notes:
137 
138  This function isn't actually much faster than crc32(), but it is smaller
139  and does not use any lookup tables.
140 */
