101 void EnvelopeGenerator::writeCONTROL_REG(unsigned char control)
102 {
103  const bool gate_next = (control & 0x01) != 0;
104 
105  if (gate_next != gate)
106  {
107  gate = gate_next;
108 
109  // The rate counter is never reset, thus there will be a delay before the
110  // envelope counter starts counting up (attack) or down (release).
111 
112  if (gate_next)
113  {
114  // Gate bit on: Start attack, decay, sustain.
115  next_state = ATTACK;
116  state = DECAY_SUSTAIN;
117  // The decay rate register is "accidentally" enabled during first cycle of attack phase
118  rate = adsrtable[decay];
119  state_pipeline = 2;
120  if (resetLfsr || (exponential_pipeline == 2))
121  {
122  envelope_pipeline = (exponential_counter_period == 1) || (exponential_pipeline == 2) ? 2 : 4;
123  }
124  else if (exponential_pipeline == 1)
125  {
126  state_pipeline = 3;
127  }
128  }
129  else
130  {
131  // Gate bit off: Start release.
132  next_state = RELEASE;
133  if (counter_enabled)
134  {
135  state_pipeline = envelope_pipeline > 0 ? 3 : 2;
136  }
137  }
138  }
139 }

141 void EnvelopeGenerator::writeATTACK_DECAY(unsigned char attack_decay)
142 {
143  attack = (attack_decay >> 4) & 0x0f;
144  decay = attack_decay & 0x0f;
145 
146  if (state == ATTACK)
147  {
148  rate = adsrtable[attack];
149  }
150  else if (state == DECAY_SUSTAIN)
151  {
152  rate = adsrtable[decay];

156 void EnvelopeGenerator::writeSUSTAIN_RELEASE(unsigned char sustain_release)
157 {
158  // From the sustain levels it follows that both the low and high 4 bits
159  // of the envelope counter are compared to the 4-bit sustain value.
160  // This has been verified by sampling ENV3.
161  //
162  // For a detailed description see:
163  // http://ploguechipsounds.blogspot.it/2010/11/new-research-on-sid-adsr.html
164  sustain = (sustain_release & 0xf0) | ((sustain_release >> 4) & 0x0f);
165 
166  release = sustain_release & 0x0f;
167 
168  if (state == RELEASE)
169  {
170  rate = adsrtable[release];
171  }
172 }
