358 int Filter8580::clock(int voice1, int voice2, int voice3)
359 {
360  voice1 = (voice1 * voiceScaleS14 >> 18) + voiceDC;
361  voice2 = (voice2 * voiceScaleS14 >> 18) + voiceDC;
362  // Voice 3 is silenced by voice3off if it is not routed through the filter.
363  voice3 = filt3 || !voice3off ? (voice3 * voiceScaleS14 >> 18) + voiceDC : 0;
364 
365  int Vi = 0;
366  int Vo = 0;
367 
368  (filt1 ? Vi : Vo) += voice1;
369  (filt2 ? Vi : Vo) += voice2;
370  (filt3 ? Vi : Vo) += voice3;
371  (filtE ? Vi : Vo) += ve;
372 
373  Vhp = currentSummer[currentResonance[Vbp] + Vlp + Vi];
374  Vbp = hpIntegrator->solve(Vhp);
375  Vlp = bpIntegrator->solve(Vbp);
376 
377  if (lp) Vo += Vlp;
378  if (bp) Vo += Vbp;
379  if (hp) Vo += Vhp;
380 
381  return currentGain[currentMixer[Vo]] - (1 << 15);
382 }
