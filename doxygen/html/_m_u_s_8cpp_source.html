65 bool detect(const uint8_t* buffer, size_t bufsize, uint_least32_t& voice3Index)
66 {
67  if ((buffer == nullptr) || (bufsize < 8))
68  return false;
69 
70  // Skip load address and 3x length entry.
71  uint_least32_t voice1Index = 2 + 3 * 2;
72  // Add length of voice 1 data.
73  voice1Index += endian_little16(&buffer[2]);
74  // Add length of voice 2 data.
75  uint_least32_t voice2Index = voice1Index + endian_little16(&buffer[4]);
76  // Add length of voice 3 data.
77  voice3Index = voice2Index + endian_little16(&buffer[6]);
78 
79  if (voice3Index > bufsize)
80  return false;
81 
82  return ((endian_big16(&buffer[voice1Index - 2]) == SIDTUNE_MUS_HLT_CMD)
83  && (endian_big16(&buffer[voice2Index - 2]) == SIDTUNE_MUS_HLT_CMD)
84  && (endian_big16(&buffer[voice3Index - 2]) == SIDTUNE_MUS_HLT_CMD));
85 }

116 bool MUS::mergeParts(buffer_t& musBuf, buffer_t& strBuf)
117 {
118  const uint_least32_t mergeLen = musBuf.size() + strBuf.size();
119 
120  // Sanity check. I do not trust those MUS/STR files around.
121  const uint_least32_t freeSpace = endian_16(player1[1], player1[0]) - SIDTUNE_MUS_DATA_ADDR;
122  if ((mergeLen - 4) > freeSpace)
123  {
124  throw loadError(ERR_SIZE_EXCEEDED);
125  }
126 
127  if (!strBuf.empty() && info->getSidChips() > 1)
128  {
129  // Install MUS data #2 _NOT_ including load address.
130  musBuf.insert(musBuf.end(), strBuf.begin(), strBuf.end());
131  }
132 
133  strBuf.clear();
134 
135  return true;
136 }

147 void MUS::installPlayer(sidmemory& mem)
148 {
149  // Install MUS player #1.
150  uint_least16_t dest = endian_16(player1[1], player1[0]);
151 
152  mem.fillRam(dest, player1 + 2, player1size - 2);
153  removeReads(mem, dest);
154  // Point player #1 to data #1.
155  mem.writeMemByte(dest + 0xc6e, (SIDTUNE_MUS_DATA_ADDR + 2) & 0xFF);
156  mem.writeMemByte(dest + 0xc70, (SIDTUNE_MUS_DATA_ADDR + 2) >> 8);
157 
158  if (info->getSidChips() > 1)
159  {
160  // Install MUS player #2.
161  dest = endian_16(player2[1], player2[0]);
162  mem.fillRam(dest, player2 + 2, player2size - 2);
163  removeReads(mem, dest);
164  // Point player #2 to data #2.
165  mem.writeMemByte(dest + 0xc6e, (SIDTUNE_MUS_DATA_ADDR + musDataLen + 2) & 0xFF);
166  mem.writeMemByte(dest + 0xc70, (SIDTUNE_MUS_DATA_ADDR + musDataLen + 2) >> 8);
167  }
168 }

176 SidTuneBase* MUS::load(buffer_t& musBuf,
177  buffer_t& strBuf,
178  uint_least32_t fileOffset,
179  bool init)
180 {
181  uint_least32_t voice3Index;
182  if (!detect(&musBuf[fileOffset], musBuf.size()-fileOffset, voice3Index))
183  return nullptr;
184 
185  std::unique_ptr<MUS> tune(new MUS());
186  tune->tryLoad(musBuf, strBuf, fileOffset, voice3Index, init);
187  tune->mergeParts(musBuf, strBuf);
188 
189  return tune.release();

192 void MUS::tryLoad(buffer_t& musBuf,
193  buffer_t& strBuf,
194  uint_least32_t fileOffset,
195  uint_least32_t voice3Index,
196  bool init)
197 {
198  if (init)
199  {
200  info->m_songs = 1;
201  info->m_startSong = 1;
202 
203  songSpeed[0] = SidTuneInfo::SPEED_CIA_1A;
204  clockSpeed[0] = SidTuneInfo::CLOCK_ANY;
205  }
206 
207  // Check setting compatibility for MUS playback
208  if ((info->m_compatibility != SidTuneInfo::COMPATIBILITY_C64)
209  || (info->m_relocStartPage != 0)
210  || (info->m_relocPages != 0))
211  {
212  throw loadError(ERR_INVALID);
213  }
214 
215  {
216  // All subtunes should be CIA
217  for (uint_least16_t i = 0; i < info->m_songs; i++)
218  {
219  if (songSpeed[i] != SidTuneInfo::SPEED_CIA_1A)
220  {
221  throw loadError(ERR_INVALID);
222  }
223  }
224  }
225 
226  musDataLen = musBuf.size();
227  info->m_loadAddr = SIDTUNE_MUS_DATA_ADDR;
228 
229  SmartPtr_sidtt<const uint8_t> spPet(&musBuf[fileOffset], musBuf.size() - fileOffset);
230 
231  // Voice3Index now is offset to text lines (uppercase Pet-strings).
232  spPet += voice3Index;
233 
234  // Extract credits
235  while (*spPet)
236  {
237  info->m_commentString.push_back(petsciiToAscii(spPet));
238  }
239 
240  spPet++;
241 
242  // If we appear to have additional data at the end, check is it's
243  // another mus file (but only if a second file isn't supplied)
244  bool stereo = false;
245  if (!strBuf.empty())
246  {
247  if (!detect(&strBuf[0], strBuf.size(), voice3Index))
248  throw loadError(ERR_2ND_INVALID);
249  spPet.setBuffer(&strBuf[0], strBuf.size());
250  stereo = true;
251  }
252  else
253  {
254  // For MUS + STR via stdin the files come combined
255  if (spPet.good())
256  {
257  const ulint_smartpt pos = spPet.tellPos();
258  if (detect(&spPet[0], spPet.tellLength()-pos, voice3Index))
259  {
260  musDataLen = static_cast<uint_least16_t>(pos);
261  stereo = true;
262  }
263  }
264  }
265 
266  if (stereo)
267  {
268  // Voice3Index now is offset to text lines (uppercase Pet-strings).
269  spPet += voice3Index;
270 
271  // Extract credits
272  while (*spPet)
273  {
274  info->m_commentString.push_back(petsciiToAscii(spPet));
275  }
276 
277  info->m_sidChipAddresses.push_back(SIDTUNE_SID2_BASE_ADDR);
278  info->m_formatString = TXT_FORMAT_STR;
279  }
280  else
281  {
282  info->m_formatString = TXT_FORMAT_MUS;
283  }
284 
285  setPlayerAddress();
286 
287  // Remove trailing empty lines.
288  const int lines = info->m_commentString.size();
289  {
290  for (int line = lines-1; line >= 0; line--)
291  {
292  if (info->m_commentString[line].length() == 0)
293  info->m_commentString.pop_back();
294  else
295  break;
296  }
297  }
298 }
