40 // Header has been extended for 'RSID' format
41 // The following changes are present:
42 // id = 'RSID'
43 // version = 2, 3 or 4
44 // play, load and speed reserved 0
45 // psidspecific flag is called C64BASIC flag
46 // init cannot be under ROMS/IO memory area
47 // load address cannot be less than $07E8
48 // info strings may be 32 characters long without trailing zero
49 
50 struct psidHeader // all values are big-endian
51 {
52  uint32_t id; // 'PSID' or 'RSID' (ASCII)
53  uint16_t version; // 1, 2, 3 or 4
54  uint16_t data; // 16-bit offset to binary data in file
55  uint16_t load; // 16-bit C64 address to load file to
56  uint16_t init; // 16-bit C64 address of init subroutine
57  uint16_t play; // 16-bit C64 address of play subroutine
58  uint16_t songs; // number of songs
59  uint16_t start; // start song out of [1..256]
60  uint32_t speed; // 32-bit speed info
61  // bit: 0=50 Hz, 1=CIA 1 Timer A (default: 60 Hz)
62  char name[PSID_MAXSTRLEN]; // ASCII strings, 31 characters long and
63  char author[PSID_MAXSTRLEN]; // terminated by a trailing zero
64  char released[PSID_MAXSTRLEN]; //
65 
66  uint16_t flags; // only version >= 2
67  uint8_t relocStartPage; // only version >= 2ng
68  uint8_t relocPages; // only version >= 2ng
69  uint8_t sidChipBase2; // only version >= 3
70  uint8_t sidChipBase3; // only version >= 4
71 };

114 SidTuneInfo::model_t getSidModel(uint_least16_t modelFlag)
115 {
116  if ((modelFlag & PSID_SIDMODEL_ANY) == PSID_SIDMODEL_ANY)
117  return SidTuneInfo::SIDMODEL_ANY;
118 
119  if (modelFlag & PSID_SIDMODEL_6581)
120  return SidTuneInfo::SIDMODEL_6581;
121 
122  if (modelFlag & PSID_SIDMODEL_8580)
123  return SidTuneInfo::SIDMODEL_8580;
124 
125  return SidTuneInfo::SIDMODEL_UNKNOWN;
126 }

131 bool validateAddress(uint_least8_t address)
132 {
133  // Only even values are valid.
134  if (address & 1)
135  return false;
136 
137  // Ranges $00-$41 ($D000-$D410) and $80-$DF ($D800-$DDF0) are invalid.
138  // Any invalid value means that no second SID is used, like $00.
139  if (address <= 0x41
140  || (address >= 0x80 && address <= 0xdf))
141  return false;
142 
143  return true;
144 }

146 SidTuneBase* PSID::load(buffer_t& dataBuf)
147 {
148  // File format check
149  if (dataBuf.size() < 4)
150  {
151  return nullptr;
152  }
153 
154  const uint32_t magic = endian_big32(&dataBuf[0]);
155  if ((magic != PSID_ID)
156  && (magic != RSID_ID))
157  {
158  return nullptr;
159  }
160 
161  psidHeader pHeader;
162  readHeader(dataBuf, pHeader);
163 
164  std::unique_ptr<PSID> tune(new PSID());
165  tune->tryLoad(pHeader);
166 
167  return tune.release();
168 }

170 void PSID::readHeader(const buffer_t &dataBuf, psidHeader &hdr)
171 {
172  // Due to security concerns, input must be at least as long as version 1
173  // header plus 16-bit C64 load address. That is the area which will be
174  // accessed.
175  if (dataBuf.size() < (psid_headerSize + 2))
176  {
177  throw loadError(ERR_TRUNCATED);
178  }
179 
180  // Read v1 fields
181  hdr.id = endian_big32(&dataBuf[0]);
182  hdr.version = endian_big16(&dataBuf[4]);
183  hdr.data = endian_big16(&dataBuf[6]);
184  hdr.load = endian_big16(&dataBuf[8]);
185  hdr.init = endian_big16(&dataBuf[10]);
186  hdr.play = endian_big16(&dataBuf[12]);
187  hdr.songs = endian_big16(&dataBuf[14]);
188  hdr.start = endian_big16(&dataBuf[16]);
189  hdr.speed = endian_big32(&dataBuf[18]);
190  memcpy(hdr.name, &dataBuf[22], PSID_MAXSTRLEN);
191  memcpy(hdr.author, &dataBuf[54], PSID_MAXSTRLEN);
192  memcpy(hdr.released, &dataBuf[86], PSID_MAXSTRLEN);
193 
194  if (hdr.version >= 2)
195  {
196  if (dataBuf.size() < (psidv2_headerSize + 2))
197  {
198  throw loadError(ERR_TRUNCATED);
199  }
200 
201  // Read v2/3/4 fields
202  hdr.flags = endian_big16(&dataBuf[118]);
203  hdr.relocStartPage = dataBuf[120];
204  hdr.relocPages = dataBuf[121];
205  hdr.sidChipBase2 = dataBuf[122];
206  hdr.sidChipBase3 = dataBuf[123];
207  }

210 void PSID::tryLoad(const psidHeader &pHeader)
211 {
212  SidTuneInfo::compatibility_t compatibility = SidTuneInfo::COMPATIBILITY_C64;
213 
214  // Require a valid ID and version number.
215  if (pHeader.id == PSID_ID)
216  {
217  switch (pHeader.version)
218  {
219  case 1:
220  compatibility = SidTuneInfo::COMPATIBILITY_PSID;
221  break;
222  case 2:
223  case 3:
224  case 4:
225  break;
226  default:
227  throw loadError(TXT_UNKNOWN_PSID);
228  }
229  info->m_formatString = TXT_FORMAT_PSID;
230  }
231  else if (pHeader.id == RSID_ID)
232  {
233  switch (pHeader.version)
234  {
235  case 2:
236  case 3:
237  case 4:
238  break;
239  default:
240  throw loadError(TXT_UNKNOWN_RSID);
241  }
242  info->m_formatString = TXT_FORMAT_RSID;
243  compatibility = SidTuneInfo::COMPATIBILITY_R64;
244  }
245 
246  fileOffset = pHeader.data;
247  info->m_loadAddr = pHeader.load;
248  info->m_initAddr = pHeader.init;
249  info->m_playAddr = pHeader.play;
250  info->m_songs = pHeader.songs;
251  info->m_startSong = pHeader.start;
252  info->m_compatibility = compatibility;
253  info->m_relocPages = 0;
254  info->m_relocStartPage = 0;
255 
256  uint_least32_t speed = pHeader.speed;
257  SidTuneInfo::clock_t clock = SidTuneInfo::CLOCK_UNKNOWN;
258 
259  bool musPlayer = false;
260 
261  if (pHeader.version >= 2)
262  {
263  const uint_least16_t flags = pHeader.flags;
264 
265  // Check clock
266  if (flags & PSID_MUS)
267  { // MUS tunes run at any speed
268  clock = SidTuneInfo::CLOCK_ANY;
269  musPlayer = true;
270  }
271  else
272  {
273  switch (flags & PSID_CLOCK)
274  {
275  case PSID_CLOCK_ANY:
276  clock = SidTuneInfo::CLOCK_ANY;
277  break;
278  case PSID_CLOCK_PAL:
279  clock = SidTuneInfo::CLOCK_PAL;
280  break;
281  case PSID_CLOCK_NTSC:
282  clock = SidTuneInfo::CLOCK_NTSC;
283  break;
284  default:
285  break;
286  }
287  }
288 
289  // These flags are only available for the appropriate
290  // file formats
291  switch (compatibility)
292  {
293  case SidTuneInfo::COMPATIBILITY_C64:
294  if (flags & PSID_SPECIFIC)
295  info->m_compatibility = SidTuneInfo::COMPATIBILITY_PSID;
296  break;
297  case SidTuneInfo::COMPATIBILITY_R64:
298  if (flags & PSID_BASIC)
299  info->m_compatibility = SidTuneInfo::COMPATIBILITY_BASIC;
300  break;
301  default:
302  break;
303  }
304 
305  info->m_clockSpeed = clock;
306 
307  info->m_sidModels[0] = getSidModel(flags >> 4);
308 
309  info->m_relocStartPage = pHeader.relocStartPage;
310  info->m_relocPages = pHeader.relocPages;
311 
312  if (pHeader.version >= 3)
313  {
314  if (validateAddress(pHeader.sidChipBase2))
315  {
316  info->m_sidChipAddresses.push_back(0xd000 | (pHeader.sidChipBase2 << 4));
317 
318  info->m_sidModels.push_back(getSidModel(flags >> 6));
319  }
320 
321  if (pHeader.version >= 4)
322  {
323  if (pHeader.sidChipBase3 != pHeader.sidChipBase2
324  && validateAddress(pHeader.sidChipBase3))
325  {
326  info->m_sidChipAddresses.push_back(0xd000 | (pHeader.sidChipBase3 << 4));
327 
328  info->m_sidModels.push_back(getSidModel(flags >> 8));
329  }
330  }
331  }
332  }
333 
334  // Check reserved fields to force real c64 compliance
335  // as required by the RSID specification
336  if (compatibility == SidTuneInfo::COMPATIBILITY_R64)
337  {
338  if ((info->m_loadAddr != 0)
339  || (info->m_playAddr != 0)
340  || (speed != 0))
341  {
342  throw loadError(ERR_INVALID);
343  }
344 
345  // Real C64 tunes appear as CIA
346  speed = ~0;
347  }
348 
349  // Create the speed/clock setting table.
350  convertOldStyleSpeedToTables(speed, clock);
351 
352  // Copy info strings.
353  info->m_infoString.push_back(std::string(pHeader.name, PSID_MAXSTRLEN));
354  info->m_infoString.push_back(std::string(pHeader.author, PSID_MAXSTRLEN));
355  info->m_infoString.push_back(std::string(pHeader.released, PSID_MAXSTRLEN));
356 
357  if (musPlayer)
358  throw loadError("Compute!'s Sidplayer MUS data is not supported yet"); // TODO
359 }

361 const char *PSID::createMD5(char *md5)
362 {
363  if (md5 == nullptr)
364  md5 = m_md5;
365 
366  *md5 = '\0';
367 
368  try
369  {
370  // Include C64 data.
371  sidmd5 myMD5;
372  myMD5.append(&cache[fileOffset], info->m_c64dataLen);
373 
374  uint8_t tmp[2];
375  // Include INIT and PLAY address.
376  endian_little16(tmp, info->m_initAddr);
377  myMD5.append(tmp, sizeof(tmp));
378  endian_little16(tmp, info->m_playAddr);
379  myMD5.append(tmp, sizeof(tmp));
380 
381  // Include number of songs.
382  endian_little16(tmp, info->m_songs);
383  myMD5.append(tmp, sizeof(tmp));
384 
385  {
386  // Include song speed for each song.
387  const unsigned int currentSong = info->m_currentSong;
388  for (unsigned int s = 1; s <= info->m_songs; s++)
389  {
390  selectSong(s);
391  const uint8_t songSpeed = static_cast<uint8_t>(info->m_songSpeed);
392  myMD5.append(&songSpeed, sizeof(songSpeed));
393  }
394  // Restore old song
395  selectSong(currentSong);
396  }
397 
398  // Deal with PSID v2NG clock speed flags: Let only NTSC
399  // clock speed change the MD5 fingerprint. That way the
400  // fingerprint of a PAL-speed sidtune in PSID v1, v2, and
401  // PSID v2NG format is the same.
402  if (info->m_clockSpeed == SidTuneInfo::CLOCK_NTSC)
403  {
404  const uint8_t ntsc_val = 2;
405  myMD5.append(&ntsc_val, sizeof(ntsc_val));
406  }
407 
408  // NB! If the fingerprint is used as an index into a
409  // song-lengths database or cache, modify above code to
410  // allow for PSID v2NG files which have clock speed set to
411  // SIDTUNE_CLOCK_ANY. If the SID player program fully
412  // supports the SIDTUNE_CLOCK_ANY setting, a sidtune could
413  // either create two different fingerprints depending on
414  // the clock speed chosen by the player, or there could be
415  // two different values stored in the database/cache.
416 
417  myMD5.finish();
418 
419  // Get fingerprint.
420  myMD5.getDigest().copy(md5, SidTune::MD5_LENGTH);
421  md5[SidTune::MD5_LENGTH] = '\0';
422  }
423  catch (md5Error const &)
424  {
425  return nullptr;
426  }
427 
428  return md5;
429 }
