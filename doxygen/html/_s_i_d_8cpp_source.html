101 void SID::voiceSync(bool sync)
102 {
103  if (sync)
104  {
105  // Synchronize the 3 waveform generators.
106  for (int i = 0; i < 3; i++)
107  {
108  voice[i]->wave()->synchronize(voice[(i + 1) % 3]->wave(), voice[(i + 2) % 3]->wave());
109  }
110  }
111 
112  // Calculate the time to next voice sync
113  nextVoiceSync = std::numeric_limits<int>::max();
114 
115  for (int i = 0; i < 3; i++)
116  {
117  const unsigned int freq = voice[i]->wave()->readFreq();
118 
119  if (voice[i]->wave()->readTest() || freq == 0 || !voice[(i + 1) % 3]->wave()->readSync())
120  {
121  continue;
122  }
123 
124  const unsigned int accumulator = voice[i]->wave()->readAccumulator();
125  const unsigned int thisVoiceSync = ((0x7fffff - accumulator) & 0xffffff) / freq + 1;
126 
127  if (thisVoiceSync < nextVoiceSync)
128  {
129  nextVoiceSync = thisVoiceSync;
130  }
131  }
132 }

134 void SID::setChipModel(ChipModel model)
135 {
136  switch (model)
137  {
138  case MOS6581:
139  filter = filter6581.get();
140  modelTTL = BUS_TTL_6581;
141  break;
142 
143  case MOS8580:
144  filter = filter8580.get();
145  modelTTL = BUS_TTL_8580;
146  break;
147 
148  default:
149  throw SIDError("Unknown chip type");
150  }
151 
152  this->model = model;
153 
154  // calculate waveform-related tables, feed them to the generator
155  matrix_t* tables = WaveformCalculator::getInstance()->buildTable(model);
156 
157  // update voice offsets
158  for (int i = 0; i < 3; i++)
159  {
160  voice[i]->envelope()->setChipModel(model);
161  voice[i]->wave()->setChipModel(model);
162  voice[i]->wave()->setWaveformModels(tables);
163  }

193 unsigned char SID::read(int offset)
194 {
195  switch (offset)
196  {
197  case 0x19: // X value of paddle
198  busValue = potX->readPOT();
199  busValueTtl = modelTTL;
200  break;
201 
202  case 0x1a: // Y value of paddle
203  busValue = potY->readPOT();
204  busValueTtl = modelTTL;
205  break;
206 
207  case 0x1b: // Voice #3 waveform output
208  busValue = voice[2]->wave()->readOSC();
209  busValueTtl = modelTTL;
210  break;
211 
212  case 0x1c: // Voice #3 ADSR output
213  busValue = voice[2]->envelope()->readENV();
214  busValueTtl = modelTTL;
215  break;
216 
217  default:
218  // Reading from a write-only or non-existing register
219  // makes the bus discharge faster.
220  // Emulate this by halving the residual TTL.
221  busValueTtl /= 2;
222  break;
223  }
224 
225  return busValue;
226 }

366 void SID::clockSilent(unsigned int cycles)
367 {
368  ageBusValue(cycles);
369 
370  while (cycles != 0)
371  {
372  int delta_t = std::min(nextVoiceSync, cycles);
373 
374  if (delta_t > 0)
375  {
376  for (int i = 0; i < delta_t; i++)
377  {
378  // clock waveform generators (can affect OSC3)
379  voice[0]->wave()->clock();
380  voice[1]->wave()->clock();
381  voice[2]->wave()->clock();
382 
383  voice[0]->wave()->output(voice[2]->wave());
384  voice[1]->wave()->output(voice[0]->wave());
385  voice[2]->wave()->output(voice[1]->wave());
386 
387  // clock ENV3 only
388  voice[2]->envelope()->clock();
389  }
390 
391  cycles -= delta_t;
392  nextVoiceSync -= delta_t;
393  }
394 
395  if (nextVoiceSync == 0)
396  {
397  voiceSync(true);
398  }
399  }
400 }
