62 const char *parseTime(const char *str, int_least32_t &result)
63 {
64  char *end;
65  const long minutes = strtol(str, &end, 10);
66 
67  if (*end != ':')
68  {
69  throw parseError();
70  }
71 
72  end++;
73  const long seconds = strtol(end, &end, 10);
74  result = ((minutes * 60) + seconds) * 1000;
75 
76  if (*end == '.')
77  {
78  end++;
79  long milliseconds = strtol(end, &end, 10);
80  if (milliseconds<10)
81  milliseconds *= 100;
82  else if (milliseconds<100)
83  milliseconds *= 10;
84  else if (milliseconds>=1000)
85  throw parseError();
86  result += milliseconds;
87  }
88 
89  while (!isspace(*end))
90  {
91  end++;
92  }
93 
94  return end;
95 }
