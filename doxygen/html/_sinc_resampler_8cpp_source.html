63 double I0(double x)
64 {
65  double sum = 1.;
66  double u = 1.;
67  double n = 1.;
68  const double halfx = x / 2.;
69 
70  do
71  {
72  const double temp = halfx / n;
73  u *= temp * temp;
74  sum += u;
75  n += 1.;
76  }
77  while (u >= I0E * sum);
78 
79  return sum;
80 }

90 int convolve(const short* a, const short* b, int bLength)
91 {
92 #ifdef HAVE_MMINTRIN_H
93  __m64 acc = _mm_setzero_si64();
94 
95  const int n = bLength / 4;
96 
97  for (int i = 0; i < n; i++)
98  {
99  const __m64 tmp = _mm_madd_pi16(*(__m64*)a, *(__m64*)b);
100  acc = _mm_add_pi16(acc, tmp);
101  a += 4;
102  b += 4;
103  }
104 
105  int out = _mm_cvtsi64_si32(acc) + _mm_cvtsi64_si32(_mm_srli_si64(acc, 32));
106  _mm_empty();
107 
108  bLength &= 3;
109 #else
110  int out = 0;
111 #endif
112 
113  for (int i = 0; i < bLength; i++)
114  {
115  out += *a++ * *b++;
116  }
117 
118  return (out + (1 << 14)) >> 15;
119 }

121 int SincResampler::fir(int subcycle)
122 {
123  // Find the first of the nearest fir tables close to the phase
124  int firTableFirst = (subcycle * firRES >> 10);
125  const int firTableOffset = (subcycle * firRES) & 0x3ff;
126 
127  // Find firN most recent samples, plus one extra in case the FIR wraps.
128  int sampleStart = sampleIndex - firN + RINGSIZE - 1;
129 
130  const int v1 = convolve(sample + sampleStart, (*firTable)[firTableFirst], firN);
131 
132  // Use next FIR table, wrap around to first FIR table using
133  // previous sample.
134  if (unlikely(++firTableFirst == firRES))
135  {
136  firTableFirst = 0;
137  ++sampleStart;
138  }
139 
140  const int v2 = convolve(sample + sampleStart, (*firTable)[firTableFirst], firN);
141 
142  // Linear interpolation between the sinc tables yields good
143  // approximation for the exact value.
144  return v1 + (firTableOffset * (v2 - v1) >> 10);
145 }

147 SincResampler::SincResampler(double clockFrequency, double samplingFrequency, double highestAccurateFrequency) :
148  sampleIndex(0),
149  cyclesPerSample(static_cast<int>(clockFrequency / samplingFrequency * 1024.)),
150  sampleOffset(0),
151  outputValue(0)
152 {
153  // 16 bits -> -96dB stopband attenuation.
154  const double A = -20. * log10(1.0 / (1 << BITS));
155  // A fraction of the bandwidth is allocated to the transition band, which we double
156  // because we design the filter to transition halfway at nyquist.
157  const double dw = (1. - 2.*highestAccurateFrequency / samplingFrequency) * M_PI * 2.;
158 
159  // For calculation of beta and N see the reference for the kaiserord
160  // function in the MATLAB Signal Processing Toolbox:
161  // http://www.mathworks.com/help/signal/ref/kaiserord.html
162  const double beta = 0.1102 * (A - 8.7);
163  const double I0beta = I0(beta);
164  const double cyclesPerSampleD = clockFrequency / samplingFrequency;
165 
166  {
167  // The filter order will maximally be 124 with the current constraints.
168  // N >= (96.33 - 7.95)/(2 * pi * 2.285 * (maxfreq - passbandfreq) >= 123
169  // The filter order is equal to the number of zero crossings, i.e.
170  // it should be an even number (sinc is symmetric with respect to x = 0).
171  int N = static_cast<int>((A - 7.95) / (2.285 * dw) + 0.5);
172  N += N & 1;
173 
174  // The filter length is equal to the filter order + 1.
175  // The filter length must be an odd number (sinc is symmetric with respect to
176  // x = 0).
177  firN = static_cast<int>(N * cyclesPerSampleD) + 1;
178  firN |= 1;
179 
180  // Check whether the sample ring buffer would overflow.
181  assert(firN < RINGSIZE);
182 
183  // Error is bounded by err < 1.234 / L^2, so L = sqrt(1.234 / (2^-16)) = sqrt(1.234 * 2^16).
184  firRES = static_cast<int>(ceil(sqrt(1.234 * (1 << BITS)) / cyclesPerSampleD));
185 
186  // firN*firRES represent the total resolution of the sinc sampling. JOS
187  // recommends a length of 2^BITS, but we don't quite use that good a filter.
188  // The filter test program indicates that the filter performs well, though.
189  }
190 
191  // Create the map key
192  std::ostringstream o;
193  o << firN << "," << firRES << "," << cyclesPerSampleD;
194  const std::string firKey = o.str();
195  fir_cache_t::iterator lb = FIR_CACHE.lower_bound(firKey);
196 
197  // The FIR computation is expensive and we set sampling parameters often, but
198  // from a very small set of choices. Thus, caching is used to speed initialization.
199  if (lb != FIR_CACHE.end() && !(FIR_CACHE.key_comp()(firKey, lb->first)))
200  {
201  firTable = &(lb->second);
202  }
203  else
204  {
205  // Allocate memory for FIR tables.
206  matrix_t tempTable(firRES, firN);
207  firTable = &(FIR_CACHE.insert(lb, fir_cache_t::value_type(firKey, tempTable))->second);
208 
209  // The cutoff frequency is midway through the transition band, in effect the same as nyquist.
210  const double wc = M_PI;
211 
212  // Calculate the sinc tables.
213  const double scale = 32768.0 * wc / cyclesPerSampleD / M_PI;
214 
215  for (int i = 0; i < firRES; i++)
216  {
217  const double jPhase = (double) i / firRES + firN / 2;
218 
219  for (int j = 0; j < firN; j++)
220  {
221  const double x = j - jPhase;
222 
223  const double xt = x / (firN / 2);
224  const double kaiserXt = fabs(xt) < 1. ? I0(beta * sqrt(1. - xt * xt)) / I0beta : 0.;
225 
226  const double wt = wc * x / cyclesPerSampleD;
227  const double sincWt = fabs(wt) >= 1e-8 ? sin(wt) / wt : 1.;
228 
229  (*firTable)[i][j] = static_cast<short>(scale * sincWt * kaiserXt);
230  }
231  }
232  }
233 }

243 bool SincResampler::input(int input)
244 {
245  bool ready = false;
246 
247  /*
248  * Clip the input as it may overflow the 16 bit range.
249  *
250  * Approximate measured input ranges:
251  * 6581: ]-20000,+20000[
252  * 8580: ]-33000,+33000[
253  */
254  sample[sampleIndex] = sample[sampleIndex + RINGSIZE] = clip<int, short>(input);
255  sampleIndex = (sampleIndex + 1) & (RINGSIZE - 1);
256 
257  if (sampleOffset < 1024)
258  {
259  outputValue = fir(sampleOffset);
260  ready = true;
261  sampleOffset += cyclesPerSample;
262  }
263 
264  sampleOffset -= 1024;
265 
266  return ready;
267 }
