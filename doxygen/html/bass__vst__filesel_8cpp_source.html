23 long fileSelOpen(BASS_VST_PLUGIN* this_, VstFileSelect* vstFs)
24 {
25  OPENFILENAME ofn;
26 
27  // allocate the memory to hold the file name
28  if( this_ == NULL || vstFs == NULL )
29  goto Cleanup;
30  #define BUFFER_BYTES 4096
31  vstFs->reserved = (int)malloc(BUFFER_BYTES+32/*32 for the faked array*/);
32  if( vstFs->reserved == NULL )
33  goto Cleanup;
34 
35  // show the file selector
36  memset(&ofn, 0, sizeof(OPENFILENAME));
37  ofn.lStructSize = sizeof(OPENFILENAME);
38  ofn.lpstrFilter = "*.*\0*.*\0\0";
39  ofn.lpstrFile = (char*)vstFs->reserved;
40  ofn.lpstrFile[0] = 0;
41  ofn.nMaxFile = BUFFER_BYTES-1;
42  ofn.lpstrTitle = vstFs->title;
43  if( vstFs->command == kVstFileSave )
44  {
45  if( !GetSaveFileName(&ofn) )
46  goto Cleanup;
47  }
48  else if( vstFs->command == kVstFileLoad
49  || vstFs->command == kVstMultipleFilesLoad
50  || vstFs->command == kVstDirectorySelect )
51  {
52  if( !GetOpenFileName(&ofn) )
53  goto Cleanup;
54  }
55  else
56  {
57  goto Cleanup;
58  }
59 
60  // set the result ...
61  if( vstFs->command == kVstMultipleFilesLoad )
62  {
63  // ... set a faked-multiple result
64  vstFs->returnMultiplePaths = (char**)(vstFs->reserved+BUFFER_BYTES);
65  vstFs->returnMultiplePaths[0] = ofn.lpstrFile;
66  vstFs->nbReturnPath = 1;
67  }
68  else
69  {
70  // ... set a single result
71  if( vstFs->command == kVstDirectorySelect )
72  {
73  // trim the file name from the returned path
74  char* ptr = strrchr(ofn.lpstrFile, '\\');
75  if( ptr )
76  {
77  ptr++;
78  *ptr = 0;
79  }
80  }
81 
82  if( vstFs->returnPath )
83  {
84  // copy data to user-provided buffer and free our buffer by calling fileSelClose()
85  // (the size check is hard-coded as not all effect modules set vst->returnPathSize)
86  if( strlen(ofn.lpstrFile) < 1000 )
87  strcpy(vstFs->returnPath, ofn.lpstrFile);
88  fileSelClose(this_, vstFs);
89  }
90  else
91  {
92  // buffer is allocated by us - fileSelClose() must be called by the user
93  vstFs->returnPath = ofn.lpstrFile;
94  }
95  }
96 
97 Cleanup:
98  return 1; // we support file selection - always return 1
99 }
