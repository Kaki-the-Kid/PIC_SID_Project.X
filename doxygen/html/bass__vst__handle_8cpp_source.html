65 BASS_VST_PLUGIN* createHandle(DWORD type, DWORD requestedHandleValue)
66 {
67  // allocate memory and init the memory
68  BASS_VST_PLUGIN* this_ = (BASS_VST_PLUGIN*)malloc(sizeof(BASS_VST_PLUGIN));
69  if( this_ == NULL )
70  return NULL;
71 
72  memset(this_, 0, sizeof(BASS_VST_PLUGIN));
73  this_->type = type;
74 
75  // calculate the handle to use - either use the given one or simply the pointer
76  if( requestedHandleValue == 0 )
77  {
78  requestedHandleValue = (DWORD)this_;
79  }
80 
81  // make sure, the handle value is okay
82  EnterCriticalSection(&s_handleCritical);
83 
84  if( sjhashFind(&s_handleHash, NULL, /*pKey, not needed*/ (int)requestedHandleValue/*nKey*/) == NULL )
85  {
86  this_->vstHandle = requestedHandleValue;
87  sjhashInsert(&s_handleHash, NULL, /*pKey, not needed*/ (int)this_->vstHandle, /*nKey*/ this_/*pData*/);
88  }
89 
90  LeaveCriticalSection(&s_handleCritical);
91 
92  if( this_->vstHandle == 0 )
93  {
94  free(this_);
95  return NULL;
96  }
97 
98  // init some basic data
99  InitializeCriticalSection(&this_->vstCritical_);
100  InitializeCriticalSection(&this_->midiCritical_);
101 
102  this_->handleUsage = 1;
103 
104  return this_;
105 }

109 static void destroyHandle(BASS_VST_PLUGIN* this_)
110 {
111  // this function is only called indirectly if the last reference to a handle is lost
112 
113  // unlink from everything ...

121  BASS_ChannelRemoveDSP(this_->channelHandle, this_->dspHandle);

128  if( this_->editorIsOpen )

144  // if the library is aleady pending, increase the unload counter
145  EnterCriticalSection(&s_idleCritical);

212 BOOL unrefHandle(DWORD handle)
213 {
214  bool ret = false;
215 
216  BASS_VST_PLUGIN* ptrToDestroy = NULL;
217 
218  EnterCriticalSection(&s_handleCritical);
219  BASS_VST_PLUGIN* this_ = (BASS_VST_PLUGIN*)sjhashFind(&s_handleHash, NULL, /*pKey, not needed*/ (int)handle/*nKey*/);
220  if( this_ )
221  {
222  ret = true;
223  this_->handleUsage--;
224  if( this_->handleUsage <= 0 )
225  {
226  sjhashInsert(&s_handleHash, NULL, /*pKey, not needed*/ (int)this_->vstHandle, /*nKey*/ 0/*pData = remove*/);
227  ptrToDestroy = this_;
228  }
229  }
230  LeaveCriticalSection(&s_handleCritical);
231 
232  if( ptrToDestroy )
233  {
234  destroyHandle(ptrToDestroy);
235  }
236 
237  return ret;
238 }

297 int validateLastValues(BASS_VST_PLUGIN* this_)
298 {
299  if( this_->aeffect->getParameter )
300  {
301  int paramCount = this_->aeffect->numParams;
302  // check, if the param count has changed
303  if (paramCount != this_->numLastValues)
304  {
305  if (paramCount > this_->numLastValues)
306  {
307  // increase the lastValues array
308  long bytesNeeded = sizeof(float) * paramCount;
309  if( (this_->lastValues = (float*)realloc(this_->lastValues, bytesNeeded)) != NULL )
310  {
311  // success, so init the newly added values
312  int paramIndex = this_->numLastValues;
313  for( ; paramIndex < paramCount; paramIndex++ )
314  {
315  this_->lastValues[paramIndex] = this_->aeffect->getParameter(this_->aeffect, paramIndex);
316  }
317  // set the new number
318  this_->numLastValues = paramCount;
319  } // else lastValues and numLastValues are left unchanged (should never happen)
320  }
321  else if (paramCount > 0)
322  {
323  // the new paramCount is smaller than the old paramCount
324  // lastValues is left unchanged (but numLastValues is set to the new value)
325  this_->numLastValues = paramCount;
326  }
327  else
328  {
329  // lastValues is left unchanged (but numLastValues is set to 0)
330  this_->numLastValues = 0;
331  }
332  }
333  }
334 
335  return this_->numLastValues;
336 }

345 void checkForwarding()
346 {
347  sjhash oldForwardReceivers;
348  sjhashInit(&oldForwardReceivers, SJHASH_INT, /*keytype*/ 0/*copyKey*/);
349 
350  EnterCriticalSection(&s_forwardCritical);
351  EnterCriticalSection(&s_handleCritical);
352 
353  // collect all "old" forward receivers
354  BASS_VST_PLUGIN* this_;
355  sjhashElem *elemThis = sjhashFirst(&s_handleHash), *elemOther;
356  while( elemThis )
357  {
358  this_ = (BASS_VST_PLUGIN*)sjhashData(elemThis);
359 
360  // reset all forwardigs
361  this_->forwardDataToOtherCnt = 0;
362 
363  // effStartProcessCalled set without an own channel handle?
364  // -> this is an old forwarding -> mark this, maybe we will remove it later
365  if( this_->type==VSTeffect && this_->channelHandle==0 && this_->effStartProcessCalled )
366  {
367  sjhashInsert(&oldForwardReceivers, NULL, /*pKey, not needed*/ (int)this_, /*nKey*/
368  (void*)1/*pData - 0 = remove*/);
369  }
370 
371  elemThis = sjhashNext(elemThis);
372  }
373 
374  // search for vst handles with no channel but with the editor opened
375  elemThis = sjhashFirst(&s_handleHash);
376  while( elemThis )
377  {
378  this_ = (BASS_VST_PLUGIN*)sjhashData(elemThis);
379 
380  if( this_->type==VSTeffect && this_->channelHandle==0 && this_->editorIsOpen )
381  {
382  elemOther = sjhashFirst(&s_handleHash);
383  while( elemOther )
384  {
385  BASS_VST_PLUGIN* other_ = (BASS_VST_PLUGIN*)sjhashData(elemOther);
386 
387  if( other_->channelHandle != 0
388  && other_->editorScope == this_->editorScope )
389  {
390  other_->forwardDataToOtherVstHandles[other_->forwardDataToOtherCnt] = this_->vstHandle;
391  if( other_->forwardDataToOtherCnt < MAX_FWD-1 ) // if the buffer (128(!) handles) is full, this will only result in display problems of VU meters, nothing really to worry about ...
392  other_->forwardDataToOtherCnt++;
393 
394  if( sjhashInsert(&oldForwardReceivers, NULL, /*pKey, not needed*/ (int)this_, /*nKey*/
395  (void*)0/*pData - 0 = remove*/) == 0 )
396  {
397  openProcess(this_, other_);
398  }
399  else if( other_->effBlockSize )
400  {
401  callMainsChanged(this_, other_->effBlockSize);
402  }
403  }
404 
405  elemOther = sjhashNext(elemOther);
406  }
407  }
408 
409  elemThis = sjhashNext(elemThis);
410  }
411 
412  // close the rest of the forwarding handles
413  elemThis = sjhashFirst(&oldForwardReceivers);
414  while( elemThis )
415  {
416  this_ = (BASS_VST_PLUGIN*)sjhashKeysize(elemThis);
417  closeProcess(this_);
418 
419  elemThis = sjhashNext(elemThis);
420  }
421 
422 
423  LeaveCriticalSection(&s_handleCritical);
424  LeaveCriticalSection(&s_forwardCritical);
425 
426  sjhashClear(&oldForwardReceivers);
427 }
