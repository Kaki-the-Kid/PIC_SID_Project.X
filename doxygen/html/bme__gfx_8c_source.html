491  y -= spr_yhotspot;
492

496  if (y + spr_ysize <= gfx_cliptop) return;
497

499  {
500  int dec = *sptr++;
501  if (dec == 255)
502  {

505  }
506  else

509  {
510  sptr += dec;
511  }
512  }
513  }
