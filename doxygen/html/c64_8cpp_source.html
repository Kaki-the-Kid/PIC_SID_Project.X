58 double c64::getCpuFreq(model_t model)
59 {
60  // The crystal clock that drives the VIC II chip is four times
61  // the color burst frequency
62  const double crystalFreq = modelData[model].colorBurst * 4.;
63 
64  // The VIC II produces the two-phase system clock
65  // by running the input clock through a divider
66  return crystalFreq/modelData[model].divider;
67 }

144 bool c64::addExtraSid(c64sid *s, int address)
145 {
146  // Check for valid address in the IO area range ($dxxx)
147  if ((address & 0xf000) != 0xd000)
148  return false;
149 
150  const int idx = (address >> 8) & 0xf;
151 
152  // Only allow second SID chip in SID area ($d400-$d7ff)
153  // or IO Area ($de00-$dfff)
154  if (idx < 0x4 || (idx > 0x7 && idx < 0xe))
155  return false;
156 
157  // Add new SID bank
158  sidBankMap_t::iterator it = extraSidBanks.find(idx);
159  if (it != extraSidBanks.end())
160  {
161  ExtraSidBank *extraSidBank = it->second;
162  extraSidBank->addSID(s, address);
163  }
164  else
165  {
166  ExtraSidBank *extraSidBank = extraSidBanks.insert(it, sidBankMap_t::value_type(idx, new ExtraSidBank()))->second;
167  extraSidBank->resetSIDMapper(ioBank.getBank(idx));
168  ioBank.setBank(idx, extraSidBank);
169  extraSidBank->addSID(s, address);
170  }
171 
172  return true;
173 }
