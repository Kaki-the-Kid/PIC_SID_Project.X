85 unsigned int exSID::delay()
86 {
87  event_clock_t cycles = eventScheduler->getTime(m_accessClk, EVENT_CLOCK_PHI1);
88  m_accessClk += cycles;
89 
90  while (cycles > 0xffff)
91  {
92  exSID_delay(0xffff);
93  cycles -= 0xffff;
94  }
95 
96  return static_cast<unsigned int>(cycles);

107 uint8_t exSID::read(uint_least8_t addr)
108 {
109  if ((addr < 0x19) || (addr > 0x1C))
110  {
111  return busValue;
112  }
113 
114  if (!readflag)
115  {
116 #ifdef DEBUG
117  printf("WARNING: Read support is limited. This file may not play correctly!\n");
118 #endif
119  readflag = true;
120 
121  // Here we implement the "safe" detection routine return values
122  if (0x1b == addr) { // we could implement a commandline-chosen return byte here
123  return (SidConfig::MOS8580 == runmodel) ? 0x02 : 0x03;
124  }
125  }
126 
127  const unsigned int cycles = delay();
128 
129  busValue = exSID_clkdread(cycles, addr); // busValue is updated on valid reads
130  return busValue;
131 }
