72 HardSID::HardSID (sidbuilder *builder) :
73  sidemu(builder),
74  Event("HardSID Delay"),
75  m_handle(0),
76  m_instance(sid++)
77 {
78  unsigned int num = 16;
79  for (unsigned int i = 0; i < 16; i++)
80  {
81  if (m_sidFree[i] == false)
82  {
83  m_sidFree[i] = true;
84  num = i;
85  break;
86  }
87  }
88 
89  // All sids in use?!?
90  if (num == 16)
91  return;
92 
93  m_instance = num;
94 
95  {
96  char device[20];
97  sprintf(device, "/dev/sid%u", m_instance);
98  m_handle = open (device, O_RDWR);
99  if (m_handle < 0)
100  {
101  if (m_instance == 0)
102  {
103  m_handle = open("/dev/sid", O_RDWR);
104  if (m_handle < 0)
105  {
106  m_error.assign("HARDSID ERROR: Cannot access \"/dev/sid\" or \"").append(device).append("\"");
107  return;
108  }
109  }
110  else
111  {
112  m_error.assign("HARDSID ERROR: Cannot access \"").append(device).append("\"");
113  return;
114  }
115  }
116  }
117 
118  m_status = true;
119  sidemu::reset();
120 }

140 event_clock_t HardSID::delay()
141 {
142  event_clock_t cycles = eventScheduler->getTime(m_accessClk, EVENT_CLOCK_PHI1);
143  m_accessClk += cycles;
144 
145  while (cycles > 0xffff)
146  {
147  ioctl(m_handle, HSID_IOCTL_DELAY, 0xffff);
148  cycles -= 0xffff;
149  }
150 
151  return cycles;
152 }

165 uint8_t HardSID::read(uint_least8_t addr)
166 {
167  if (!m_handle)
168  return 0;
169 
170  const event_clock_t cycles = delay();
171 
172  unsigned int packet = ((cycles & 0xffff) << 16) | ((addr & 0x1f) << 8);
173  ioctl(m_handle, HSID_IOCTL_READ, &packet);
174 
175  return static_cast<uint8_t>(packet & 0xff);
176 }

179 {

191 void HardSID::voice(unsigned int num, bool mute)
192 {
193  // Only have 3 voices!
194  if (num >= voices)
195  return;
196  muted[num] = mute;
197 
198  int cmute = 0;
199  for (unsigned int i = 0; i < voices; i++)
200  cmute |= (muted[i] << i);
201  ioctl(m_handle, HSID_IOCTL_MUTE, cmute);
202 }

204 void HardSID::event()
205 {
206  event_clock_t cycles = eventScheduler->getTime(m_accessClk, EVENT_CLOCK_PHI1);
207  if (cycles < HARDSID_DELAY_CYCLES)
208  {
209  eventScheduler->schedule(*this, HARDSID_DELAY_CYCLES - cycles,
210  EVENT_CLOCK_PHI1);
211  }
212  else
213  {
214  m_accessClk += cycles;
215  ioctl(m_handle, HSID_IOCTL_DELAY, static_cast<unsigned int>(cycles));
216  eventScheduler->schedule(*this, HARDSID_DELAY_CYCLES, EVENT_CLOCK_PHI1);
217  }
218 }
