70 {

72  {

740 void MOS6510::sh_instr(uint8_t offset)
741 {
742  const uint8_t tmp = Cycle_Data & (endian_16hi8(Cycle_EffectiveAddress - offset) + 1);
743 
744 #ifdef CORRECT_SH_INSTRUCTIONS
745  /*
746  * When a DMA is going on (the CPU is halted by the VIC-II)
747  * while the instruction sha/shx/shy executes then the last
748  * term of the ANDing (ADH+1) drops off.
749  *
750  * http://sourceforge.net/p/vice-emu/bugs/578/
751  */
752  if (rdyOnThrowAwayRead)
753  {
754  Cycle_Data = tmp;
755  }
756 #endif
757 
758  /*
759  * When the addressing/indexing causes a page boundary crossing
760  * the highbyte of the target address becomes equal to the value stored.
761  */
762  if (adl_carry)
763  endian_16hi8(Cycle_EffectiveAddress, tmp);
764  PutEffAddrDataByte();

811 void MOS6510::doADC()
812 {
813  const unsigned int C = flags.getC() ? 1 : 0;
814  const unsigned int A = Register_Accumulator;
815  const unsigned int s = Cycle_Data;
816  const unsigned int regAC2 = A + s + C;
817 
818  if (flags.getD())
819  { // BCD mode
820  unsigned int lo = (A & 0x0f) + (s & 0x0f) + C;
821  unsigned int hi = (A & 0xf0) + (s & 0xf0);
822  if (lo > 0x09)
823  lo += 0x06;
824  if (lo > 0x0f)
825  hi += 0x10;
826 
827  flags.setZ(!(regAC2 & 0xff));
828  flags.setN(hi & 0x80);
829  flags.setV(((hi ^ A) & 0x80) && !((A ^ s) & 0x80));
830  if (hi > 0x90)
831  hi += 0x60;
832 
833  flags.setC(hi > 0xff);
834  Register_Accumulator = (hi | (lo & 0x0f));
835  }
836  else
837  { // Binary mode
838  flags.setC(regAC2 > 0xff);
839  flags.setV(((regAC2 ^ A) & 0x80) && !((A ^ s) & 0x80));
840  flags.setNZ(Register_Accumulator = regAC2 & 0xff);
841  }
842 }

847 void MOS6510::doSBC()
848 {
849  const unsigned int C = flags.getC() ? 0 : 1;
850  const unsigned int A = Register_Accumulator;
851  const unsigned int s = Cycle_Data;
852  const unsigned int regAC2 = A - s - C;
853 
854  flags.setC(regAC2 < 0x100);
855  flags.setV(((regAC2 ^ A) & 0x80) && ((A ^ s) & 0x80));
856  flags.setNZ(regAC2);
857 
858  if (flags.getD())
859  { // BCD mode
860  unsigned int lo = (A & 0x0f) - (s & 0x0f) - C;
861  unsigned int hi = (A & 0xf0) - (s & 0xf0);
862  if (lo & 0x10)
863  {
864  lo -= 0x06;
865  hi -= 0x10;
866  }
867  if (hi & 0x100)
868  hi -= 0x60;
869  Register_Accumulator = (hi | (lo & 0x0f));
870  }
871  else
872  { // Binary mode
873  Register_Accumulator = regAC2 & 0xff;
874  }

937 void MOS6510::branch_instr(bool condition)
938 {
939  // 2 cycles spent before arriving here. spend 0 - 2 cycles here;
940  // - condition false: Continue immediately to FetchNextInstr.
941  //
942  // Otherwise read the byte following the opcode (which is already scheduled to occur on this cycle).
943  // This effort is wasted. Then calculate address of the branch target. If branch is on same page,
944  // then continue at that insn on next cycle (this delays IRQs by 1 clock for some reason, allegedly).
945  //
946  // If the branch is on different memory page, issue a spurious read with wrong high byte before
947  // continuing at the correct address.
948  if (condition)
949  {
950  // issue the spurious read for next insn here.
951  cpuRead(Register_ProgramCounter);
952 
953  Cycle_EffectiveAddress = endian_16lo8(Register_ProgramCounter);
954  Cycle_EffectiveAddress += Cycle_Data;
955  adl_carry = (Cycle_EffectiveAddress > 0xff) != (Cycle_Data > 0x7f);
956  endian_16hi8(Cycle_EffectiveAddress, endian_16hi8(Register_ProgramCounter));
957 
958  Register_ProgramCounter = Cycle_EffectiveAddress;
959 
960  // Check for page boundary crossing
961  if (!adl_carry)
962  {
963  // Skip next throw away read
964  cycleCount++;
965 
966  // Hack: delay the interrupt past this instruction.
967  if (interruptCycle >> 3 == cycleCount >> 3)
968  interruptCycle += 2;
969  }
970  }
971  else
972  {
973  // branch not taken: skip the following spurious read insn and go to FetchNextInstr immediately.
974  interruptsAndNextOpcode();
975  }
976 }

1469 void MOS6510::buildInstructionTable()
1470 {
1471  for (unsigned int i = 0; i < 0x100; i++)
1472  {
1473 #if DEBUG > 1
1474  printf("Building Command %d[%02x]... ", i, i);
1475 #endif
1476 
1477  /*
1478  * So: what cycles are marked as stealable? Rules are:
1479  *
1480  * - CPU performs either read or write at every cycle. Reads are
1481  * always stealable. Writes are rare.
1482  *
1483  * - Every instruction begins with a sequence of reads. Writes,
1484  * if any, are at the end for most instructions.
1485  */
1486 
1487  unsigned int buildCycle = i << 3;
1488 
1489  typedef enum { WRITE, READ } AccessMode;
1490  AccessMode access = WRITE;
1491  bool legalMode = true;
1492  bool legalInstr = true;
1493 
1494  switch (i)
1495  {
1496  // Accumulator or Implied addressing
1497  case ASLn: case CLCn: case CLDn: case CLIn: case CLVn: case DEXn:
1498  case DEYn: case INXn: case INYn: case LSRn: case NOPn_: case PHAn:
1499  case PHPn: case PLAn: case PLPn: case ROLn: case RORn:
1500  case SECn: case SEDn: case SEIn: case TAXn: case TAYn:
1501  case TSXn: case TXAn: case TXSn: case TYAn:
1502  instrTable[buildCycle++].func = &MOS6510::throwAwayFetch;
1503  break;
1504 
1505  // Immediate and Relative Addressing Mode Handler
1506  case ADCb: case ANDb: case ANCb_: case ANEb: case ASRb: case ARRb:
1507  case BCCr: case BCSr: case BEQr: case BMIr: case BNEr: case BPLr:
1508  case BRKn: case BVCr: case BVSr: case CMPb: case CPXb: case CPYb:
1509  case EORb: case LDAb: case LDXb: case LDYb: case LXAb: case NOPb_:
1510  case ORAb: case SBCb_: case SBXb: case RTIn: case RTSn:
1511  instrTable[buildCycle++].func = &MOS6510::FetchDataByte;
1512  break;
1513 
1514  // Zero Page Addressing Mode Handler - Read & RMW
1515  case ADCz: case ANDz: case BITz: case CMPz: case CPXz: case CPYz:
1516  case EORz: case LAXz: case LDAz: case LDXz: case LDYz: case ORAz:
1517  case NOPz_: case SBCz:
1518  case ASLz: case DCPz: case DECz: case INCz: case ISBz: case LSRz:
1519  case ROLz: case RORz: case SREz: case SLOz: case RLAz: case RRAz:
1520  access = READ;
1521  case SAXz: case STAz: case STXz: case STYz:
1522  instrTable[buildCycle++].func = &MOS6510::FetchLowAddr;
1523  break;
1524 
1525  // Zero Page with X Offset Addressing Mode Handler
1526  // these issue extra reads on the 0 page, but we don't care about it
1527  // because there are no detectable effects from them. These reads
1528  // occur during the "wasted" cycle.
1529  case ADCzx: case ANDzx: case CMPzx: case EORzx: case LDAzx: case LDYzx:
1530  case NOPzx_: case ORAzx: case SBCzx:
1531  case ASLzx: case DCPzx: case DECzx: case INCzx: case ISBzx: case LSRzx:
1532  case RLAzx: case ROLzx: case RORzx: case RRAzx: case SLOzx: case SREzx:
1533  access = READ;
1534  case STAzx: case STYzx:
1535  instrTable[buildCycle++].func = &MOS6510::FetchLowAddrX;
1536  // operates on 0 page in read mode. Truly side-effect free.
1537  instrTable[buildCycle++].func = &MOS6510::WasteCycle;
1538  break;
1539 
1540  // Zero Page with Y Offset Addressing Mode Handler
1541  case LDXzy: case LAXzy:
1542  access = READ;
1543  case STXzy: case SAXzy:
1544  instrTable[buildCycle++].func = &MOS6510::FetchLowAddrY;
1545  // operates on 0 page in read mode. Truly side-effect free.
1546  instrTable[buildCycle++].func = &MOS6510::WasteCycle;
1547  break;
1548 
1549  // Absolute Addressing Mode Handler
1550  case ADCa: case ANDa: case BITa: case CMPa: case CPXa: case CPYa:
1551  case EORa: case LAXa: case LDAa: case LDXa: case LDYa: case NOPa:
1552  case ORAa: case SBCa:
1553  case ASLa: case DCPa: case DECa: case INCa: case ISBa: case LSRa:
1554  case ROLa: case RORa: case SLOa: case SREa: case RLAa: case RRAa:
1555  access = READ;
1556  case JMPw: case SAXa: case STAa: case STXa: case STYa:
1557  instrTable[buildCycle++].func = &MOS6510::FetchLowAddr;
1558  instrTable[buildCycle++].func = &MOS6510::FetchHighAddr;
1559  break;
1560 
1561  case JSRw:
1562  instrTable[buildCycle++].func = &MOS6510::FetchLowAddr;
1563  break;
1564 
1565  // Absolute With X Offset Addressing Mode Handler (Read)
1566  case ADCax: case ANDax: case CMPax: case EORax: case LDAax:
1567  case LDYax: case NOPax_: case ORAax: case SBCax:
1568  access = READ;
1569  instrTable[buildCycle++].func = &MOS6510::FetchLowAddr;
1570  instrTable[buildCycle++].func = &MOS6510::FetchHighAddrX2;
1571  // this cycle is skipped if the address is already correct.
1572  // otherwise, it will be read and ignored.
1573  instrTable[buildCycle++].func = &MOS6510::throwAwayRead;
1574  break;
1575 
1576  // Absolute X (RMW; no page crossing handled, always reads before writing)
1577  case ASLax: case DCPax: case DECax: case INCax: case ISBax:
1578  case LSRax: case RLAax: case ROLax: case RORax: case RRAax:
1579  case SLOax: case SREax:
1580  access = READ;
1581  case SHYax: case STAax:
1582  instrTable[buildCycle++].func = &MOS6510::FetchLowAddr;
1583  instrTable[buildCycle++].func = &MOS6510::FetchHighAddrX;
1584  instrTable[buildCycle++].func = &MOS6510::throwAwayRead;
1585  break;
1586 
1587  // Absolute With Y Offset Addresing Mode Handler (Read)
1588  case ADCay: case ANDay: case CMPay: case EORay: case LASay:
1589  case LAXay: case LDAay: case LDXay: case ORAay: case SBCay:
1590  access = READ;
1591  instrTable[buildCycle++].func = &MOS6510::FetchLowAddr;
1592  instrTable[buildCycle++].func = &MOS6510::FetchHighAddrY2;
1593  instrTable[buildCycle++].func = &MOS6510::throwAwayRead;
1594  break;
1595 
1596  // Absolute Y (No page crossing handled)
1597  case DCPay: case ISBay: case RLAay: case RRAay: case SLOay:
1598  case SREay:
1599  access = READ;
1600  case SHAay: case SHSay: case SHXay: case STAay:
1601  instrTable[buildCycle++].func = &MOS6510::FetchLowAddr;
1602  instrTable[buildCycle++].func = &MOS6510::FetchHighAddrY;
1603  instrTable[buildCycle++].func = &MOS6510::throwAwayRead;
1604  break;
1605 
1606  // Absolute Indirect Addressing Mode Handler
1607  case JMPi:
1608  instrTable[buildCycle++].func = &MOS6510::FetchLowPointer;
1609  instrTable[buildCycle++].func = &MOS6510::FetchHighPointer;
1610  instrTable[buildCycle++].func = &MOS6510::FetchLowEffAddr;
1611  instrTable[buildCycle++].func = &MOS6510::FetchHighEffAddr;
1612  break;
1613 
1614  // Indexed with X Preinc Addressing Mode Handler
1615  case ADCix: case ANDix: case CMPix: case EORix: case LAXix: case LDAix:
1616  case ORAix: case SBCix:
1617  case DCPix: case ISBix: case SLOix: case SREix: case RLAix: case RRAix:
1618  access = READ;
1619  case SAXix: case STAix:
1620  instrTable[buildCycle++].func = &MOS6510::FetchLowPointer;
1621  instrTable[buildCycle++].func = &MOS6510::FetchLowPointerX;
1622  instrTable[buildCycle++].func = &MOS6510::FetchLowEffAddr;
1623  instrTable[buildCycle++].func = &MOS6510::FetchHighEffAddr;
1624  break;
1625 
1626  // Indexed with Y Postinc Addressing Mode Handler (Read)
1627  case ADCiy: case ANDiy: case CMPiy: case EORiy: case LAXiy:
1628  case LDAiy: case ORAiy: case SBCiy:
1629  access = READ;
1630  instrTable[buildCycle++].func = &MOS6510::FetchLowPointer;
1631  instrTable[buildCycle++].func = &MOS6510::FetchLowEffAddr;
1632  instrTable[buildCycle++].func = &MOS6510::FetchHighEffAddrY2;
1633  instrTable[buildCycle++].func = &MOS6510::throwAwayRead;
1634  break;
1635 
1636  // Indexed Y (No page crossing handled)
1637  case DCPiy: case ISBiy: case RLAiy: case RRAiy: case SLOiy:
1638  case SREiy:
1639  access = READ;
1640  case SHAiy: case STAiy:
1641  instrTable[buildCycle++].func = &MOS6510::FetchLowPointer;
1642  instrTable[buildCycle++].func = &MOS6510::FetchLowEffAddr;
1643  instrTable[buildCycle++].func = &MOS6510::FetchHighEffAddrY;
1644  instrTable[buildCycle++].func = &MOS6510::throwAwayRead;
1645  break;
1646 
1647  default:
1648  legalMode = false;
1649  break;
1650  }
1651 
1652  if (access == READ)
1653  {
1654  instrTable[buildCycle++].func = &MOS6510::FetchEffAddrDataByte;
1655  }
1656 
1657  //---------------------------------------------------------------------------------------
1658  // Addressing Modes Finished, other cycles are instruction dependent
1659  //---------------------------------------------------------------------------------------
1660  switch(i)
1661  {
1662  case ADCz: case ADCzx: case ADCa: case ADCax: case ADCay: case ADCix:
1663  case ADCiy: case ADCb:
1664  instrTable[buildCycle++].func = &MOS6510::adc_instr;
1665  break;
1666 
1667  case ANCb_:
1668  instrTable[buildCycle++].func = &MOS6510::anc_instr;
1669  break;
1670 
1671  case ANDz: case ANDzx: case ANDa: case ANDax: case ANDay: case ANDix:
1672  case ANDiy: case ANDb:
1673  instrTable[buildCycle++].func = &MOS6510::and_instr;
1674  break;
1675 
1676  case ANEb: // Also known as XAA
1677  instrTable[buildCycle++].func = &MOS6510::ane_instr;
1678  break;
1679 
1680  case ARRb:
1681  instrTable[buildCycle++].func = &MOS6510::arr_instr;
1682  break;
1683 
1684  case ASLn:
1685  instrTable[buildCycle++].func = &MOS6510::asla_instr;
1686  break;
1687 
1688  case ASLz: case ASLzx: case ASLa: case ASLax:
1689  instrTable[buildCycle].nosteal = true;
1690  instrTable[buildCycle++].func = &MOS6510::asl_instr;
1691  instrTable[buildCycle].nosteal = true;
1692  instrTable[buildCycle++].func = &MOS6510::PutEffAddrDataByte;
1693  break;
1694 
1695  case ASRb: // Also known as ALR
1696  instrTable[buildCycle++].func = &MOS6510::alr_instr;
1697  break;
1698 
1699  case BCCr:
1700  instrTable[buildCycle++].func = &MOS6510::bcc_instr;
1701  instrTable[buildCycle++].func = &MOS6510::fix_branch;
1702  break;
1703 
1704  case BCSr:
1705  instrTable[buildCycle++].func = &MOS6510::bcs_instr;
1706  instrTable[buildCycle++].func = &MOS6510::fix_branch;
1707  break;
