219  inline void checkVblank()
220  {

222  if (rasterY == (maxRasters - 1))
223  {
224  vblanking = true;
225  }

227  // Check DEN bit on first cycle of the line following the first DMA line
228  if (rasterY == FIRST_DMA_LINE
229  && !areBadLinesEnabled
