385 {

393  if (rate > 0x3c)
394  {
395  rate = 0x3c;
396  }
397  return rate;

480 static void OPL3_EnvelopeCalc(opl3_slot *slot)
481 {
482  Bit8u rate_h, rate_l;
483  Bit8u inc = 0;
484  rate_h = slot->eg_rate >> 2;
485  rate_l = slot->eg_rate & 3;
486  if (eg_incsh[rate_h] > 0)
487  {
488  if ((slot->chip->timer & ((1 << eg_incsh[rate_h]) - 1)) == 0)
489  {
490  inc = eg_incstep[eg_incdesc[rate_h]][rate_l]
491  [((slot->chip->timer) >> eg_incsh[rate_h]) & 0x07];
492  }
493  }
494  else
495  {
496  inc = eg_incstep[eg_incdesc[rate_h]][rate_l]
497  [slot->chip->timer & 0x07] << (-eg_incsh[rate_h]);
498  }
499  slot->eg_inc = inc;
500  slot->eg_out = slot->eg_rout + (slot->reg_tl << 2)
501  + (slot->eg_ksl >> kslshift[slot->reg_ksl]) + *slot->trem;
502  envelope_gen[slot->eg_gen](slot);
503 }
