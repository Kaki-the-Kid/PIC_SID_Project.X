115 void ReSIDfp::sampling(float systemclock, float freq,
116  SidConfig::sampling_method_t method, bool)
117 {
118  reSIDfp::SamplingMethod sampleMethod;
119  switch (method)
120  {
121  case SidConfig::INTERPOLATE:
122  sampleMethod = reSIDfp::DECIMATE;
123  break;
124  case SidConfig::RESAMPLE_INTERPOLATE:
125  sampleMethod = reSIDfp::RESAMPLE;
126  break;
127  default:
128  m_status = false;
129  m_error = ERR_INVALID_SAMPLING;
130  return;
131  }
132 
133  try
134  {
135  // Round half frequency to the nearest multiple of 5000
136  const int halfFreq = 5000*((static_cast<int>(freq)+5000)/10000);
137  m_sid.setSamplingParameters(systemclock, sampleMethod, freq, std::min(halfFreq, 20000));
138  }
139  catch (reSIDfp::SIDError const &)
140  {
141  m_status = false;
142  m_error = ERR_UNSUPPORTED_FREQ;
143  return;
144  }
145 
146  m_status = true;
147 }

151 {

157  break;

160  if (digiboost)

162  break;
163  default:
164  m_status = false;
165  m_error = ERR_INVALID_CHIP;
166  return;
167  }
168 
169  m_sid.setChipModel(chipModel);
170  m_status = true;
171 }
