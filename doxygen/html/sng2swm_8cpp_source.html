80 //slide (FREQMODL,FREQMODH) calculation in SID-Wizard 1.2 (if non-zero):
81 // ExpTabIndex=(Input/2)+DPITCH; //(pitch-dependent)
82 // if (ExpPabIndex<ExpTreshold) { FREQMODL=SWExpTabL[ExpTabIndex]; FREQMODH=0}
83 // else { FREQMODL=SWExpTabL[ExpTabIndex-ExpTreshold]; FREQMODH=SWExpTabH[ExpTabIndex-ExpTreshold]
84 // this slide-value converter function has to do the inverse of this operation:
85 int GTsldToSWsld(int GTpos, unsigned char Dpitch, sng_table * tbl)
86 {
87  int i, PitchTemp, SWslide, GTslide = (tbl->left[GTpos-1])*256 + tbl->right[GTpos-1];
88  if (GTpos==0 || GTslide==0) return 0x00;
89  else if (GTslide<0x100) for (i=EXPTRESHOLD-1; i>=0; i--) //if small pitch slide (fits in low byte of pitch)
90  {
91  if (SWexpTabH[i] <= GTslide) { SWslide=(i-Dpitch)*2; break; } //if found the matching index, calculate slide-value and exit the loop
92  }
93  else
94  {
95  for (i=FREQTB_SIZE-1; i>=0; i--) //if big pitch-slide using both low&high bytes of SID-pitch
96  {
97  PitchTemp=(SWexpTabH[i+FREQTBH_POS]*0x100+SWexpTabL[i]);
98  if ( PitchTemp <= GTslide ) { SWslide=((i+EXPTRESHOLD)-Dpitch)*2; break; } //if found the matching index, calculate slide-value and exit the loop
99  else SWslide=(EXPTRESHOLD-Dpitch)*2; //smallest value for big pitch-slide
100  }
101  }
102  //printf("%x-%X-%X-%x ",i,PitchTemp,GTslide, SWslide);
103  if (SWslide < 0 ) SWslide=0; //take care of minimal limit which is 0 of course
104  if (SWslide > 0xFF) SWslide=0xFF; //take care of maximal limit if dpitch causes overload
105  return SWslide;
106 }

108 //vibrato-amplitude (FREQMODL,FREQMODH) calculation in SID-Wizard 1.2 (if non-zero): the same as slide but Input value is halved
109 int GTvibToSWvib(int GTpos, unsigned char Dpitch, sng_table * tbl)
110 {
111  int i,SWamp, GTfreq=tbl->left[GTpos-1], GTamp = tbl->right[GTpos-1], SWvibr;
112  if (GTpos==0 || GTamp==0) return 0x00;
113  if (GTfreq>0xF) GTfreq=0xF; //take care of limit in SID-Wizard
114  //GT has calculated vibrato if GTamp is bigger than $7F (bit7 set) - should be converted differently (omitting 'Dpitch' from the equation)
115  for (i=EXPTRESHOLD-1; i>=0; i--) //convert GT-amplitude to SW-amplitude (i.e. slide-speed)
116  {
117  if (SWexpTabH[i] <= GTamp) { SWamp=(i-Dpitch)*4; break; } //if found the matching index, calculate slide-value and exit the loop
118  }
119  //printf("%x-%X-%X ",i,GTamp, SWamp);
120  if (SWamp < 0x10 ) SWamp=0x10; //take care of minimal limit
121  if (SWamp > 0xFF) SWamp=0xFF; //take care of maximal limit if dpitch causes overload
122  SWvibr = (SWamp&0xF0) + SWvibFreq[GTfreq];
123  return SWvibr;
124 }

127 unsigned char GTtblToSWtbl(sng_table * gt2_table, unsigned char GTstartIndex, swm_instrument * sw_instr, char TableType, bool debug)
128 { //TableType: 0:Waveform-table, 1:Pulse-table, 2:Filter-table
129  unsigned int SWtableStart, SWtableIndex;
130  unsigned char GTvalLeft, GTvalRight, GTvalThird, prevGTvalLeft=0, PrevReso=0xF,PrevBand=0x9,PrevCutoff=0xFF, SkippedRowCount=0;
131  int GTtableIndex;
132 
133  if (!(TableType&0x80))
134  {
135  SWtableStart=SWtableIndex = sw_instr->tables.size(); //init target (SW) index
136  }
137  else
138  {
139  switch (TableType&0x7F)
140  {
141  case 0: SWtableStart=SWtableIndex = 0;
142  break;
143  case 1: SWtableStart=SWtableIndex = 0;
144  break;
145  case 2: SWtableStart=SWtableIndex = 0;
146  break;
147  }
148  return 0;
149  }
150 
151 
152 if ( (GTtableIndex=GTstartIndex) ) //convert GT-table
153  {
154  SkippedRowCount=0; //when more table-rows brought into one (for filter)
155  for ( ;GTtableIndex<=0xFF; GTtableIndex++)
156  {
157  GTvalLeft=gt2_table->left[GTtableIndex-1]; GTvalRight=gt2_table->right[GTtableIndex-1]; GTvalThird=0;
158  if (debug) printf("[%.2X: %.2X %.2X]", GTtableIndex, GTvalLeft, GTvalRight);
159  if (!(TableType&0x7F) && GTvalLeft==0) GTvalLeft=prevGTvalLeft; //avoid inaudible waveform (when $FE loops)
160  if (GTvalLeft==GT2_TABLE_JUMP)
161  {
162  if (GTvalRight==0) {if (debug) printf(" -> [(%.2X)%.2X: %.2X\n",SWtableIndex, SW1_INSTRUMENT_PARAMSIZE+SWtableIndex, SW1_TABLE_END); break;} //$FF 00 ending
163  else if (!(TableType&0x7F) && GTvalRight==GTtableIndex) { GTvalLeft=SW1_TABLE_JUMP; GTvalRight=SW1_INSTRUMENT_PARAMSIZE+0xC0; } //self-jumping in Wave-table needs special treatment
164  else if (GTvalRight < GTstartIndex || GTvalRight > GTtableIndex) // - or merge if jumped outside the already converted table
165  { GTtableIndex=GTstartIndex=GTvalRight-1; SWtableStart=SWtableIndex-3; SkippedRowCount=0; if(debug)printf(" -> merging table-jump!\n"); continue; }
166  else {GTvalLeft=SW1_TABLE_JUMP; GTvalRight=SW1_INSTRUMENT_PARAMSIZE+SWtableStart + 3*((GTvalRight-GTstartIndex)-SkippedRowCount); } //$FF xx loop/jump ending -> convert to SW: $FE xx
167  }
168  else if (!(TableType&0x7F) && GTvalLeft>=GT2_WFTABLE_FX) GTvalLeft=GTvalRight=0; //no Wavetable-FX support in SW
169  else if (!(TableType&0x7F)) //in case of normal waveform or speed-setting handle arp/pitch values
170  {
171  if(GT2_ARP_REL_MIN<=GTvalRight && GTvalRight<GT2_ARP_NOP) GTvalRight += (SW1_ARP_REL_MIN-GT2_ARP_REL_MIN); //convert negative relative pitches
172  else if (GT2_ARP_ABS_MIN<=GTvalRight && GTvalRight<GT2_ARP_ABS_MAX) GTvalRight+=1; //convert absolute-pitches (starts from C#0 in GT, and from C-0 in SW)
173  }
174  else if ((TableType&0x7F)==2) //filtertable?
175  {
176  if (GTvalLeft==0) GTvalLeft=PrevBand*0x10 + PrevReso; //convert GT cutoff-setting to SW band+reso+cutoff setting
177  else if (GTvalLeft&0x80) //band/reso/filterswitch setting in GT?
178  {
179  PrevReso=GTvalRight/0x10; PrevBand=(GTvalLeft&0xF0)/0x10; GTvalLeft = PrevBand*0x10 + PrevReso; GTvalThird=(GTvalRight&0xF)+0x80;
180  if (gt2_table->left[(GTtableIndex-1)+1]==0) { GTvalRight=gt2_table->right[(GTtableIndex-1)+1]; GTtableIndex++; SkippedRowCount++; } //cutoff-setting in next row? set and skip next row
181  else GTvalRight=PrevCutoff; //if no cutoff-setting follows, use previous cutoff
182  }
183  else
184  { //sweep is only in 8 bit resolution, convert it to SW which sweeps in 11bit resolution
185  if (GTvalRight<=0x7F) GTvalRight = (GTvalRight*8<=0x7F) ? GTvalRight*8 : 0x7F; //take care of maximal value
186  else if (GTvalRight>=0x80) GTvalRight = ((0x100-GTvalRight)*8<0x80) ? 0x100-(0x100-GTvalRight)*8 :0x80; //take care of minimal value (can be overridden in GT's non-11bit-fine slides)
187  }
188  }
189  if (!(TableType&0x80))
190  {
191  sw_instr->tables.push_back(GTvalLeft);
192  sw_instr->tables.push_back(GTvalRight);
193  sw_instr->tables.push_back(GTvalThird);
194  }
195  else
196  {
197  //sw_instr->tables.insert(SWtableIndex, GTvalThird);
198  //sw_instr->tables.insert(SWtableIndex, GTvalRight);
199  //sw_instr->tables.insert(SWtableIndex, GTvalLeft);
200  }
201  if (debug) printf(" -> [(%.2X)%.2X: %.2X %.2X %.2X]\n", SWtableIndex, SW1_INSTRUMENT_PARAMSIZE+SWtableIndex, sw_instr->tables[SWtableIndex], sw_instr->tables[SWtableIndex+1], sw_instr->tables[SWtableIndex+2]);
202  SWtableIndex+=3; prevGTvalLeft=GTvalLeft;
203  if (gt2_table->left[GTtableIndex-1]==GT2_TABLE_JUMP) break; //$FF xx loop/jump ending
204  }
205  }
206  sw_instr->tables.push_back(SW1_TABLE_END);SWtableIndex++;
207  return SWtableIndex;
208 }

211 void GTseqToSWseq (sng_orderlist * gt_seq , swm_sequence * sw_seq, bool debug)
212 {
213  int j,k,size=0;
214  if(debug) printf("\n\n*GT : ");
215  for (j=0; j<gt_seq->length; j++)
216  {
217  if(debug) printf("%X ",gt_seq->data[j]);
218  if (gt_seq->data[j]==SW1_SEQUENCE_LOOPSONG) sw_seq->data.push_back(gt_seq->data[j]);
219  else if ( gt_seq->data[j] >= GT2_ORDERLIST_TRANS_MIN ) sw_seq->data.push_back(gt_seq->data[j]-(GT2_ORDERLIST_TRANS-SW1_SEQUENCE_TRANS));
220  else if (gt_seq->data[j] >= GT2_ORDERLIST_REPEAT)
221  {
222  for (k=0;k<(gt_seq->data[j]&0xF);k++) sw_seq->data.push_back(gt_seq->data[j+1]+1); //SW pattern-numbers count from 1
223  size += k-1;
224  }
225  else sw_seq->data.push_back(gt_seq->data[j]+1); //SW pattern-numbers count from 1
226  size++;
227  }
228  sw_seq->data.push_back(gt_seq->restart); size++;
229  sw_seq->size = size;
230  if (debug) { printf("\n*SW : "); for(j=0;j<sw_seq->size;j++) printf("%X ",sw_seq->data.at(j)); }
231 }

234 void GTptnToSWptn (sng_pattern * gt_ptn , swm_pattern * sw_ptn , sng_table * spdtbl , bool debug)
235 {
236  unsigned char j,GTnote,SWnote,GTins,SWins,GTfx,SWfx,GTval,SWval,SWindex=0;
237  unsigned char Dpitch=(char)(SW1_NOTE_MAX/2); //buffer for valid notes (discrete pitch) - init to the middle of the note-region
238  unsigned char PrevSWins=0, PrevSWfx=0, PrevSWval=0; //used to eliminate repetitive GT effects
239  for (j=0; j<gt_ptn->rows-1; j++)
240  {
241  GTnote=gt_ptn->data[j].note;
242  if (GTnote==GT2_PATTERN_REST) SWnote=SW1_PTNCOL1_REST;
243  else if (GTnote==GT2_PATTERN_KEYOFF) SWnote=SW1_PTNCOL1_KEYOFF;
244  else if (GTnote==GT2_PATTERN_KEYON) SWnote=SW1_PTNCOL1_KEYON;
245  else if (GTnote>=GT2_PATTERN_FIRSTPACKEDREST) SWnote=SW1_PTNCOL1_REST;
246  else if (GTnote>=GT2_PATTERN_NOTE) //register valid note
247  { SWnote=Dpitch=GTnote-(GT2_PATTERN_NOTE-SW1_PTNCOL1_NOTE); PrevSWfx=PrevSWval=0; }
248  else SWnote=SW1_PTNCOL1_REST;
249 
250  SWins=SWfx=SWval=0; //init
251 
252  if (SWindex < SW1_PATTERNSIZE_MAX)
253  {
254  GTins=gt_ptn->data[j].instrument;
255  if (SW1_INSTRUMENT_MIN<=GTins && GTins<=SW1_INSTRUMENT_MAX && PrevSWins!=GTins)
256  { SWins=PrevSWins=GTins; SWnote|=0x80; } //sign in note-column
257  else SWins=0;
258 
259  GTfx=gt_ptn->data[j].command; GTval=gt_ptn->data[j].parameter;
260 
261  switch (GTfx)
262  { //continuous GT commands - convert them to one-shot form of SW
263  case GT2_COMMAND_NOP: //00?
264  if (GTnote!=GT2_PATTERN_END && (PrevSWfx==SW1_BIGFX_PORTUP || PrevSWfx==SW1_BIGFX_PORTDOWN || PrevSWfx==SW1_BIGFX_VIBRATO) )
265  { SWfx=PrevSWfx; SWval=0x00; } break; //stop slideup/slide-down/vibrato if ends in GT (portamento stops automatically)
266  case GT2_COMMAND_PORTUP:
267  if (GTval) { SWfx=SW1_BIGFX_PORTUP; SWval=GTsldToSWsld(GTval,Dpitch,spdtbl); } break;
268  case GT2_COMMAND_PORTDOWN:
269  if (GTval) { SWfx=SW1_BIGFX_PORTDOWN; SWval=GTsldToSWsld(GTval,Dpitch,spdtbl); } break;
270  case GT2_COMMAND_TONEPORT:
271  if (GTval) { SWfx=SW1_BIGFX_TONEPORT; SWval=GTsldToSWsld(GTval,Dpitch,spdtbl); }
272  else { SWfx=0; SWins=SW1_LEGATO; SWnote|=0x80; } //change fx-legato to instrument-column legato
273  break;
274  case GT2_COMMAND_VIBRATO: SWfx=SW1_BIGFX_VIBRATO; SWval=GTvibToSWvib(GTval,Dpitch,spdtbl); break;
275  default: SWfx=SWval=0; break;
276  }
277  if (SWfx==PrevSWfx && SWval==PrevSWval) { SWfx=SWval=0; } //eliminate repetitive GT effects
278  else { PrevSWfx=SWfx; PrevSWval=SWval; }
279 
280  switch (GTfx)
281  { //one-shot GT commands - table-jump commands should jump inside corresponding instrument-tables in SW
282  case GT2_COMMAND_AD: SWfx=SW1_BIGFX_AD; SWval=GTval; break;
283  case GT2_COMMAND_SR: SWfx=SW1_BIGFX_SR; SWval=GTval; break;
284  case GT2_COMMAND_WAVEREG: SWfx=SW1_BIGFX_WAVEREG; SWval=GTval; break;
285 
286  case GT2_COMMAND_WAVETBL:
287  SWfx=SW1_BIGFX_WAVETBL; //this FX was added in SW 1.4
288  printf("\nInserting more into waveform-arp-table of SW instrument $%.2X:\n",PrevSWins);
289  SWval = GTtblToSWtbl(&gt2_wavetable, GTval, &sw_instruments[PrevSWins], 0x80, debug); //TableType bit7 set means 'insert into the table'
290  break;
291  case GT2_COMMAND_PULSETBL:
292  SWfx=SW1_BIGFX_PULSETBL;
293  printf("\nInserting more into pulse-table of SW instrument $%.2X:\n",PrevSWins);
294  SWval = GTtblToSWtbl(&gt2_pulsetable, GTval, &sw_instruments[PrevSWins], 0x81, debug); //TableType bit7 set means 'insert into the table'
295  break;
296  case GT2_COMMAND_FILTERTBL:
297  SWfx=SW1_BIGFX_FILTERTBL;
298  printf("\nInserting more into filter-table of SW instrument $%.2X:\n",PrevSWins);
299  SWval = GTtblToSWtbl(&gt2_filtertable, GTval, &sw_instruments[PrevSWins], 0x82, debug); //TableType bit7 set means 'insert into the table'
300  break;
301 
302  case GT2_COMMAND_FILTERCTRL:
303  SWfx=SW1_BIGFX_FILT_CONTROL; //this FX was added in SW 1.4
304  break;
305  case GT2_COMMAND_FILTERCUT:
306  SWfx=SW1_BIGFX_FILT_CUTOFF; SWval=GTval; //this FX was added in SW 1.4
307  break;
308  case GT2_COMMAND_VOLUME:
309  SWfx=SW1_SMALLFX_MAINVOL|(GTval&0xF);
310  break;
311  case GT2_COMMAND_FUNK:
312  SWfx=SW1_BIGFX_FUNKTEMPO;
313  SWval= (spdtbl->left[GTval-1]&0xF)*16 + (spdtbl->right[GTval-1]&0xF) ;
314  break;
315  case GT2_COMMAND_TEMPO:
316  if (!(GTval&0x80)) {SWfx=SW1_BIGFX_TEMPO; SWval=GTval;} //tune-tempo setting
317  else {SWfx=SW1_BIGFX_TRACKTEMPO; SWval=GTval&0x7F;} //track-tempo setting
318  break;
319  };
320 
321  if (SWfx) { SWins|=0x80; SWnote|=0x80; }//sign FX column
322  }
323 
324  if(debug) printf("%2X %2X %2X %2X , ",SWnote,SWins,SWfx,SWval);
325 
326  //push converted pattern-row to target
327  sw_ptn->data.push_back(SWnote);SWindex++;
328  if (SWnote&0x80) {sw_ptn->data.push_back(SWins);SWindex++;}
329  if (SWins&0x80) {sw_ptn->data.push_back(SWfx);SWindex++;}
330  if (0<SWfx && SWfx<SW1_SMALLFX_MIN) {sw_ptn->data.push_back(SWval);SWindex++;}
331  }
332  sw_ptn->data.push_back(SW1_PTNCOL1_END); //the closing $FF for SW pattern
333  sw_ptn->length = gt_ptn->rows-1;
334  sw_ptn->size=SWindex+1;
335 }
